56  public ArgumentParseResults parse(IRegisteredCommand command, String[] args)
57  throws CommandException {
58 
59  PreCon.notNull(command);
60  PreCon.notNull(args);
61 
62  ArgumentParseResults results = new ArgumentParseResults(command);
63 
64  Deque<String> arguments = ArrayUtils.toDeque(args);
65  Deque<ICommandParameter> staticParameters = new ArrayDeque<>(
66  command.getInfo().getStaticParams());
67 
68  // parse arguments for static parameters.
69  parseStaticArgs(command, results, staticParameters, arguments);
70 
71  // check if there are floating parameters for the command.
72  if (command.getInfo().getRawFloatingParams().length == 0 &&
73  command.getInfo().getRawFlagParams().length == 0) {
74 
75  // if there are no more parameters but there are still
76  // arguments left, then there are too many arguments.
77  if (!arguments.isEmpty())
78  throw CommandException.tooManyArgs(command);
79 
80 
81  // nothing left to do
82  return results;
83  }
84 
85  // get ready to parse floating and flag parameters.
86  RetrievableSet<ICommandParameter> floating = new RetrievableSet<>(command.getInfo().getFloatingParams());
87  RetrievableSet<IFlagParameter> flags = new RetrievableSet<>(command.getInfo().getFlagParams());
88 
89  // parse arguments for non static parameters.
90  parseNonStaticArgs(command, results, floating, flags, arguments);
91 
92  return results;
93 
94  }

96  // parse arguments for static parameters
97  private void parseStaticArgs(IRegisteredCommand command,
98  ArgumentParseResults results,
99  Deque<ICommandParameter> staticParameters,
100  Deque<String> arguments)
101  throws CommandException {
102 
103  while (!staticParameters.isEmpty()) {
104 
105  ICommandParameter parameter = staticParameters.removeFirst();
106 
107  String name = parameter.getName();
108  String value = null;
109 
110  if (!arguments.isEmpty()) {
111 
112  String arg = arguments.removeFirst();
113 
114  // Should not be any floating or flag parameters before required arguments
115  if (arg.startsWith(COMMON_PREFIX)) {
116 
117  // the last static parameter might be optional, in which
118  // case it should have a default value.
119 
120  // If there are still more static parameters, it means this
121  // is not the last parameter and the value is incorrect.
122  if (!staticParameters.isEmpty()) {
123  throw CommandException.invalidArgument(command, parameter.getName());
124  }
125 
126  // No default value defined means a discreet value is expected.
127  else if (!parameter.hasDefaultValue()) {
128  throw CommandException.missingRequiredArg(command, parameter);
129  }
130 
131  // re-insert floating argument so the other parsers
132  // will see it. Since this is the last static parameter,
133  // this is the end of the loop.
134  else {
135  arguments.addFirst(arg);
136  }
137  }
138  else {
139  // get parameter value
140  value = parseArgValue(arg, arguments);
141  }
142  }
143 
144  // add argument
145  if (value != null || parameter.hasDefaultValue()) {
146  Argument argument = new Argument(parameter, value);
147 
148  if (results.getArgMap().containsKey(name))
149  throw CommandException.duplicateArg(command, parameter);
150 
151 
152  results.getStaticArgs().add(argument);
153  results.getArgMap().put(name, argument);
154  }
155  else {
156  throw CommandException.missingRequiredArg(command, parameter);
157  }
158  }
159  }

161  // parse arguments for floating parameters and flags
162  private void parseNonStaticArgs(IRegisteredCommand command,
163  ArgumentParseResults results,
164  RetrievableSet<ICommandParameter> parameters,
165  RetrievableSet<IFlagParameter> flags,
166  Deque<String> arguments)
167  throws CommandException {
168 
169  while (!arguments.isEmpty()) {
170 
171  String paramName = arguments.removeFirst();
172 
173  if (!paramName.startsWith(COMMON_PREFIX))
174  throw CommandException.invalidArgument(command, paramName);
175 
176  // check for flag
177  if (paramName.startsWith(FLAG_PREFIX)) {
178 
179  paramName = paramName.substring(FLAG_PREFIX.length());
180 
181  IFlagParameter parameter = flags.removeRetrieve(new Flag(paramName, -1));
182  if (parameter == null)
183  throw CommandException.invalidFlag(command, paramName);
184 
185 
186  results.setFlag(paramName, true);
187  }
188  // check if floating parameter
189  else if (paramName.startsWith(FLOATING_PREFIX)) {
190 
191  paramName = paramName.substring(FLOATING_PREFIX.length());
192 
193  ICommandParameter parameter = parameters.removeRetrieve(new Parameter(paramName, null));
194  if (parameter == null)
195  throw CommandException.invalidParam(command, paramName);
196 
197  if (arguments.isEmpty())
198  throw CommandException.missingRequiredArg(command, parameter);
199 
200  String arg = parseArgValue(arguments.removeFirst(), arguments);
201  Argument commandArgument = new Argument(parameter, arg);
202 
203  if (results.getArgMap().containsKey(paramName))
204  throw CommandException.duplicateArg(command, parameter);
205 
206 
207  results.getFloatingArgs().add(commandArgument);
208  results.getArgMap().put(paramName, commandArgument);
209  }
210  }
211 
212  if (!parameters.isEmpty()) {
213 
214  for (ICommandParameter param : parameters) {
215  if (param.hasDefaultValue()) {
216  Argument commandArgument = new Argument(param, null);
217  results.getFloatingArgs().add(commandArgument);
218  results.getArgMap().put(param.getName(), commandArgument);
219  } else {
220  throw CommandException.missingRequiredArg(command, param);
221  }
222  }
223  }
224  }

232  private String parseArgValue(String currentArg, Deque<String> argsQueue) {
233 
234  // check to see if parsing a literal
235  String quote = null;
236 
237  // detect double quote
238  if (currentArg.startsWith("\"")) {
239  quote = "\"";
240  }
241  // detect single quote
242  else if (currentArg.startsWith("'")) {
243  quote = "'";
244  }
245 
246  // check for quoted literal
247  if (quote != null) {
248 
249 
250  String firstWord = currentArg.substring(1); // remove quotation
251 
252  // make sure the literal isn't closed on the same word
253  if (firstWord.endsWith(quote)) {
254 
255  // remove end quote
256  return firstWord.substring(0, firstWord.length() - 1);
257  }
258  // otherwise parse ahead until end of literal
259  else {
260 
261  StringBuilder literal = new StringBuilder(currentArg.length() * argsQueue.size());
262  literal.append(firstWord);
263 
264  while (!argsQueue.isEmpty()) {
265  String nextArg = argsQueue.removeFirst();
266 
267  // check if this is the final word in the literal
268  if (nextArg.endsWith(quote)) {
269 
270  //remove end quote
271  nextArg = nextArg.substring(0, nextArg.length() - 1);
272 
273  literal.append(' ');
274  literal.append(nextArg);
275  break;
276  }
277 
278  literal.append(' ');
279  literal.append(nextArg);
280  }
281  return literal.toString();
282  }
283  }
284  // value is unquoted argument
285  else {
286  return currentArg;
287  }
288  }
