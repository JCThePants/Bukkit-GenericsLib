112  protected float calculateG() {
113 
114  float g = 0;
115  AStarNode parentNode;
116  AStarNode currentNode = _node;
117 
118  while ((parentNode = getParentNode(currentNode)) != null) {
119 
120  ICoords3Di parent = parentNode.getCoords();
121  ICoords3Di current = currentNode.getCoords();
122 
123  int deltaX = Math.abs(current.getX() - parent.getX());
124  int deltaY = Math.abs(current.getY() - parent.getY());
125  int deltaZ = Math.abs(current.getZ() - parent.getZ());
126 
127  if (deltaX == 1 && deltaY == 1 && deltaZ == 1) {
128  g += 0.7;
129  } else if (((deltaX == 1 || deltaZ == 1) && deltaY == 1) ||
130  ((deltaX == 1 || deltaZ == 1) && deltaY == 0)) {
131  g += 0.4;
132  } else {
133  g += 0.1;
134  }
135 
136  // move backwards
137  currentNode = parentNode;
138  }
139  return g;
140  }
