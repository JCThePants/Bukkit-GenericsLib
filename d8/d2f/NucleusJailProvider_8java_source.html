284  public void run(boolean silent, boolean doLateReleases) {
285  if (_sessionMap.isEmpty())
286  return;
287 
288  if (!hasProcessedLateReleases) {
289  doLateReleases = true;
290  hasProcessedLateReleases = true;
291  }
292 
293  List<IJailSession> jailSessions = new ArrayList<>(_sessionMap.values());
294 
295  Date now = new Date();
296 
297  for (IJailSession session : jailSessions) {
298 
299  boolean isLateRelease = _lateReleases.containsKey(session.getPlayerId());
300  if (isLateRelease && !doLateReleases)
301  continue;
302 
303  if (isLateRelease || session.isExpired()) {
304 
305  Player p = PlayerUtils.getPlayer(session.getPlayerId());
306  if (p == null) {
307 
308  // register player to be released at next login
309  Location releaseLocation = getReleaseLocation(session);
310  if (releaseLocation == null) {
311  NucMsg.warning("Failed to find a jail release location");
312  continue;
313  }
314 
315  registerLateRelease(session.getPlayerId(), releaseLocation);
316  continue;
317  }
318 
319  _sessionMap.remove(session.getPlayerId());
320 
321  Location releaseLoc = unregisterLateRelease(session.getPlayerId());
322  if (releaseLoc == null)
323  releaseLoc = getReleaseLocation(session);
324 
325  if (!session.isReleased())
326  session.release();
327 
328  if (releaseLoc != null)
329  p.teleport(releaseLoc);
330  }
331  else if (!silent) {
332 
333  long releaseMinutes = DateUtils
334  .getDeltaMinutes(now, session.getExpiration(), TimeRound.ROUND_UP);
335 
336  Long lastMessageMin = session.getMeta().get(RELEASE_MESSAGE_META);
337  if (lastMessageMin == null ||
338  (lastMessageMin != releaseMinutes &&
339  (releaseMinutes <= 5 || releaseMinutes % 10 == 0))) {
340 
341  Player p = PlayerUtils.getPlayer(session.getPlayerId());
342 
343  if (p != null) {
344  NucMsg.tellAnon(p, NucLang.get(_RELEASE_TIME, releaseMinutes));
345  }
346 
347  session.getMeta().setKey(RELEASE_MESSAGE_META, releaseMinutes);
348  }
349  }
350  }
351  }
