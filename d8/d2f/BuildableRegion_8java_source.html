124  public final boolean build(BuildSpeed buildSpeed, Collection<? extends ChunkSnapshot> snapshots) {
125 
126  // already building
127  if (_isBuilding)
128  return false;
129 
130  if (!isDefined())
131  return false;
132 
133  _isBuilding = true;
134 
135  QueueProject project = new QueueProject(getPlugin());
136 
137  for (ChunkSnapshot snapshot : snapshots) {
138 
139  World world = Bukkit.getWorld(snapshot.getWorldName());
140  if (world == null) {
141  NucMsg.debug(getPlugin(),
142  "Failed to get world named '{0}' while building region '{1}'.",
143  snapshot.getWorldName(), getName());
144  continue;
145  }
146 
147  Chunk chunk = world.getChunkAt(snapshot.getX(), snapshot.getZ());
148  if (chunk == null) {
149  NucMsg.debug(getPlugin(),
150  "Failed to get chunk ({0}, {1}) in world '{0}' while building region '{1}'.",
151  snapshot.getX(), snapshot.getZ(), snapshot.getWorldName(), getName());
152  continue;
153  }
154 
155  if (!chunk.isLoaded()) {
156  chunk.load();
157  }
158 
159  // get region chunk section calculations
160  RegionChunkSection section = new RegionChunkSection(this, snapshot);
161 
162  // get build chunk iterator task
163  BuildChunkIterator iterator = new BuildChunkIterator(this, snapshot, -1,
164  section.getStartChunkX(), section.getStartY(), section.getStartChunkZ(),
165  section.getEndChunkX(), section.getEndY(), section.getEndChunkZ());
166 
167  // add task to project
168  project.addTask(iterator);
169  }
170 
171  switch (buildSpeed) {
172  case PERFORMANCE:
173  QueueWorker.get().addTask(project);
174  break;
175 
176  case BALANCED:
177  project.run();
178  break;
179 
180  case FAST:
181  List<QueueTask> tasks = project.getTasks();
182  for (QueueTask task : tasks) {
183  task.run();
184  }
185  break;
186  }
187 
188  project.getResult().onStatus(new FutureSubscriber() {
189  @Override
190  public void on(FutureStatus status, @Nullable CharSequence message) {
191  _isBuilding = false;
192  }
193  });
194 
195  return true;
196  }

207  public BuildChunkIterator (Region region, ChunkSnapshot snapshot, long segmentSize,
208  int xStart, int yStart, int zStart,
209  int xEnd, int yEnd, int zEnd) {
210 
211  super(region.getPlugin(), TaskConcurrency.ASYNC, segmentSize, xStart, yStart, zStart, xEnd, yEnd, zEnd);
212 
213  this.blocks = new ArrayDeque<>((int)this.getVolume());
214  this.snapshot = snapshot;
215 
216  //noinspection ConstantConditions
217  this.chunk = region.getWorld().getChunkAt(snapshot.getX(), snapshot.getZ());
218  }
219 
220  @Override
221  public void onIterateItem(int x, int y, int z) {
222 
223  Material type = Material.getMaterial(snapshot.getBlockTypeId(x, y, z));
224  int data = snapshot.getBlockData(x, y, z);
225 
226  Block block = chunk.getBlock(x, y, z);
227  BlockState state = block.getState();
228 
229  if (state.getType() != type || (type != Material.AIR && state.getData().getData() != data)) {
230  state.setType(type);
231  state.setRawData((byte)data);
232  this.blocks.add(state);
233  }
234  }
