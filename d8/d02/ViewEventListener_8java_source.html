73  // Handle view closing.
74  @EventHandler
75  private void onInventoryClose(InventoryCloseEvent event) {
76 
77  ViewSession session = _sessions.get(event.getPlayer());
78  if (session == null)
79  return;
80 
81  // A disposed session should not be in the sessions map
82  if (session.isDisposed()) {
83  throw new IllegalStateException("A view session that is disposed is still registered.");
84  }
85 
86  // get the current session view.
87  View currentView = session.getCurrent();
88  if (currentView == null)
89  throw new AssertionError();
90 
91  switch (currentView.getCloseReason()) {
92 
93  // Player pressed escape to go back to the previous view.
94  case ESCAPE:
95  session.escaped();
96  if (session.getCurrent() != null) {
97  openView(ViewOpenReason.PREV, session.getCurrent());
98  }
99  break;
100 
101  // Called to go back to the previous view.
102  case PREV:
103  View prevView = session.getPrev();
104  if (prevView != null) {
105  openView(ViewOpenReason.PREV, prevView);
106  }
107  break;
108 
109  // Closing and re-opening the current view.
110  case REFRESH:
111  openView(ViewOpenReason.REFRESH, currentView);
112  break;
113 
114  // Closing the current view to open the next view.
115  case NEXT:
116  View nextView = session.getNext();
117  if (nextView != null) {
118  openView(ViewOpenReason.NEXT, nextView);
119  }
120  break;
121 
122  default:
123  throw new RuntimeException("The value returned from a views getCloseReason method " +
124  "cannot return null.");
125  }
126 
127  currentView.resetCloseReason();
128  }
