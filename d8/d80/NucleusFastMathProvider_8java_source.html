34  public static final String NAME = "NucleusFastMath";
35  static final XORShiftRandom XOR_RANDOM = new XORShiftRandom();
36 
37  private static final int SIN_BITS = 12;
38  private static final int SIN_MASK = ~(-1 << SIN_BITS);
39  private static final float DEG_TO_INDEX;
40  private static final float[] SIN_TABLE, COS_TABLE;
41 
42  private static final int ATAN2_BITS = 12;
43  private static final int ATAN2_BITS2 = ATAN2_BITS << 1;
44  private static final int ATAN2_MASK = ~(-1 << ATAN2_BITS2);
45  private static final int ATAN2_COUNT = ATAN2_MASK + 1;
46  private static final int ATAN2_DIM = (int) Math.sqrt(ATAN2_COUNT);
47  private static final float ATAN2_DIM_MINUS_1 = (ATAN2_DIM - 1);
48  private static final float[] ATAN2_TABLE = new float[ATAN2_COUNT];
49 
50  static {
51 
52  // cos/sin code from http://riven8192.blogspot.com/2009/08/fastmath-sincos-lookup-tables.html
53  // licensed: http://creativecommons.org/licenses/by/3.0/
54  int sinCount = SIN_MASK + 1;
55  float degFull = (float) (360.0);
56  float radFull = (float) (Math.PI * 2.0);
57  DEG_TO_INDEX = sinCount / degFull;
58  SIN_TABLE = new float[sinCount];
59  COS_TABLE = new float[sinCount];
60 
61  for (int i = 0; i < sinCount; i++) {
62  SIN_TABLE[i] = (float) Math.sin((i + 0.5f) / sinCount * radFull);
63  COS_TABLE[i] = (float) Math.cos((i + 0.5f) / sinCount * radFull);
64  }
65  for (int i = 0; i < 360; i += 90) {
66  SIN_TABLE[(int)(i * DEG_TO_INDEX) & SIN_MASK] = (float)Math.sin(i * Math.PI / 180.0);
67  COS_TABLE[(int)(i * DEG_TO_INDEX) & SIN_MASK] = (float)Math.cos(i * Math.PI / 180.0);
68  }
69 
70  // atan2 code from http://riven8192.blogspot.com/2009/08/fastmath-atan2-lookup-table.html
71  // licensed: http://creativecommons.org/licenses/by/3.0/
72  for (int i = 0; i < ATAN2_DIM; i++) {
73  for (int j = 0; j < ATAN2_DIM; j++) {
74 
75  float x = (float) i / ATAN2_DIM;
76  float y = (float) j / ATAN2_DIM;
77 
78  ATAN2_TABLE[j * ATAN2_DIM + i] = (float) Math.atan2(y, x);
79  }
80  }
81  }

118  @Override
119  public float atan2(double y, double x) {
120 
121  float add, mul;
122 
123  if (x < 0.0f) {
124 
125  if (y < 0.0f) {
126  x = -x;
127  y = -y;
128  mul = 1.0f;
129  } else {
130  x = -x;
131  mul = -1.0f;
132  }
133  add = -3.141592653f;
134 
135  } else {
136 
137  if (y < 0.0f) {
138  y = -y;
139  mul = -1.0f;
140  }
141  else {
142  mul = 1.0f;
143  }
144  add = 0.0f;
145  }
146 
147  float invDiv = ATAN2_DIM_MINUS_1 / (float)((x < y) ? y : x);
148  int xi = Math.min((int) (x * invDiv), (int)ATAN2_DIM_MINUS_1);
149  int yi = Math.min((int) (y * invDiv), (int)ATAN2_DIM_MINUS_1);
150 
151  return (ATAN2_TABLE[yi * ATAN2_DIM + xi] + add) * mul;
152  }
