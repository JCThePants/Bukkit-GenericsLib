67  @Nullable
68  public static PotionEffectType getPotionEffectType(Object object) {
69 
70  if (object instanceof String) {
71  String name = ((String)object).toUpperCase();
72 
73  PotionEffectType type = PotionEffectType.getByName(name);
74  if (type == null)
75  return null;
76 
77  return type;
78  }
79  else if (object instanceof Number) {
80  int id = ((Number)object).intValue();
81 
82  PotionType type = PotionType.getByDamageValue(id & 15);
83  if (type == null)
84  return null;
85 
86  return type.getEffectType();
87  }
88  else if (object instanceof PotionEffectType) {
89  return (PotionEffectType)object;
90  }
91  else if (object instanceof PotionEffect) {
92  PotionEffect effect = (PotionEffect)object;
93  return effect.getType();
94  }
95  else {
96  return null;
97  }
