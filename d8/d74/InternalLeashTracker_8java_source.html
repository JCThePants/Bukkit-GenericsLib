74  @Override
75  public <T extends Collection<Entity>> T getLeashed(Player player, T output) {
76  PreCon.notNull(player);
77  PreCon.notNull(output);
78 
79  synchronized (_playerMap) {
80  Set<Entity> entities = _playerMap.get(player.getUniqueId());
81  if (entities == null) {
82 
83  Chunk chunk = player.getLocation().getChunk();
84  Entity[] entityArray = chunk.getEntities();
85  entities = new HashSet<>(7);
86  for (Entity entity : entityArray) {
87  if (!(entity instanceof LivingEntity) || entity instanceof Player)
88  continue;
89 
90  if (entity.isDead() || !entity.isValid())
91  continue;
92 
93  LivingEntity living = (LivingEntity)entity;
94  if (living.isLeashed() && living.getLeashHolder().equals(player)) {
95  entities.add(living);
96  }
97  }
98 
99  _playerMap.put(player.getUniqueId(), entities);
100  }
101 
102  Iterator<Entity> iterator = entities.iterator();
103  while (iterator.hasNext()) {
104 
105  LivingEntity entity = (LivingEntity)iterator.next();
106 
107 
108  if (entity.isDead() || !entity.isValid() || !player.equals(entity.getLeashHolder())) {
109  iterator.remove();
110  }
111  else {
112  output.add(entity);
113  }
114  }
115  }
116 
117  return output;
118  }
