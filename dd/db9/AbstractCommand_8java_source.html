469  protected void setSetting(CommandSender sender, final ISettingsManager settings,
470  ICommandArguments args, String propertyArgName,
471  String valueArgName, @Nullable final Runnable onSuccess)
472  throws CommandException {
473 
474  PreCon.notNull(sender);
475  PreCon.notNull(settings);
476  PreCon.notNull(args);
477  PreCon.notNullOrEmpty(propertyArgName);
478  PreCon.notNullOrEmpty(valueArgName);
479 
480  final String settingName = args.getString(propertyArgName);
481  Object value;
482 
483  // get settings definitions
484  final PropertyDefinition propertyDefinition = settings.getDefinitions().get(settingName);
485  if (propertyDefinition == null)
486  throw new CommandException(NucLang.get(_UNRECOGNIZED_PROPERTY, settingName));
487 
488  PropertyValueType valueType = propertyDefinition.getValueType();
489 
490  switch (valueType.getType()) {
491 
492  case LOCATION:
493 
494  // get location value to use
495  args.getLocation(sender, valueArgName, new ILocationHandler() {
496 
497  @Override
498  public void onLocationRetrieved (Player player, Location location) {
499 
500  if (settings.set(settingName, location)) {
501  IChatMessage successMessage = NucLang.get(_SET_PROPERTY_SUCCESS, settingName,
502  TextUtils.formatLocation(location, true));
503 
504  tellSuccess(player, successMessage);
505 
506  if (onSuccess != null)
507  onSuccess.run();
508 
509  } else {
510 
511  tellError(player, NucLang.get(_SET_PROPERTY_FAILED, settingName));
512  tell(player, NucLang.get(_PROPERTY_DESCRIPTION,
513  propertyDefinition.getName(), propertyDefinition.getDescription()));
514  }
515  }
516 
517  });
518  return; // finish
519 
520  case ITEM_STACK_ARRAY:
521  value = args.getItemStack(sender, valueArgName);
522  break;
523 
524  default:
525  value = args.getString(valueArgName);
526  break;
527  }
528 
529  // make sure the result is valid
530  if (!settings.set(settingName, value)) {
531  throw new CommandException(
532  NucLang.get(_INVALID_PROPERTY_VALUE, value, settingName).toString() + '\n' +
533  NucLang.get(_PROPERTY_DESCRIPTION,
534  propertyDefinition.getName(), propertyDefinition.getDescription())
535  );
536  }
537 
538  tellSuccess(sender, NucLang.get(_SET_PROPERTY_SUCCESS, settingName, args.getString(valueArgName)));
539 
540  if (onSuccess != null)
541  onSuccess.run();
542  }
