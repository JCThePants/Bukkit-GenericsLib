159  public boolean take(Entity entity, ItemStackMatcher matcher, int qty) {
160  PreCon.notNull(entity);

162  PreCon.greaterThanZero(qty);
163 
164  List<ItemStack> itemsToTake = new ArrayList<>(_items.size() + 4);
165 
166  itemsToTake.addAll(_items);
167 
168  if (_armor[0] != null)

171  if (_armor[1] != null)
172  itemsToTake.add(getChestplate());
173 
174  if (_armor[2] != null)
175  itemsToTake.add(getLeggings());
176 
177  if (_armor[3] != null)
178  itemsToTake.add(getBoots());
179 
180  if (entity instanceof InventoryHolder) {
181 
182  InventoryHolder holder = (InventoryHolder)entity;
183 
184  // check entity has all required items
185  for (ItemStack item : itemsToTake) {
186  if (!InventoryUtils.has(holder.getInventory(), item, matcher, qty))
187  return false;
188  }
189 
190  // take items
191  for (ItemStack item : itemsToTake) {
192  InventoryUtils.removeAmount(holder.getInventory(), item, matcher, qty);
193  }
194  return true;
195  }
196 
197  return false;
198  }

337  protected void addAnyItems(Collection<ItemStack> items) {
338  PreCon.notNull(items);
339 
340  List<ItemStack> clone = new ArrayList<>(items);
341 
342  ItemStack helmet = null;
343  ItemStack chestplate = null;
344  ItemStack leggings = null;
345  ItemStack boots = null;
346 
347  Iterator<ItemStack> iterator = clone.iterator();
348  while (iterator.hasNext()) {
349  ItemStack item = iterator.next();
350 
351  Set<MaterialProperty> properties = Materials.getProperties(item.getType());
352 
353  if (properties.contains(MaterialProperty.ARMOR)) {
354 
355  if (properties.contains(MaterialProperty.HELMET)) {
356  helmet = item;
357  iterator.remove();
358  }
359  else if (properties.contains(MaterialProperty.CHESTPLATE)) {
360  chestplate = item;
361  iterator.remove();
362  }
363  else if (properties.contains(MaterialProperty.LEGGINGS)) {
364  leggings = item;
365  iterator.remove();
366  }
367  else if (properties.contains(MaterialProperty.BOOTS)) {
368  boots = item;
369  iterator.remove();
370  }
371  }
372 
373  if (helmet != null && chestplate != null && leggings != null && boots != null)
374  break;

378  setHelmet(helmet);
379 
380  if (chestplate != null)
381  setChestplate(chestplate);
382 
383  if (leggings != null)
384  setLeggings(leggings);
385 
386  if (boots != null)

388

423  protected boolean removeAnyItems(ItemStack... items) {
424  PreCon.notNull(items);

426  List<ItemStack> clone = new ArrayList<>(items.length);
427  Collections.addAll(clone, items);
428 
429  boolean[] armorFlags = new boolean[4];
430 
431  Iterator<ItemStack> iterator = clone.iterator();
432  while (iterator.hasNext()) {
433  ItemStack item = iterator.next();
434 
435  Set<MaterialProperty> properties = Materials.getProperties(item.getType());
436 
437  if (properties.contains(MaterialProperty.ARMOR)) {
438  int index;
439 
440  if (properties.contains(MaterialProperty.HELMET)) {
441  setHelmet(null);
442  index = 0;
443  } else if (properties.contains(MaterialProperty.CHESTPLATE)) {
444  setChestplate(null);
445  index = 1;
446  } else if (properties.contains(MaterialProperty.LEGGINGS)) {
447  setLeggings(null);
448  index = 2;
449  } else if (properties.contains(MaterialProperty.BOOTS)) {
450  setBoots(null);
451  index = 3;
452  }
453  else {
454  continue;
455  }
456 
457  armorFlags[index] = true;
458  iterator.remove();
459  }
460 
461  if (armorFlags[0] && armorFlags[1] && armorFlags[2] && armorFlags[3])
462  break;
463  }

466 
467  if (!InventoryUtils.remove(kitItems, items).isEmpty()) {
