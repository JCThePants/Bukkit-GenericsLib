57 class AddSubCommand extends AbstractCommand implements IExecutableCommand {
58 
59  @Localizable static final String _LEVEL_NOT_FOUND = "A friend level named '{0}' was not found.";
60  @Localizable static final String _PLAYER_NOT_FOUND = "A player named '{0: friend name}' was not found.";
61  @Localizable static final String _SUCCESS_ADD = "Player '{0}' added to your friends list.";
62  @Localizable static final String _SUCCESS_SET = "Player '{0}' friendship level changed.";
63 
64  @Override
65  public void execute(CommandSender sender, ICommandArguments args) throws CommandException {
66 
67  CommandException.checkNotConsole(getPlugin(), this, sender);
68 
69  String name = args.getString("friendName");
70  int rawLevel;
71 
72  if (args.hasInteger("level")) {
73  rawLevel = args.getInteger("level");
74  }
75  else {
76 
77  String levelName = args.getString("level");
78 
79  IFriendLevel level = Nucleus.getProviders().getFriends().getLevel(levelName);
80  if (level == null)
81  throw new CommandException(NucLang.get(_LEVEL_NOT_FOUND, levelName));
82 
83  rawLevel = level.getRawLevel();
84  }
85 
86  UUID friendId = PlayerUtils.getPlayerId(name);
87  if (friendId == null)
88  throw new CommandException(NucLang.get(_PLAYER_NOT_FOUND, name));
89 
90  Player player = (Player)sender;
91 
92  IFriend friend = Friends.get(player, friendId);
93  if (friend != null) {
94  friend.setRawLevel(rawLevel);
95  tellSuccess(sender, NucLang.get(_SUCCESS_SET, name));
96  }
97  else {
98  Friends.add(player, friendId, rawLevel);
99  tellSuccess(sender, NucLang.get(_SUCCESS_ADD, name));
100  }
101  }
102 }
