158  List<Player> visible = new ArrayList<>(30);
159 
160  for (Player player : players) {

162  if (!player.getWorld().equals(location.getWorld()))
163  continue;

165  Location playerLocation = player.getLocation(LOCATIONS.get());

222  private boolean showTo(Collection<? extends Player> players,
223  double x, double y, double z, int count) {
224 
225  INmsParticleEffectHandler handler = NmsUtils.getParticleEffectHandler();
226  if (handler == null)
227  return false;
228 
229  if (this instanceof AbstractRGBColorParticle) {
230  ((AbstractRGBColorParticle)this).showColoredTo(handler, players, x, y, z, count);
231  return true;
232  }
233  else {
234  showParticleTo(handler, players, x, y, z, count);
235  }
236 
237  return true;
