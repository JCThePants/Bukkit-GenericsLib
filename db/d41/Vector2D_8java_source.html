472  private static double calculateMagnitude(ICoords2D coords) {
473  if (coords instanceof ICoords3D) {
474  ICoords3D coords3D = (ICoords3D)coords;
475  return FastMath.sqrt(coords.getX() * coords.getX()
476  + coords3D.getY() * coords3D.getY()
477  + coords.getZ() * coords.getZ());
478  }
479  else {
480  return FastMath.sqrt(coords.getX() * coords.getX() + coords.getZ() * coords.getZ());
481  }
482  }
