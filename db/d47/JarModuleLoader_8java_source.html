146  File folder = getModuleFolder();

148  return;

154  new IValidator<File>() {

207  protected List<Class<T>> getModuleClasses(Collection<File> files) {
208 
209  if (files.isEmpty())
210  return new ArrayList<>(0);
211 
212  List<Class<T>> results = new ArrayList<>(30);
213 
214  for (File file : files) {
215 
216  ClassLoadMethod method = getLoadMethod(file);
217  if (method == null)
218  continue;
219 
220  switch (method) {
221 
222  case DIRECT_OR_SEARCH:
223  // fall through
224 
225  case DIRECT:
226 
227  Class<T> module;
228  try {
229  module = getModuleClass(file, method);
230  } catch (IOException e) {
231  e.printStackTrace();
232  continue;
233  }
234 
235  if (module != null) {
236  results.add(module);
237  break;
238  }
239  else if (method != ClassLoadMethod.DIRECT_OR_SEARCH) {
240  break;
241  }
242  // else fall through
243 
244  case SEARCH:
245  List<Class<T>> modules;
246  // get module classes from jar file
247  try {
248  modules = searchModuleClasses(file, method);
249  } catch (IOException e) {
250  e.printStackTrace();
251  continue;
252  }
253 
254  results.addAll(modules);
255  break;
256 
257  default:
258  throw new AssertionError();
259  }
260  }
261 
262  return results;
263  }

273  protected List<Class<T>> searchModuleClasses(File file, ClassLoadMethod loadMethod) throws IOException {
274 
275  List<Class<T>> moduleClasses = _moduleClassMap.get(file.getAbsolutePath());
276  if (moduleClasses != null)
277  return moduleClasses;
278 
279  JarFile jarFile = new JarFile(file);
280 
281  // validate jar file
282  if (!isValidJarFile(jarFile))
283  return new ArrayList<>(0);
284 
285  Enumeration<JarEntry> entries = jarFile.entries();
286 
287  URL[] urls = { new URL("jar:file:" + file + "!/") };
288 
289  // a search loader to be discarded when the search is complete
290  URLClassLoader classLoader = URLClassLoader.newInstance(urls, getClass().getClassLoader());
291 
292  // set to cache class names, if a module is found
293  // these are added to _loadedClasses set.
294  Set<String> searchResults = new HashSet<>(10);
295 
296  moduleClasses = new ArrayList<>(10);
297 
298  while (entries.hasMoreElements()) {
299 
300  JarEntry entry = entries.nextElement();
301 
302  // make sure entry is a class file
303  if (entry.isDirectory() || !entry.getName().endsWith(".class"))
304  continue;
305 
306  // get the class name
307  String className = entry.getName().substring(0, entry.getName().length() - 6);
308  className = className.replace('/', '.');
309 
310  // check if the class has already been loaded
311  if (_loadedClasses.contains(className))
312  continue;
313 
314  // validate the class before loading it
315  if (!isValidClassName(className))
316  continue;
317 
318  Class<?> c;
319  try {
320  c = classLoader.loadClass(className);
321  } catch (ClassNotFoundException e) {
322  e.printStackTrace();
323  continue;
324  }
325 
326  searchResults.add(className);
327 
328  // check if type is a module
329  if (getModuleClass().isAssignableFrom(c)) {
330 
331  @SuppressWarnings("unchecked")
332  Class<T> clazz = (Class<T>)c;
333 
334  // validate type
335  if (isValidType(clazz)) {
336  moduleClasses.add(clazz);
337  }
338  }
339  }
340 
341  jarFile.close();
342 
343  // module class not found
344  if (moduleClasses.isEmpty()) {
345  return moduleClasses;
346  }
347 
348  // cache module class
349  _moduleClassMap.put(file.getAbsolutePath(), moduleClasses);
350 
351  // add class names to prevent reloading
352  _loadedClasses.addAll(searchResults);
353 
354  if (moduleClasses.size() == 1 &&
355  loadMethod == ClassLoadMethod.DIRECT_OR_SEARCH) {
356  // save class name for direct load next time
357  saveClassName(file, moduleClasses.get(0));
358  }
359 
360  return moduleClasses;
361  }
