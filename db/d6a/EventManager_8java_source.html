302  public <T> T call(@Nullable Object caller, T event) {
303 
304  if (isDisposed())
305  throw new RuntimeException("Cannot use a disposed event manager.");
306 
307  // prevent redirected events from bubbling back to
308  // an event manager its was already called on.
309  synchronized (_sync) {
310  if (_calledEvents.contains(event))
311  return event;
312 
313  _calledEvents.add(event);
314  }
315 
316  // call event on parent first
317  if (_parent != null) {
318  _parent.call(caller, event);
319  }
320 
321  _callAgent.update(event);
322 
323  EventAgent agent = getEventAgent(event.getClass(), false);
324  if (agent == null)
325  return event;
326 
327  // check for Bukkit cancellable event
328  if (event instanceof Cancellable) {
329  agent.call(caller, new BukkitEventWrapper<T>(event));
330  }
331  else {
332  agent.call(caller, event);
333  }
334 
335  return event;
336  }

376  /*
377  * Extract all valid event subscribers from a listener.
378  */
379  private List<IEventSubscriber> extractSubscribers(IEventListener listener) {
380 
381  synchronized (_sync) {
382  if (_listeners.containsKey(listener))
383  throw new RuntimeException("Listener already registered.");
384  }
385 
386  // get all methods from listener
387  Method[] methods = listener.getClass().getDeclaredMethods();
388  List<IEventSubscriber> subscribers = new ArrayList<>(methods.length);
389 
390  for (Method method : methods) {
391 
392  EventMethod annotation = method.getAnnotation(EventMethod.class);
393  if (annotation == null)
394  continue;
395 
396  // event handlers must have exactly one parameter
397  Class<?>[] paramTypes = method.getParameterTypes();
398  if (paramTypes == null || paramTypes.length != 1)
399  continue;
400 
401  Class<?> eventClass = paramTypes[0];
402 
403  IEventSubscriber<?> wrapper = new EventMethodWrapper<>(listener, eventClass, method, annotation);
404  subscribers.add(wrapper);
405  }
406 
407  return subscribers;
408  }
