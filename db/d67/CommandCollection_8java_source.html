227  @Nullable
228  public String addCommand(RegisteredCommand container) {
229  PreCon.notNull(container);
230 
231  if (container.getDispatcher() != null) {
232  throw new RuntimeException("The command is already initialized and cannot be added.");
233  }
234 
235  // make sure command has required command info annotation
236  CommandInfo info = container.getCommand().getClass().getAnnotation(CommandInfo.class);
237  if (info == null) {
238  throw new RuntimeException(
239  "Could not find required CommandInfo annotation for command class: " +
240  container.getCommand().getClass().getName());
241  }
242 
243  String[] commandNames = info.command();
244  String primaryName = null;
245 
246  for (String commandName : commandNames) {
247 
248  if (_commandMap.containsKey(commandName))
249  continue;
250 
251  if (primaryName == null) {
252  primaryName = commandName;
253  _classMap.put(container.getCommand().getClass(), container);
254 
255  // clear sorted commands cache
256  _sortedCommands = null;
257  }
258 
259  _commandMap.put(commandName, container);
260  }
261
