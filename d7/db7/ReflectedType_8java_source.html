403  @Override
404  @Nullable
405  public <V> V invoke(@Nullable Object instance, String methodName, Object... arguments) {
406  PreCon.notNullOrEmpty(methodName, "methodName");
407  PreCon.notNull(arguments, "arguments");
408 
409  Method method = _aliasMethods != null ? _aliasMethods.get(methodName) : null;
410 
411  if (method == null) {
412 
413  _methodOutput.clear();
414  Collection<Method> methods = _cached.methodsByName(methodName, _methodOutput);
415 
416  // get method definition
417  if (methods.size() == 1) {
418  method = methods.iterator().next();
419  } else {
420  method = Reflection.findMethodByArgs(methods, methodName, arguments);
421  if (method == null)
422  throw new RuntimeException("Method '" + methodName +
423  "' not found in type " + _cached.getHandle().getCanonicalName());
424  }
425  }
426 
427  if (instance == null && !Modifier.isStatic(method.getModifiers())) {
428  throw new RuntimeException("Method " + methodName + " is not static.");
429  }
430  else if (instance != null && Modifier.isStatic(method.getModifiers())) {
431  throw new RuntimeException("Method " + methodName + " is static.");
432  }
433 
434  Object result;
435 
436  try {
437  result = method.invoke(instance, arguments);
438  } catch (IllegalAccessException | InvocationTargetException e) {
439  e.printStackTrace();
440  return null;
441  }
442 
443  @SuppressWarnings("unchecked")
444  V castedResult = (V)result;
445
