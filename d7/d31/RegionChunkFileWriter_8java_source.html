91  public RegionChunkFileWriter (IRegion region, int chunkX, int chunkZ) {
92  if (!region.isDefined())
93  throw new RuntimeException("Cannot get a snapshot from an undefined region.");
94 
95  //noinspection ConstantConditions
96  Chunk chunk = region.getWorld().getChunkAt(chunkX, chunkZ); // region.getWorld() is not null because region is defined
97  _plugin = region.getPlugin();
98  _region = region;
99  _world = region.getWorld();
100  _snapshot = chunk.getChunkSnapshot();
101  _section = new RegionChunkSection(region, _snapshot);
102 
103  // get tile entities from chunk
104  BlockState[] tileEntities = chunk.getTileEntities();
105  _tileEntities = new ArrayDeque<>(tileEntities.length);
106 
107  for (BlockState tile : tileEntities) {
108 
109  // make sure the tile entity is contained within the section
110  if (_section.containsBlockCoords(tile.getX(), tile.getY(), tile.getZ()))
111  _tileEntities.add(new SerializableBlockEntity(tile));
112  }
113 
114  // get entities from chunk
115  Entity[] entities = chunk.getEntities();
116  _entities = new ArrayDeque<>(25);
117 
118  Location entityLocation = new Location(null, 0, 0, 0);
119 
120  for (Entity entity : entities) {
121 
122  if (!entity.isValid())
123  continue;
124 
125  if (!SerializableFurnitureEntity.isFurnitureEntity(entity))
126  continue;
127 
128  entity.getLocation(entityLocation);
129 
130  // make sure the entity is contained within the section
131  if (!_section.containsBlockCoords(
132  entityLocation.getBlockX(), entityLocation.getBlockY(), entityLocation.getBlockZ()))
133  continue;
134 
135  _entities.add(new SerializableFurnitureEntity(entity));
136  }
137  }

175  public IFuture saveData(File file, @Nullable QueueProject project) {
176  PreCon.notNull(file);
177 
178  boolean runNow = project == null;
179 
180  if (project == null)
181  project = new QueueProject(_plugin);
182 
183  if (isSaving())
184  return project.cancel("Cannot save region because it is already saving.");
185 
186  _isSaving = true;
187 
188  SaveChunkIterator iterator = new SaveChunkIterator(file, 8192,
189  _section.getStartChunkX(), _section.getStartY(), _section.getStartChunkZ(),
190  _section.getEndChunkX(), _section.getEndY(), _section.getEndChunkZ());
191 
192  project.addTask(iterator);
193 
194  if (runNow) {
195  project.run();
196  }
197 
198  return project.getResult().onStatus(new FutureSubscriber() {
199  @Override
200  public void on(FutureStatus status, @Nullable CharSequence message) {
201  _isSaving = false;
202  }
203  });
204  }

257  @Override
258  public void onIterateItem(int x, int y, int z) {
259 
260  Material type = Material.getMaterial(_snapshot.getBlockTypeId(x, y, z));
261  int data = _snapshot.getBlockData(x, y, z);
262  int light = _snapshot.getBlockEmittedLight(x, y, z);
263  int skylight = _snapshot.getBlockSkyLight(x, y, z);
264 
265  int ls = (light << 4) | skylight;
266 
267  try {
268  writer.writeSmallString(type.name());
269  writer.write((short)data);
270  writer.write((byte)ls);
271  }
272  catch (IOException ioe) {
273  ioe.printStackTrace();
274 
275  fail("IOException while writing region data.");
276  _isSaving = false;
277  }
278  }

284  @Override

292  SerializableBlockEntity tileEntity = _tileEntities.remove();
293  writer.write(tileEntity);

301  writer.write(entity);

307  }
