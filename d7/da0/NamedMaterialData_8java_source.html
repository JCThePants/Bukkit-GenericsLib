62  @Nullable
63  public static MaterialData get(String materialName) {
64  PreCon.notNull(materialName);
65 
66  String str = materialName.toUpperCase();
67 
68  if (str.startsWith("MINECRAFT:"))
69  str = str.substring(10);
70 
71  // check for alternate value
72  MaterialData data = _alternateNames.getValue(str);
73  if (data == null) {
74 
75  // check for enum name
76  try {
77  Material material = Material.valueOf(str);
78  if (material != null)
79  return new MaterialData(material);
80  }
81  catch (Exception ignore) {}
82  return null;
83  }
84  return new MaterialData(data.getItemType(), data.getData());
85  }

96  public static String get(MaterialData materialData) {
97  PreCon.notNull(materialData);
98 
99  String name = _alternateNames.getKey(materialData);
100 
101  if (name == null) {
102  String result = materialData.getItemType().name();
103  if (materialData.getData() != 0)
104  result += ":" + materialData.getData();
105  return result;
106  } else {
107  return name;
108  }
109  }
