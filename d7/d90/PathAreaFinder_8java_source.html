61  public PathAreaResults search(AStar astar, Location start) {

64  start = LocationUtils.getBlockLocation(start);
65  LocationUtils.findSurfaceBelow(start, start);
66 
67  FinderContext context = new FinderContext(astar, start);
68 
69  // Add start node to open nodes
70  context.validNodes.add(start);
71 
72  // Add valid adjacent nodes to open list
73  searchAdjacent(context, start);
74 
75  Iterator<Location> iterator = context.validNodes.iterator();
76  while (iterator.hasNext()) {
77  Location location = iterator.next();
78  if (context.search(location).getStatus() != AStarResult.AStarResultStatus.RESOLVED) {

81  }
82  }
83

87  /*
88  * Search for valid nodes adjacent to the specified node.
89  */
90  private void searchAdjacent(FinderContext context, Location node) {
91 
92  LinkedList<StackItem> stack = new LinkedList<>();
93  stack.push(new StackItem(node));
94  StackItem curr = stack.peek();
95 
96  byte dropHeight = (byte)(-context.astar.getMaxDropHeight());
97 
98  while (!stack.isEmpty()) {
99 
100  start:
101  {
102  for (; curr.y >= dropHeight; curr.y--) {
103  for (; curr.x <= 1; curr.x++) {
104  for (; curr.z <= 1; curr.z++) {
105 
106  if (!curr.columns[curr.x + 1][curr.z + 1])
107  continue;
108 
109  // get instance of candidate node
110  Location candidate = curr.node.clone().add(curr.x, curr.y, curr.z);
111 
112  // check if candidate is already checked
113  if (context.invalidNodes.contains(candidate)) {
114  continue;
115  }
116 
117  if (context.validNodes.contains(candidate)) {
118  curr.columns[curr.x + 1][curr.z + 1] = false;
119  continue;
120  }
121 
122  int xRange = Math.abs(context.start.getBlockX() - candidate.getBlockX());
123  int yRange = Math.abs(context.start.getBlockY() - candidate.getBlockY());
124  int zRange = Math.abs(context.start.getBlockZ() - candidate.getBlockZ());
125 
126  // check x & z range
127  if ((context.astar.getRange() - xRange < 0) ||
128  (context.astar.getRange() - zRange < 0)) {
129 
130  curr.columns[curr.x + 1][curr.z + 1] = false;
131  continue;
132  }
133 
134  // check y range
135  if ((context.astar.getRange() - yRange < 0)) {
136  continue;
137  }
138 
139  // Check for diagonal obstruction
140  if (curr.x != 0 && curr.z != 0 && curr.y >= 0) {
141  Location diagX = curr.node.clone().add(curr.x, curr.y, (short) 0),
142  diagZ = curr.node.clone().add((short) 0, curr.y, curr.z);
143 
144  if (!isValid(diagX) && !isValid(diagZ)) {
145  curr.columns[curr.x + 1][curr.z + 1] = false;
146  continue;
147  }
148  }
149 
150  // check candidate to see if its valid
151  if (!isValid(candidate)) {
152 
153  // invalidate column if material is NOT transparent
154  if (!Materials.isTransparent(candidate.getBlock().getType())) {
155  curr.columns[curr.x + 1][curr.z + 1] = false;
156  }
157 
158  context.invalidNodes.add(candidate);
159  continue;
160  }
161 
162  context.validNodes.add(candidate);
163 
164  //searchAdjacent(context, candidate);
165  curr.z++;
166  stack.push(new StackItem(candidate));
167  curr = stack.peek();
168 
169 
170  break start;
171  }
172  curr.z = -1;
173  }
174  curr.x = -1;
175  }
176 
177  stack.pop();
178  if (stack.isEmpty()) {
179  return;
180  }
181  else {
182  curr = stack.peek();
183  }
184  }
185 
186  // "start" break location
187  }
188  }

191  * Determine if a node is a valid location.
192  */
193  private boolean isValid(Location loc) {
194 
195  Block block = loc.getBlock();
196 
197  // check if block is a surface
198  if (!Materials.isSurface(block.getType()))
199  return false;
200 
201  Block above = block.getRelative(0, 1, 0);
202  if (!Materials.isTransparent(above.getType()))
203  return false;
204 
205  Block above1 = block.getRelative(0, 2, 0);
206  return Materials.isTransparent(above1.getType());
207  }
