178  for (Constructor<?> c : candidates) {
179 
180  Class<?>[] ptypes = c.getParameterTypes();
181  if (ptypes.length != args.length)
182  continue;
183 
184  boolean isMatch = true;
185 
186  for (int i=0; i < ptypes.length; i++) {
187 
188  Class<?> arg = args[i].getClass();
189 
190  if (ptypes[i].isPrimitive() && !arg.isPrimitive()) {
191  arg = WRAPPERS_TO_PRIMITIVES.get(args[i].getClass());
192  if (arg == null)
193  continue;
194  }
195 
196  if (!ptypes[i].isAssignableFrom(arg)) {
197  isMatch = false;
198  break;
199  }
200  }
201 
202  if (isMatch) {
203  return c;
204  }
205  }
206 
207  return null;
