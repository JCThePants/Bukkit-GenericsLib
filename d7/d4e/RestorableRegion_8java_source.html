129  public IFuture saveData() throws IOException {
130 
131  Collection<IChunkCoords> chunks = this.getChunkCoords();
132 
133  if (chunks.size() == 0)
134  return new FutureAgent()
135  .cancel("Cannot save region because there are no chunks to save.");
136 
137  if (isSaving() || isRestoring())
138  return new FutureAgent()
139  .cancel("Cannot save region while it is already saving or restoring.");
140 
141  _isSaving = true;
142  onPreSave();
143 
144  IRegionFileWriter writer = getFileFormat().getWriter(this, getFileFactory());
145  return writer.save().onStatus(new FutureSubscriber() {
146  @Override

148  _isSaving = false;
149  onSaveComplete();
150  }
151  });

170  public IFuture restoreData(LoadSpeed loadSpeed) throws IOException {
171 
172  Collection<IChunkCoords> chunks = getChunkCoords();
173 
174  if (chunks.size() == 0)
175  return new FutureAgent().cancel("No chunks to restore.");
176 
177  if (isSaving())
178  return new FutureAgent().cancel("Region is still saving.");
179 
180  if (isRestoring())
181  return new FutureAgent().cancel("Region is still restoring.");
182 
183  if (!canRestore())
184  return new FutureAgent().cancel("Region cannot restore without restore files.");
185 
186  _isRestoring = true;
187  onPreRestore();
188 
189  removeEntities(Item.class, Monster.class, Animals.class);
190  removeEntities(SerializableFurnitureEntity.getFurnitureClasses());
191 
192  IRegionFileData fileData = new WorldBuilder(getPlugin(), getWorld());
193 
194  return getFileFormat()
195  .getLoader(this, getFileFactory())
196  .load(LoadType.MISMATCHED, loadSpeed, fileData)
197  .onStatus(new FutureSubscriber() {
198  @Override
199  public void on(FutureStatus status, @Nullable CharSequence message) {
200  _isRestoring = false;
201  onRestoreComplete();
202  }
203  });
204  }
