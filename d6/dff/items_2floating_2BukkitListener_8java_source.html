106  @EventHandler(priority = EventPriority.HIGHEST, ignoreCancelled = true)
107  private void onPlayerTryPickup(final PlayerPickupItemEvent event) {
108 
109  final FloatingItem item = _floatingItems.get(event.getItem().getUniqueId());
110  if (item == null)
111  return;
112 
113  ItemStack itemToGive = event.getItem().getItemStack();
114 
115  // restore display name
116  String displayName = item.getItem().getItemMeta().getDisplayName();
117  ItemMeta meta = itemToGive.getItemMeta();
118  meta.setDisplayName(displayName);
119  itemToGive.setItemMeta(meta);
120 
121  // prevent or allow pickup
122  if (!item.canPickup() && !item.isPickupSimulated()) {
123  event.setCancelled(true);
124  }
125 
126  FloatingItemPickUpEvent fiEvent = new FloatingItemPickUpEvent(item, event.getPlayer());
127  fiEvent.setCancelled(event.isCancelled());
128 
129  Nucleus.getEventManager().callBukkit(this, fiEvent);
130 
131  event.setCancelled(fiEvent.isCancelled());
132 
133  item.onTryPickup(event.getPlayer());
134 
135  // check for cancelled event.
136  if (event.isCancelled()) {
137 
138  // set meta back to uuid to prevent merging
139  meta.setDisplayName(UUID.randomUUID().toString());
140  itemToGive.setItemMeta(meta);
141  return;
142  }
143  else if (item.isPickupSimulated()) {
144  event.setCancelled(true);
145  item.despawn();
146  }
147 
148  final Location location = item.getLocation();
149 
150  if (location != null) {
151  _respawner.queue.add(new RespawnEntry(item,
152  System.currentTimeMillis() + (item.getRespawnTimeSeconds() * 1000)));
153  }
154  }

193  private static class Respawner implements Runnable {
194 
195  CircularQueue<RespawnEntry> queue = new CircularQueue<>();
196 
197  @Override
198  public void run() {
199 
200  int size = queue.size();
201 
202  for (int i=0; i < size; i++) {
203 
204  RespawnEntry entry = queue.peekFirst();
205  assert entry != null;
206 
207  if (entry.item.isDisposed() || entry.time <= System.currentTimeMillis()) {
208  queue.removeFirst();
209 
210  if (!entry.item.isDisposed()) {
211  entry.item.spawn();
212  }
213  }
214  else {
215  queue.next();
216  }
217  }
218  }
219  }
