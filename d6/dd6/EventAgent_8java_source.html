49  @Override
50  public synchronized void call(@Nullable Object caller, Object event) {
51 
52  if (isDisposed())
53  throw new RuntimeException("Cannot use a disposed EventProducer");
54 
55  ICancellable cancellable = null;
56  boolean isCancelled = false;
57  Object callEvent;
58 
59  if (event instanceof ICancellable) {
60  cancellable = (ICancellable)event;
61  isCancelled = cancellable.isCancelled();
62  }
63 
64  callEvent = event instanceof IEventWrapper ? ((IEventWrapper) event).getEvent() : event;
65 
66  List<IEventSubscriber> skippedSubscribers = new ArrayList<>(_eventSubscribers.size());
67  ListIterator<IEventSubscriber> iterator = new ArrayList<>(_eventSubscribers).listIterator();
68 
69  while (iterator.hasNext()) {
70 
71  boolean preCancel = isCancelled;
72 
73  @SuppressWarnings("unchecked")
74  IEventSubscriber<Object> subscriber = iterator.next();
75 
76  // check for cancelled event and determine if the subscriber
77  // will still be notified of the event.
78  if (subscriber.getPriority() != EventSubscriberPriority.WATCHER &&
79  isCancelled && !subscriber.isInvokedForCancelled()) {
80 
81  skippedSubscribers.add(subscriber);
82  continue;
83  }
84 
85  // notify the subscriber of the event
86  subscriber.onEvent(caller, callEvent);
87 
88  // check if the event is cancelled
89  if (cancellable != null) {
90  isCancelled = cancellable.isCancelled();
91  }
92 
93  boolean postCancel = isCancelled;
94 
95  // Make sure that WATCHER subscribers honor their priority type.
96  if (postCancel != preCancel && subscriber.getPriority() == EventSubscriberPriority.WATCHER)
97  throw new RuntimeException("Event subscribers with WATCHER priority " +
98  "cannot cancel or un-cancel an event.");
99 
100  // if cancelled event is uncancelled, run skipped subscribers next
101  if (!preCancel && postCancel && !skippedSubscribers.isEmpty()) {
102  for (IEventSubscriber skipped : skippedSubscribers) {
103  iterator.add(skipped);
104  }
105  skippedSubscribers.clear();
106  }
107  }
108  }
