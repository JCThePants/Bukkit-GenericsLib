107  public static <T> List<T> textSearch(Collection<T> candidates, String searchTerm,
108  ISearchTextGetter<T> textGetter) {
109  PreCon.notNull(candidates);
110  PreCon.notNull(searchTerm);
111  PreCon.notNull(textGetter);
112 
113  if (candidates.size() == 0)
114  return new ArrayList<>(0);
115 
116  PriorityQueue<WeightedSearchResult<T>> queue = new PriorityQueue<>(candidates.size());
117 
118  String caseSearch = searchTerm.toUpperCase();
119 
120  for (T candidate : candidates) {
121 
122  String text = textGetter.getText(candidate);
123  if (text == null)
124  continue;
125 
126  int weight = text.indexOf(searchTerm);
127  if (weight == -1) {
128 
129  String caseText = text.toUpperCase();
130  weight = caseText.indexOf(caseSearch);
131  if (weight == -1) {
132  continue;
133  }
134  else {
135  weight++; // increase weight of non-case matching
136  }
137  }
138 
139  queue.add(new WeightedSearchResult<T>(weight, candidate, text));
140  }
141 
142  List<T> results = new ArrayList<>(queue.size());
143  while (!queue.isEmpty()) {
144  results.add(queue.remove().item);
145  }
146 
147  return results;
148  }
