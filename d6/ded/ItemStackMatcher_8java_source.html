182  public boolean isMatch(@Nullable ItemStack stack1, @Nullable ItemStack stack2) {
183 
184  // if either stack is null, they must not be the same.
185  if (stack1 == null || stack2 == null)
186  return false;
187 
188  // same instance is always the same
189  if (stack1 == stack2)
190  return true;
191 
192  // compare type
193  if (_matchType) {
194  if (stack1.getType() != stack2.getType())
195  return false;
196 
197  Material type = stack1.getType();
198 
199  // check potion type
200  if (Materials.hasSubMaterialDurability(type)
201  && stack1.getDurability() != stack2.getDurability()) {
202  return false;
203  }
204 
205  // match data if type has color or sub material data
206  if ((Materials.hasColorData(type) || Materials.hasSubMaterialData(type)) &&
207  !stack1.getData().equals(stack2.getData())) {
208  return false;
209  }
210  }
211 
212  // compare meta data
213  if (_matchMeta) {
214 
215  if (stack1.hasItemMeta() != stack2.hasItemMeta())
216  return false;
217 
218  if (stack1.hasItemMeta() && !Bukkit.getItemFactory().equals(
219  stack1.getItemMeta(), stack2.getItemMeta())) {
220  return false;
221  }
222  }
223 
224  // compare durability
225  if (_matchDurability && stack1.getDurability() != stack2.getDurability())
226  return false;
227 
228  // compare amount
229  if (_matchAmount && stack1.getAmount() != stack2.getAmount())
230  return false;
231 
232  return true;
233  }
