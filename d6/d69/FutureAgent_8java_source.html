181  String agentName, FutureStatus status, @Nullable CharSequence message, Object[] args) {
182 
183  if (message != null)

248 
249  if (parent._finalStatus != null) {
250

252 
253  if (agentName.equals("onStatus") ||
254  (status == FutureStatus.SUCCESS && agentName.equals("onSuccess")) ||
255  (status == FutureStatus.CANCEL && agentName.equals("onCancel")) ||
256  (status == FutureStatus.ERROR && agentName.equals("onError"))) {
257  subscriber.on(parent._finalStatus, parent._finalMessage);
258  }
259  }
260

263  parent._hasFutureSubscribers = true;
