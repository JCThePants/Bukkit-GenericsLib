191  public ItemStack createItem(Object materialName, @Nullable Object data) {
192  PreCon.notNull(materialName, "materialName");
193 
194  MaterialData materialData;
195 
196  if (materialName instanceof String) {
197  materialData = NamedMaterialData.get((String)materialName);
198  PreCon.isValid(materialData != null, "materialName is invalid: {0}", materialName);
199  }
200  else {
201  Material material = EnumUtils.getEnum(materialName, Material.class);
202  materialData = new MaterialData(material);
203  }
204 
205  if (data != null) {
206 
207  if (data instanceof Number) {
208  materialData.setData(((Number)data).byteValue());
209  }
210  else if (data instanceof String) {
211  materialData.setData(TextUtils.parseByte((String)data, (byte)0));
212  }
213  }
214 
215  return materialData.toItemStack();
216  }
