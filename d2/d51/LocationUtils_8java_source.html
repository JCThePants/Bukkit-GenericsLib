648  @Nullable IValidator<Location> validator) {
649  PreCon.notNull(source);
650  PreCon.notNull(locations);
651 
652  Location closest = null;
653  double closestDist = Double.MAX_VALUE;
654 
655  for (Location loc : locations) {
656  if (validator != null && !validator.isValid(loc))
657  continue;
658 
659  double dist;
660  if ((dist = source.distanceSquared(loc)) < closestDist) {
661  closest = loc;
662  closestDist = dist;
663  }
664  }
665 
666  return closest;

681  public static boolean isInRange(Location source, Location target,

685  PreCon.positiveNumber(radiusX);

696  double deltaZ = Math.abs(source.getZ() - target.getZ());
697 
698  return deltaX <= radiusX && deltaY <= radiusY && deltaZ <= radiusZ;
699  }

865  public static Location rotate(Location axis, Location location,
866  double rotationX, double rotationY, double rotationZ,
867  Location output) {

869  PreCon.notNull(location);
870  PreCon.notNull(output);

872  Vector3D vector = VECTOR.copyFrom3D(location).subtract3D(axis);
873 
874  if (rotationX != 0.0D) {
875  IRotationMatrix xMatrix = FastMath.getRotationMatrix((float) rotationX);
876  xMatrix.rotateX(vector, vector);
877  }
878 
879  if (rotationY != 0.0D) {
880  IRotationMatrix yMatrix = FastMath.getRotationMatrix((float) rotationY);

890  output.setYaw(location.getYaw() + (float)rotationY);
891  output.setPitch(location.getPitch());
