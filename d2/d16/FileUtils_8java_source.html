85  public static List<File> getFiles(File folder,
86  DirectoryTraversal traversal,
87  @Nullable IValidator<File> fileValidator) {
88  PreCon.notNull(folder);
89  PreCon.isValid(folder.isDirectory(), "folder argument must be a folder.");
90  PreCon.notNull(traversal);
91 
92  File[] files = folder.listFiles();
93  if (files == null)
94  return new ArrayList<>(0);
95 
96  List<File> results = new ArrayList<File>(files.length);
97 
98  for (File file : files) {
99 
100  if (file.isDirectory() && traversal == DirectoryTraversal.RECURSIVE) {
101 
102  List<File> traversed = getFiles(file, DirectoryTraversal.RECURSIVE, fileValidator);
103  results.addAll(traversed);
104  }
105  else if (!file.isDirectory()) {
106 
107  if (fileValidator != null && !fileValidator.isValid(file))
108  continue;
109 
110  results.add(file);
111  }
112  }
113 
114  return results;
