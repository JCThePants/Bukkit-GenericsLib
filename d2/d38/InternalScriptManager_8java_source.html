326  public boolean evaluate(IScript script) {
327 
328  IEvaluatedScript current = _evaluated.remove(script.getName().toLowerCase());
329  if (current != null)
330  current.dispose();
331 
332  IEvaluatedScript evaluated;
333 
334  try {
335 
336  evaluated = script.evaluate(_api);
337  if (evaluated == null)
338  return false;
339  }
340  catch (Exception e) {
341  e.printStackTrace();
342  return false;
343  }
344 
345  _evaluated.put(script.getName().toLowerCase(), evaluated);
346 
347  return true;
348  }

350  public boolean reload(String scriptName) {
351  PreCon.notNullOrEmpty(scriptName);
352 
353  IScript script = getScript(scriptName);
354  if (script == null) {
355 
356  script = ScriptUtils.loadScript(Nucleus.getPlugin(),
357  _scriptFolder,
358  new File(_scriptFolder, scriptName + ".js"),
359  getScriptFactory());
360  }
361  else if (script.getFile() != null) {
362 
363  script = ScriptUtils.loadScript(Nucleus.getPlugin(),
364  _scriptFolder, script.getFile(), getScriptFactory());
365  }
366 
367  if (script == null)
368  return false;
369 
370  _scripts.put(script.getName().toLowerCase(), script);
371 
372  evaluate(script);
373  return true;
374  }

376  public boolean unload(String scriptName) {
377  PreCon.notNullOrEmpty(scriptName);
378 
379  IScript script = getScript(scriptName);
380  if (script == null)
381  return false;
382 
383  _scripts.remove(script.getName().toLowerCase());
384 
385  IEvaluatedScript evaluated = _evaluated.remove(script.getName().toLowerCase());
386  if (evaluated != null)
387  evaluated.dispose();
388 
389  return true;
390  }
