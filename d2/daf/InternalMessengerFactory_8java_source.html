115  @Override
116  public void tellImportant(final Player player, boolean clearMessages) {
117  PreCon.notNull(player);
118 
119  List<IDataNode> dataNodes = new ArrayList<>(_importantNodeCache.values());
120  final List<String> messages = new ArrayList<>(10);
121 
122  for (IDataNode data : dataNodes) {
123 
124  IDataNode playerData = data.getNode(player.getUniqueId().toString());
125 
126  boolean save = false;
127 
128  for (IDataNode contextData : playerData) {
129 
130  String prefix = contextData.getString("prefix", "");
131  String message = contextData.getString("message", "");
132  int lineLen = contextData.getInteger("lineLen", 60);
133 
134  assert prefix != null;
135  assert message != null;
136 
137  if (!message.isEmpty()) {
138  String[] lines = TextUtils.PATTERN_NEW_LINE.split(message);
139  for (String line : lines) {
140  messages.addAll(TextUtils.paginateString(line, prefix, lineLen, true));
141  }
142  }
143 
144  if (clearMessages) {
145  contextData.remove();
146  save = true;
147  }
148  }
149 
150  if (save)
151  data.save();
152  }
153 
154  if (!messages.isEmpty()) {
155  Scheduler.runTaskLater(Nucleus.getPlugin(), new Runnable() {
156  @Override
157  public void run() {
158  player.sendRawMessage(NucLang.get(_MESSAGES_HEADER).toString());
159 
160  for (String msg : messages) {
161  player.sendRawMessage(msg);
162  }
163 
164  player.sendRawMessage(NucLang.get(_MESSAGES_FOOTER).toString());
165  }
166  });
167  }
168  }
