108  public ParsedTabComplete parseTabComplete(ICommandOwner commandOwner,
109  CommandSender sender, String[] args) {
110  PreCon.notNull(commandOwner);
111  PreCon.notNull(sender);
112  PreCon.notNull(args);
113 
114  // get the primary command from the first argument
115  ParsedCommand parsed = parse(commandOwner, args, true);
116  if (parsed == null || args.length == 1) {
117 
118  List<String> result;
119 
120  if (args[0].isEmpty()) {
121  result = args.length == 1
122  // get all commands (that can be seen)
123  ? filterCommandNames(sender, "", commandOwner)
124  : new ArrayList<String>(5);
125  } else {
126 
127  // get possible command matches
128  result = filterCommandNames(sender, args[0], commandOwner);
129  }
130 
131  return new ParsedTabComplete(result, null, ArrayUtils.EMPTY_STRING_ARRAY);
132  }
133 
134  final RegisteredCommand command = parsed.getCommand();
135  final String[] arguments = parsed.getArguments();
136  ParsedTabComplete tabComplete;
137 
138  if (arguments.length == 1 &&
139  command.getCommands().size() > 0) {
140 
141  // generate list of sub command names the player has permission to use
142  List<String> names = filterCommandNames(sender, arguments[0], parsed.getCommand());
143 
144  tabComplete = new ParsedTabComplete(names, command, arguments);
145  }
146  else if (arguments.length == 0 && parsed.getCommand().getParent() != null) {
147 
148  // generate list of command names the player has permission to use
149  List<String> names = filterCommandNames(sender, "", parsed.getCommand().getParent());
150 
151  tabComplete = new ParsedTabComplete(names, command, arguments);
152 
153  }
154  else {
155  tabComplete = new ParsedTabComplete(new ArrayList<String>(5), command, arguments);
156  }

165  private List<String> filterCommandNames(final CommandSender sender,
166  String searchName,
167  final ICommandOwner commandOwner) {
168 
169  final String caseSearchName = searchName.toLowerCase();

179  element.toLowerCase().startsWith(caseSearchName);

187  private ParsedCommand parse(ICommandOwner commandCollection, String[] components, boolean isStrict) {
188  PreCon.notNull(commandCollection);
189  PreCon.notNull(components);
190 
191  // get the primary command from the first argument
192  RegisteredCommand command = null;
193 
194  if (components.length > 0 && !components[0].isEmpty()) {
195  command = (RegisteredCommand)commandCollection.getCommand(components[0]);
196  }
197 
198  // primary command not found
199  if (command == null) {
200 
201  if (!isStrict) {
202  return new ParsedCommand(_rootCommand, components, 0);
203  }
204 
205  return null;
206  }
207 
208  // trim the first element from the array
209  String[] args = ArrayUtils.reduceStart(1, components);
210 
211  // parse arguments and get command and command specific arguments
212  return getCommand(command, args, 1);
213  }

215  /*
216  * Recursively parses a String[] of arguments for the specified
217  * parent command and return a ParsedCommand containing the
218  * {@link AbstractCommand} implementation that should be used to execute the command
219  * as well as the arguments to be used for the returned command.
220  */
221  private ParsedCommand getCommand(RegisteredCommand parentCommand, @Nullable String[] args, int depth) {
222  PreCon.notNull(parentCommand);
223 
224  if (args == null || args.length == 0)
225  return new ParsedCommand(parentCommand, new String[0], depth);
226 
227  String subCmd = args[0].toLowerCase();
228  String[] params = ArrayUtils.reduceStart(1, args);
229 
230  RegisteredCommand subCommand = subCmd.isEmpty() ? null : parentCommand.getCommand(subCmd);
231  if (subCommand == null)
232  return new ParsedCommand(parentCommand, args, depth);
233  else {
234  ParsedCommand p = getCommand(subCommand, params, depth + 1);
235  if (p == null)
236  return new ParsedCommand(parentCommand, args, depth);
237 
238  return p;
239  }
