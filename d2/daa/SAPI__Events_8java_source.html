100  public boolean on(String eventName, String priority, final IScriptEventSubscriber handler) {
101  PreCon.notNullOrEmpty(eventName);
102  PreCon.notNullOrEmpty(priority);
103  PreCon.notNull(handler);
104 
105  String[] priorityComp = TextUtils.PATTERN_COLON.split(priority);
106  boolean ignoreCancelled = true;
107 
108  if (priorityComp.length == 2) {
109  if (priorityComp[1].equalsIgnoreCase("invokeForCancelled")) {
110  ignoreCancelled = false;
111  }
112  }
113 
114  Class<?> eventClass;
115 
116  try {
117  eventClass = Class.forName(eventName);
118  } catch (ClassNotFoundException e) {
119  e.printStackTrace();
120  return false;
121  }
122 
123  // check for and register bukkit events
124  if (Event.class.isAssignableFrom(eventClass)) {
125  Class<? extends Event> bukkitEventClass = eventClass.asSubclass(Event.class);
126  return registerBukkitEvent(bukkitEventClass, priorityComp[0], ignoreCancelled, handler);
127  }
128  else {
129  return registerNucleusEvent(eventClass, priorityComp[0], ignoreCancelled, handler);
130  }
131  }
