121  /*
122  * Invoked to check for and process a potential response.
123  */
124  private boolean onResponse(CommandSender sender, String message) {
125 
126  String[] messageComp = TextUtils.PATTERN_SPACE.split(message);
127 
128  if (messageComp.length > 2 || messageComp[0].length() == 0)
129  return false; // finished
130 
131  String commandName = messageComp[0].substring(1);
132  String contextName = messageComp.length > 1 ? messageComp[1].toLowerCase() : null;
133 
134  if (commandName.isEmpty())
135  return false;
136 
137  ResponseType type = ResponseType.from(commandName);
138 
139  List<IRequestContext> allRequests = getRequests(sender);
140  if (allRequests == null || allRequests.isEmpty())
141  return false; // finished
142 
143  List<IRequestContext> contexts = new ArrayList<>(allRequests.size());
144 
145  for (IRequestContext request : allRequests) {
146  if (request.contains(type) &&
147  (contextName == null || request.getSearchName().equals(contextName))) {
148  contexts.add(request);
149  }
150  }
151 
152  if (contexts.isEmpty())
153  return false; // finished
154 
155  if (contexts.size() == 1) {
156  IRequestContext context = contexts.get(0);
157  context.respond(type);
158  }
159  else if (contextName == null) {
160  tellMultipleRequests(sender, type, contexts);
161  }
162  else {
163  return false;
164  }
165 
166  return true;
167  }
