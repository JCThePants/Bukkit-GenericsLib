211  public IFuture enableProviders() {
212 
213  final FutureAgent agent = new FutureAgent();
214 
215  DependencyRunner<IDependantRunnable> runner =
216  new DependencyRunner<IDependantRunnable>(Nucleus.getPlugin());
217 
218  Iterator<IProvider> iterator = _toEnable.iterator();
219  while (iterator.hasNext()) {
220 
221  IProvider provider = iterator.next();
222 
223  try {
224  provider.registerTypes();
225  }
226  catch (Throwable e) {
227  e.printStackTrace();
228  iterator.remove();
229  }
230  }
231 
232  for (final IProvider provider : _toEnable) {
233  try {
234  provider.enable();
235  }
236  catch (Throwable e) {
237  e.printStackTrace();
238  continue;
239  }
240 
241  runner.add(new IDependantRunnable() {
242  @Override
243  public DependencyStatus getDependencyStatus() {
244  return provider.isLoaded()
245  ? DependencyStatus.READY
246  : DependencyStatus.NOT_READY;
247  }
248 
249  @Override
250  public void run() {
251  // do nothing
252  }
253  });
254  }
255 
256  runner.onFinish(new IFinishHandler<IDependantRunnable>() {
257  @Override
258  public void onFinish(List<IDependantRunnable> notRun) {
259  agent.success();
260  }
261  });
262 
263  _toEnable.clear();
264 
265  runner.start();
266 
267  return agent.getFuture();
268  }
