268 
269  if (nodePath.isEmpty())

272  TreeEntryNode<String, Object> node = _node;
273 
274  String[] pathNames = TextUtils.PATTERN_DOT.split(nodePath);
275 
276  for (int i=0; i < pathNames.length; i++) {
277  String pathName = pathNames[i];
278  TreeEntryNode<String, Object> child = node.getChild(pathName);
279 
280  if (child == null) {

284  else {
