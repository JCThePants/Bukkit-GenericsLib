72  PreCon.notNull(array1);

74 
75  int len = array1.length + array2.length;
76 
77  @SuppressWarnings("unchecked")
78  Class<T> componentClass = (Class<T>) array1.getClass().getComponentType();
79 
80  T[] result = newArray(componentClass, len);
81 
82  for (int i=0, j=0; i < len; i++) {
83  if (i < array1.length) {
84  result[i] = array1[i];
85  }
86  else {
87  result[i] = array2[j];
88  j++;
89  }
