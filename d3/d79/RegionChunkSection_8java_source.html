312  // initialize all variables
313  private void init(Location p1, Location p2) {
314 
315  _chunkBlockX = _chunkX * 16;
316  _chunkBlockZ = _chunkZ * 16;
317 
318  _chunkBlockEndX = _chunkBlockX + 15;
319  _chunkBlockEndZ = _chunkBlockZ + 15;
320 
321  //noinspection ConstantConditions
322  int xStart = _regionStartX = Math.min(p1.getBlockX(), p2.getBlockX());
323  int xEnd = _regionEndX = Math.max(p1.getBlockX(), p2.getBlockX());
324 
325  _yStart = Math.min(p1.getBlockY(), p2.getBlockY());
326  _yEnd = Math.max(p1.getBlockY(), p2.getBlockY());
327 
328  int zStart = _regionStartZ = Math.min(p1.getBlockZ(), p2.getBlockZ());
329  int zEnd = _regionEndZ = Math.max(p1.getBlockZ(), p2.getBlockZ());
330 
331  _startChunkX = (_chunkBlockX > xStart) ? 0 : xStart - _chunkBlockX;
332  _endChunkX = (xEnd > _chunkBlockEndX) ? 15 : 15 - (_chunkBlockEndX - xEnd);
333 
334  _startChunkZ = (_chunkBlockZ > zStart) ? 0 : zStart - _chunkBlockZ;
335  _endChunkZ = (zEnd > _chunkBlockEndZ) ? 15 : 15 - (_chunkBlockEndZ - zEnd);
336 
337  _startBlockX = _chunkBlockX + _startChunkX;
338  _endBlockX = _startBlockX + _endChunkX;
339 
340  _startBlockZ = _chunkBlockZ + _startChunkZ;
341  _endBlockZ = _startBlockZ + _endChunkZ;
342 
343  _p1 = new Location(p1.getWorld(), _startBlockX, _yStart, _startBlockZ);
344  _p2 = new Location(p2.getWorld(), _endBlockX, _yEnd, _endBlockZ);
345  }
