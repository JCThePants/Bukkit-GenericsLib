105  private String getCommand(int newPage) throws InvalidArgumentException {
106  IChatMessage cmdText = Nucleus.getCommandManager().getUsageGenerator().generate(_command,
107  "/{0: root command}{1: command path}{2: command}");
108 
109  StringBuilder output = new StringBuilder(cmdText);
110 
111  for (ICommandParameter parameter : _command.getInfo().getStaticParams()) {
112 
113  output.append(' ');
114  if (parameter.getName().equalsIgnoreCase(_pageName)) {
115  output.append(newPage);
116  }
117  else {
118  try {
119  output.append(_arguments.getString(parameter.getName()));
120  }
121  catch (CommandException ignore) {}
122  }
123  }
124 
125  for (ICommandParameter parameter : _command.getInfo().getFloatingParams()) {
126  if (parameter.getName().equalsIgnoreCase(_pageName)) {
127  output.append(" -");
128  output.append(parameter.getName());
129  output.append(' ');
130  output.append(newPage);
131  }
132  else {
133  ICommandArgument arg = _arguments.get(parameter.getName());
134  if (arg == null || arg.isDefaultValue() || arg.getValue() == null) {
135  continue;
136  }
137  output.append(" -");
138  output.append(parameter.getName());
139  output.append(' ');
140  output.append(arg.getValue());
141  }
142  }
143 
144  for (IFlagParameter parameter : _command.getInfo().getFlagParams()) {
145  if (_arguments.getBoolean(parameter.getName())) {
146  output.append(" --");
147  output.append(parameter.getName());
148  }
149  }
150 
151  return output.toString();
152  }
