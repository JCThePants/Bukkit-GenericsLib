102  return false;
103 
104  Converter<?> converter = definition.getConverter();
105  if (converter != null) {
106  value = converter.convert(value);
107  }
108 
109  IValidator<Object> validator = definition.getValidator();
110  if (!(validator != null && !validator.isValid(value)) && _dataNode.set(propertyName, value)) {
111 
112  if (_agents.hasAgent("onChange"))
113  _agents.update("onChange", new PropertyValue(this, definition, value));
114 
115  return true;

120  @Nullable
121  @Override
122  public <T> T get(String propertyName) {
123  PreCon.notNull(propertyName);
124 
125  PropertyDefinition definition = _definitions.get(propertyName);
126  if (definition == null)
127  return null;
128 
129  @SuppressWarnings("unchecked")
130  T result = (T)_dataNode.get(propertyName);
131 
132  if (result == null) {
133 
134  @SuppressWarnings("unchecked")
135  T defaultValue = (T) definition.getDefaultValue();
136 
137  return defaultValue;
138  }
139  else {
140  switch (definition.getValueType().getType()) {
141  case BOOLEAN:
142  if (result instanceof String) {
143 
144  @SuppressWarnings("unchecked")
145  T bool = (T)(Boolean) TextUtils.parseBoolean((String) result);
146  return bool;
147  }
148  break;
149  case INTEGER:
150  if (result instanceof String) {
151 
152  @SuppressWarnings("unchecked")
153  T integer = (T)(Integer)TextUtils.parseInt((String)result, 0);
154  return integer;
155  }
156  else if (result instanceof Long) {
157 
158  @SuppressWarnings("unchecked")
159  T integerByte = (T)(Integer)((Long)result).intValue();
160  return integerByte;
161  }
162  break;
163 
164  case LONG:
165  if (result instanceof String) {
166 
167  @SuppressWarnings("unchecked")
168  T l = (T)(Long)TextUtils.parseLong((String)result, 0L);
169  return l;
170  }
171  else if (result instanceof Integer) {
172 
173  @SuppressWarnings("unchecked")
174  T longInteger = (T)(Long)((Integer)result).longValue();
175  return longInteger;
176  }
177  break;
178 
179  case ITEM_STACK_ARRAY:
180  if (result instanceof String) {
181 
182  try {
183  @SuppressWarnings("unchecked")
184  T itemStacks = (T) ItemStackUtils.parse((String) result);
185  return itemStacks;
186  } catch (InvalidItemStackStringException e) {
187  e.printStackTrace();
188  return null;
189  }
190  }
191  break;
192 
193  case LOCATION:
194  if (result instanceof String) {
195 
196  @SuppressWarnings("unchecked")
197  T location = (T) LocationUtils.parseLocation((String) result);
198  return location;
199  }
200  break;
201 
202  case UNIQUE_ID:
203  if (result instanceof String) {
204 
205  @SuppressWarnings("unchecked")
206  T uuid = (T) TextUtils.parseUUID((String) result);
207  return uuid;
208  }
209  break;
210 
211  case ENUM:
212  if (result instanceof String) {
213 
214  @SuppressWarnings("unchecked")
215  T enumeration = (T) EnumUtils.searchEnum(
216  (String) result, (Class<Enum>) definition.getValueType().getTypeClass());
217 
218  return enumeration;
219  }
220  break;
221  }
222  }
223 
224  return result;
225  }
