42  private static final int ANGLE_PRECISION = 64;
43  private static final int ANGLE_MODULUS = 361 * ANGLE_PRECISION;
44  private static NucleusRotationMatrix[] MATRICES = new NucleusRotationMatrix[ANGLE_MODULUS];
45 
46  static {
47  for (int i=0; i < MATRICES.length; i++) {
48  MATRICES[i] = new NucleusRotationMatrix((float)(i / ANGLE_PRECISION) - 180f);
49  }
50  }

52  public static NucleusRotationMatrix get(float rotation) {
53 
54  while (rotation > 180f)
55  rotation -= 360f;
56  while (rotation < -180f)
57  rotation += 360f;
58 
59  int index = (int)((rotation + 180) * ANGLE_PRECISION + 0.5f);
60  return MATRICES[index];
61  }

67  NucleusRotationMatrix(float rotation) {
68  _rotation = rotation;
69  _cos = FastMath.cos(_rotation);
70  _sin = FastMath.sin(_rotation);
71  /*
72  newX = x * matrix[0][0] + y * matrix[1][0] + z * matrix[2][0];
73  newY = x * matrix[0][1] + y * matrix[1][1] + z * matrix[2][1];
74  newZ = x * matrix[0][2] + y * matrix[1][2] + z * matrix[2][2];
75  */
