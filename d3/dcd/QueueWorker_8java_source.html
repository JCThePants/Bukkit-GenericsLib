120  // make sure the current task is finished before
121  // starting the next one
122  if (_currentTask != null && !_currentTask.isEnded())
123  return;
124 
125  // check if all tasks are completed
126  if (_queue.isEmpty()) {
127  _currentTask = null;
128  cancelTask();
129  return;
130  }
131 
132  // get next item in queue
133  _currentTask = _queue.removeFirst();
134 
135  // make sure the task project hasn't been cancelled
136  if (_currentTask.isCancelled()) {
137 
138  _currentTask = null;
139  run(); // try the next task
140  }
141  else {
142 
143  switch (_currentTask.getConcurrency()) {
144  case MAIN_THREAD:
145  // already on main thread
146  // fall through
147 
148  case CURRENT_THREAD:
149  _currentTask.run();
150  break;
151 
152  case ASYNC:
153  Scheduler.runTaskLaterAsync(_currentTask.getPlugin(), 1, new Runnable() {
154  @Override
155  public void run() {
156  _currentTask.run();
157  }
158  });
