458  void remove() {
459 
460  Scheduler.runTaskLater(getPlugin(), new Runnable() {
461  @Override
462  public void run() {
463 
464  // check if player should be removed from queue immediately or
465  // wait for current song to end.
466  Player player = getPlayer();
467  boolean removeNow = player == null;
468  if (player != null) {
469 
470  World world = player.getWorld();
471  removeNow = world == null || !world.equals(_world);
472  }
473 
474  if (removeNow) {
475  removeNow();
476  }
477 
478  _isRemoved = true;
479  }
480  });
481  }
482 
488  void removeNow() {
489 
490  Player player = getPlayer();
491  if (player == null)
492  return;
493 
494  if (_queue != null)
495  _queue.clear();
496 
497  _playerQueues.remove(player);
498 
499  Set<PlayList> playLists = _instances.get(player);
500  if (playLists != null) {
501  playLists.remove(PlayList.this);
502  }
503 
504  _isRemoved = true;
505  }
506 
512  @Nullable
513  IResourceSound next() {
514 
515  Player player = getPlayer();
516  IResourceSound prev = _current;
517 
518  // check for first time use of queue
519  if (_queue == null) {
520  _queue = new LinkedList<>();
521  }
522  // check for end of queue or loop
523  else if (_isRemoved ||
524  player == null ||
525  (_queue.isEmpty() && !_isLoop)) {
526 
527  return _current = null;
528  }
529 
530  // refill queue if empty
531  if (_queue.isEmpty()) {
532 
533  refill();
534  if (_queue.isEmpty())
535  return _current = null;
536  }
537 
538  // check for linear or random playback
539  if (_isRandom) {
540  int index = Rand.getInt(0, _queue.size() - 1);
541  _current = _queue.remove(index);
542  }
543  else {
544  _current = _queue.pollFirst();
545  }
546 
547  return _current = onTrackChange(this, prev, _current);
548  }
549 
550  // refill queue
551  private void refill() {
552 
553  List<IResourceSound> sounds = getSounds(this, _loopCount);
554  _queue.addAll(sounds);
555 
556  onLoop(this, _queue, _loopCount);
557 
558  _loopCount++;
559  }
560  }
