214  private String getCommand(Player p, TitleCommandType type) {
215  StringBuilder buffer = new StringBuilder(100);
216 
217  buffer.append("title ");
218  buffer.append(p.getName());
219  buffer.append(' ');
220 
221  buffer.append(type.name());
222  buffer.append(' ');
223 
224  switch (type) {
225  case TITLE:
226  SimpleJSONBuilder.text(buffer, getTitleComponents());
227  break;
228  case SUBTITLE:
229  //noinspection ConstantConditions
230  SimpleJSONBuilder.text(buffer, getSubTitleComponents());
231  break;
232  case TIMES:
233  buffer.append(_fadeInTime);
234  buffer.append(' ');
235  buffer.append(_stayTime);
236  buffer.append(' ');
237  buffer.append(_fadeOutTime);
238  break;
239  case CLEAR:
240  break;
241  case RESET:
242  break;
243  default:
244  throw new AssertionError();
245  }
246 
247  return buffer.toString();
248  }
