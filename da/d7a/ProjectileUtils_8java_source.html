153  @Nullable
154  public static Double getBallisticAngle(Location source, Location target,
155  double elevation, double velocity, double gravity) {
156  PreCon.notNull(source);
157  PreCon.notNull(target);
158  PreCon.positiveNumber(velocity);
159 
160  if (!Objects.equals(source.getWorld(), target.getWorld())) {
161  return null;
162  }
163 
164  if (source.distanceSquared(target) <= 2.0D) {
165  return null;
166  }
167 
168  double velocitySquared = velocity * velocity;
169  double velocityQuad = velocitySquared * velocitySquared;
170 
171  double distanceSquared = source.distanceSquared(target);
172  double distance = Math.sqrt(distanceSquared);
173 
174  double formulaComponent = gravity * ((gravity * distanceSquared) + (2 * elevation * velocitySquared));
175 
176  // make sure formula can be correctly used
177  if (velocityQuad < formulaComponent)
178  return null;
179 
180  return Math.atan(
181  (velocitySquared - Math.sqrt(velocityQuad - formulaComponent))
182  /
183  (gravity * distance)
184  );
185  }

296  public static <T extends Projectile> T shootBallistic(Location source, Location target,
297  double sourceRadius, double velocity, double gravity,
298  Class<T> projectileClass) {
299  PreCon.notNull(source);
300  PreCon.notNull(target);
301  PreCon.positiveNumber(velocity);
302  PreCon.notNull(projectileClass);
303 
304  Vector vector = target.clone().subtract(source).toVector();
305  Double distance = source.distance(target);
306 
307  Vector radiusVector = vector.clone();
308 
309  double magnitude = Math.sqrt(
310  (radiusVector.getX() * radiusVector.getX()) +
311  (radiusVector.getY() * radiusVector.getY()) +
312  (radiusVector.getZ() * radiusVector.getZ()));
313 
314  magnitude -= sourceRadius;
315  magnitude = Math.max(0, magnitude);
316 
317  if (magnitude != 0)
318  radiusVector = radiusVector.multiply(1 / magnitude);
319 
320  source = source.add(radiusVector);
321 
322  Double launchAngle = getBallisticAngle(source, target, vector.getY(), velocity, gravity);
323  if (launchAngle == null) {
324  launchAngle = 0.0D;
325  }
326 
327  vector.setY(Math.tan(launchAngle) * distance);
328 
329  T projectile = source.getWorld().spawn(source, projectileClass);
330  projectile.setVelocity(vector);
331 
332  return projectile;
333  }
