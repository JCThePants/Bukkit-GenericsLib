72  /*
73  * Search adjacent locations around the specified location
74  * and add valid and invalid locations to their respective map.
75  */
76  protected void searchAdjacent(FinderContext context, Location node) {
77 
78  // column validations, work from top down, skip columns that are false
79  boolean[][] columns = new boolean[][] {
80  { true, true, true },
81  { true, true, true },
82  { true, true, true }
83  };
84 
85  boolean isBelowStart = node.getBlockY() <= context.start.getBlockY();
86 
87  byte yStart = (byte)(isBelowStart ? 1 : -1);
88 
89  for (byte y = yStart; isBelowStart ? y >= -1 : y <= 1; y += (isBelowStart ? -1 : 1)) {
90  for (byte x = -1; x <= 1; x++) {
91  for (byte z = -1; z <= 1; z++) {
92 
93  if (x == 0 && z == 0 && y == 0)
94  continue;
95 
96  // get instance of candidate node
97  Location candidate = node.clone().add(x, y, z);
98 
99  // check if candidate is already considered
100  if (context.invalidNodes.contains(candidate)) {
101  columns[x + 1][z + 1] = false;
102  continue;
103  }
104 
105  if (context.validNodes.contains(candidate)) {
106  continue;
107  }
108 
109  if (!columns[x + 1][z + 1]) {
110  continue;
111  }
112 
113  // make sure candidate is within boundaries
114  if (!context.boundaries.contains(candidate)) {
115  continue;
116  }
117 
118  // make sure candidate is air
119  if (candidate.getBlock().getType() != Material.AIR) {
120  context.invalidNodes.add(candidate);
121  columns[x + 1][z + 1] = false;
122  continue;
123  }
124 
125  // Check for diagonal obstruction
126  if (x != 0 && z != 0) {
127  Location diagX = node.clone().add(x, y, 0),
128  diagZ = node.clone().add(0, y, z);
129 
130  if(!Materials.isTransparent(diagX.getBlock().getType()) &&
131  !Materials.isTransparent(diagZ.getBlock().getType())) {
132 
133  context.invalidNodes.add(candidate);
134  columns[x + 1][z + 1] = false;
135  continue;
136  }
137  }
138 
139  // check for adjacent obstruction
140  if (y != 0) {
141  Location middle = node.clone().add(0, y, 0),
142  below = node.clone().add(x, 0, z);
143 
144  if (!Materials.isTransparent(middle.getBlock().getType()) &&
145  !Materials.isTransparent(below.getBlock().getType())) {
146  continue;
147  }
148  }
149 
150  // check for corner obstruction
151  if (x != 0 && y != 0 && z != 0) {
152  Location adjac1 = node.clone().add(x, 0, 0),
153  adjac2 = node.clone().add(0, 0, z),
154  middle = node.clone().add(0, y, 0);
155 
156  if (!Materials.isTransparent(adjac1.getBlock().getType()) &&
157  !Materials.isTransparent(adjac2.getBlock().getType()) &&
158  !Materials.isTransparent(middle.getBlock().getType())) {
159  continue;
160  }
161  }
162 
163  context.validNodes.add(candidate);
164 
165  searchAdjacent(context, candidate);
166 
167  }
168  }
169  }
