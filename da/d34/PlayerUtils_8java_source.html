368  public static <T extends Collection<Player>> T getNearbyPlayers(Location loc, double maxDistance,
369  @Nullable IValidator<Player> validator, T output) {
370  PreCon.notNull(loc, "loc");
371  PreCon.notNull(loc.getWorld(), "loc world");
372  PreCon.notNull(output);
373 
374  World world = loc.getWorld();
375  List<Player> players = world.getPlayers();
376  if (players.isEmpty())
377  return output;
378 
379  maxDistance *= maxDistance;
380 
381  if (output instanceof ArrayList)
382  ((ArrayList) output).ensureCapacity((int)(players.size() * 0.5D) + output.size());
383 
384  for (Player player : players) {
385  if (Npcs.isNpc(player))
386  continue;
387 
388  Location playerLoc = player.getLocation(PLAYER_LOCATION);
389  if (loc.distanceSquared(playerLoc) > maxDistance)
390  continue;
391 
392  if (validator != null && !validator.isValid(player))
393  continue;
394 
395  output.add(player);
396  }
397 
398  return output;
399  }

463  @Nullable
464  public static Player getClosestPlayer(Location loc, double radius,
465  @Nullable IValidator<Player> validator) {
466  PreCon.notNull(loc, "loc");
467  PreCon.notNull(loc.getWorld(), "loc world");
468 
469  World world = loc.getWorld();
470  List<Player> players = world.getPlayers();
471  if (players.isEmpty())
472  return null;
473 
474  radius *= radius;
475 
476  Player closest = null;
477  double closestDistSq = 0;
478 
479  for (Player player : players) {
480  if (Npcs.isNpc(player))
481  continue;
482 
483  Location playerLoc = player.getLocation(PLAYER_LOCATION);
484  double distanceSq = loc.distanceSquared(playerLoc);
485  if (distanceSq > radius)
486  continue;
487 
488  if (validator != null && !validator.isValid(player))
489  continue;
490 
491  if (closest == null || distanceSq < closestDistSq) {
492  closest = player;
493  closestDistSq = distanceSq;
494  }
495  }
496 
497  return closest;
