78  public boolean apply(Player player, ManagedScoreboard scoreboard) {
79  PreCon.notNull(player);
80  PreCon.notNull(scoreboard);
81 
82  // get the list of scoreboard instances applied to the player
83  PlayerScoreboards scoreboards;
84 
85  synchronized (_sync) {
86 
87  if (scoreboard.getLifespan() == ScoreboardLifespan.TRANSIENT) {
88  player.setScoreboard(scoreboard.getHandle());
89  return true;
90  }
91 
92  scoreboards = _playerMap.get(player.getUniqueId());
93 
94  // add a new list if one doesn't exist
95  if (scoreboards == null) {
96  scoreboards = new PlayerScoreboards();
97  _playerMap.put(player.getUniqueId(), scoreboards);
98  }
99 
100  // apply scoreboard
101  player.setScoreboard(scoreboard.getHandle());
102 
103  // make sure the scoreboard isn't already the most recent scoreboard
104  if (!scoreboards.isEmpty()) {
105  IManagedScoreboard current = scoreboards.peek();
106  if (scoreboard.equals(current)) {
107  return false;
108  }
109  }
110 
111  // push scoreboard
112  scoreboards.push(scoreboard);
113  }
114 
115  return true;
116  }

132  public boolean remove(Player player, ManagedScoreboard scoreboard) {
133  PreCon.notNull(player);
134  PreCon.notNull(scoreboard);
135 
136  synchronized (_sync) {
137 
138  // get the list of scoreboard instances applied to the player
139  PlayerScoreboards scoreboards = _playerMap.get(player.getUniqueId());
140  if ((scoreboards == null || scoreboards.isEmpty()) &&
141  scoreboard.getLifespan() != ScoreboardLifespan.TRANSIENT) {
142  return false;
143  }
144 
145  if (scoreboard.getLifespan() == ScoreboardLifespan.TRANSIENT) {
146 
147  ManagedScoreboard managed = scoreboards != null && !scoreboards.isEmpty()
148  ? scoreboards.peek()
149  : null;
150 
151  if (scoreboard.getHandle().equals(player.getScoreboard())) {
152  setScoreboard(player, managed != null ? managed.getHandle() : null);
153  }
154  return true;
155  }
156 
157  ManagedScoreboard current = scoreboards.peek();
158  ManagedScoreboard next = null;
159  boolean isRemoved;
160 
161  if (scoreboard.equals(current)) {
162  scoreboards.pop();
163  isRemoved = true;
164 
165  if (!scoreboards.isEmpty()) {
166  next = scoreboards.peek();
167  }
168  } else {
169  isRemoved = scoreboards.removeLastOccurrence(scoreboard);
170  }
171 
172  if (next != null) {
173  setScoreboard(player, next.getHandle());
174  }
175  else if (scoreboard.getHandle().equals(player.getScoreboard())) {
176  setScoreboard(player, null);
177  }
178 
179  return isRemoved;
180  }
181  }
