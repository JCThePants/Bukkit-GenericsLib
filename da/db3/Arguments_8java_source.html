568  @Override
569  public ItemStack[] getItemStack(@Nullable CommandSender sender, String parameterName)
570  throws InvalidArgumentException {
571  PreCon.notNullOrEmpty(parameterName);
572 
573  // get the raw argument
574  String arg = getRawArgument(parameterName);
575 
576  // make sure the argument was provided
577  if (arg == null)
578  throw invalidArg(parameterName, ArgumentValueType.ITEMSTACK);
579 
580  // Check for "inhand" keyword as argument
581  if (arg.equalsIgnoreCase("inhand")) {
582 
583  // sender must be a player to use "inhand" keyword
584  if (!(sender instanceof Player))
585  throw invalidArg(parameterName, ArgumentValueType.ITEMSTACK);
586 
587 
588  Player p = (Player)sender;
589 
590  ItemStack inhand = p.getItemInHand();
591 
592  // sender must have an item in hand
593  if (inhand == null || inhand.getType() == Material.AIR) {
594  throw invalidArg(parameterName, ArgumentValueType.ITEMSTACK);
595  }
596 
597  return new ItemStack[] { inhand }; // finished
598  }
599 
600  // Check for "hotbar" keyword as argument
601  if (arg.equalsIgnoreCase("hotbar")) {
602 
603  // sender must be a player to use "hotbar" keyword
604  if (!(sender instanceof Player))
605  throw invalidArg(parameterName, ArgumentValueType.ITEMSTACK);
606 
607  Player p = (Player)sender;
608 
609  Inventory inventory = p.getInventory();
610 
611  Set<MatchableItem> wrappers = new HashSet<MatchableItem>(9);
612 
613  // iterate and add players hotbar items to wrapper set.
614  for (int i=0; i < 9; i++) {
615  ItemStack item = inventory.getItem(i);
616  if (item != null && item.getType() != Material.AIR) {
617  MatchableItem wrapper = new MatchableItem(item, ItemStackMatcher.getDefault());
618  wrappers.add(wrapper);
619  }
620  }
621 
622  // generate result {@link ItemStack} array
623  ItemStack[] result = new ItemStack[wrappers.size()];
624  int index = 0;
625  for (MatchableItem wrapper : wrappers) {
626  result[index] = wrapper.getItem();
627  index++;
628  }
629 
630  return result; // finished
631  }
632 
633  // Check for "inventory" keyword as argument
634  if (arg.equalsIgnoreCase("inventory")) {
635 
636  // sender must be player to use "chest" keyword
637  if (!(sender instanceof Player))
638  throw invalidArg(parameterName, ArgumentValueType.ITEMSTACK);
639 
640 
641  Player p = (Player)sender;
642 
643  Inventory inventory = p.getInventory();
644 
645  int len = inventory.getContents().length;
646  Set<MatchableItem> wrappers = new HashSet<MatchableItem>(len);
647 
648  // iterate and add players chest items to wrapper set
649  for (int i=0; i < len; i++) {
650  ItemStack item = inventory.getItem(i);
651  if (item != null && item.getType() != Material.AIR) {
652 
653  // use item wrapper to prevent duplicates in set
654  MatchableItem wrapper = new MatchableItem(item);
655  wrappers.add(wrapper);
656  }
657  }
658 
659  // generate result {@link ItemStack} array.
660  ItemStack[] result = new ItemStack[wrappers.size()];
661  int index = 0;
662  for (MatchableItem wrapper : wrappers) {
663  result[index] = wrapper.getItem();
664  index++;
665  }
666 
667  return result; // finished
668  }
669 
670  // no keywords used, attempt to parse the argument into an {@link ItemStack} array.
671  ItemStack[] stacks = null;
672 
673  try {
674  stacks = ItemStackUtils.parse(arg);
675  } catch (InvalidItemStackStringException ignore) {
676  // do nothing
677  }
678 
679  if (stacks == null)
680  throw invalidArg(parameterName, ArgumentValueType.ITEMSTACK);
681 
682  // return result
683  return stacks;
684  }
