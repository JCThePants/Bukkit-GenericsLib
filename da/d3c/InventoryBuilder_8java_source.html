257  public Inventory build(@Nullable InventoryHolder holder) {
258 
259  Inventory inventory = _title != null
260  ? Bukkit.createInventory(holder, _rows * 9, _title)
261  : Bukkit.createInventory(holder, _rows * 9);
262 
263  if (_layoutArray != null) {
264  fillFromLayoutArray(inventory);
265  }
266  else if (_layout != null) {
267  prepLayout();
268  fillFromLayoutArray(inventory);
269  }
270  else if (_items != null) {

273  ItemStack item = _items.get(i);
