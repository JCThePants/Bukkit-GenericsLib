505  public static String titleCase(String text, TitleCaseOption... options) {
506  PreCon.notNull(text);
507 
508  if (text.length() < 2) {
509  return text;
510  }
511 
512  boolean forceCasing = false;
513  boolean removeUnderscores = false;
514  boolean ignoreExcluded = false;
515 
516  for (TitleCaseOption option : options) {
517  switch (option) {
518  case ALL:
519  removeUnderscores = true;
520  ignoreExcluded = true;
521  forceCasing = true;
522  break;
523  case REMOVE_UNDERSCORE:
524  removeUnderscores = true;
525  break;
526  case IGNORE_EXCLUDED_WORDS:
527  ignoreExcluded = true;
528  break;
529  case FORCE_CASING:
530  forceCasing = true;
531  break;
532  }
533  }
534 
535  if (removeUnderscores) {
536  text = text.replace('_', ' ');
537  }
538 
539  String[] words = PATTERN_SPACE.split(text);
540  StringBuilder resultSB = new StringBuilder(text.length() + 10);
541 
542  for (int i=0; i < words.length; i++) {
543 
544  if (i != 0)
545  resultSB.append(' ');
546 
547  String word = words[i];
548 
549  String firstLetter = word.substring(0, 1);
550  String wordEnd = word.substring(1, word.length());
551 
552  if (i == 0 || ignoreExcluded || !isTitleCaseExcluded(word)) {
553  firstLetter = firstLetter.toUpperCase();
554  }
555 
556  if (forceCasing) {
557  wordEnd = wordEnd.toLowerCase();
558  }
559 
560  resultSB.append(firstLetter);
561  resultSB.append(wordEnd);
562  }
563 
564  return resultSB.toString();
565  }

740  public static List<String> paginateString(String str, @Nullable String linePrefix,
741  int maxLineLen, boolean excludeColorsFromLen) {
742  PreCon.notNull(str);
743  PreCon.isValid(maxLineLen > 1);
744 
745  if (linePrefix != null)
746  str = str.replace(linePrefix, "");
747  else
748  linePrefix = "";
749 
750  String[] words = PATTERN_SPACE.split(str);
751 
752  List<String> results = new ArrayList<String>(str.length() / maxLineLen);
753 
754  StringBuilder line = new StringBuilder(maxLineLen);
755  line.append(linePrefix);
756 
757  int prefixSize = linePrefix.length();
758 
759  boolean wordAddedToLine = false;
760 
761  for (int i=0; i < words.length; i++) {
762 
763  int lineLength = excludeColorsFromLen
764  ? TextColor.remove(line).length()
765  : line.length();
766 
767  int wordLength = excludeColorsFromLen
768  ? TextColor.remove(words[i]).length()
769  : words[i].length();
770 
771  if (lineLength + wordLength + 1 <= maxLineLen &&
772  prefixSize + wordLength < maxLineLen) { // append to current line
773 
774  if (wordAddedToLine)
775  line.append(' ');
776 
777  line.append(words[i]);
778  wordAddedToLine = true;
779  }
780  else { // create new line
781 
782  String format = null;
783 
784  if (line.length() != 0 && i != 0) {
785  String finishedLine = line.toString();
786  format = ChatColor.getLastColors(finishedLine);
787  results.add(finishedLine);
788  }
789 
790  line.setLength(0);
791  line.append(linePrefix);
792  if (format != null)
793  line.append(format);
794 
795  if (prefixSize + wordLength >= maxLineLen) {
796  line.append(words[i]);
797  wordAddedToLine = true;
798  }
799  else {
800 
801  if (i != 0) {
802  i = i - 1;
803  }
804 
805  wordAddedToLine = false;
806  }
807  }
808  }
809 
810  // make sure last line is added.
811  // get left behind if number of words runs out
812  // and there is still room for more.
813  if (wordAddedToLine) {
814  results.add(line.toString());
815  }
816 
817  return results;
818  }
