63  public static List<ItemStack> add(ItemStack[] array, ItemStack... itemStacks) {
64 
65  List<ItemStack> toAdd = new ArrayList<>(itemStacks.length);
66  Collections.addAll(toAdd, itemStacks);
67  ItemStackMatcher matcher = ItemStackMatcher.getTypeMetaDurability();
68 
69  for (int i=0; i < array.length; i++) {
70  ItemStack item = array[i];
71 
72  Iterator<ItemStack> iterator = toAdd.iterator();
73  while (iterator.hasNext()) {
74  ItemStack newItem = iterator.next();
75 
76  if (newItem == null || newItem.getType() == Material.AIR) {
77  iterator.remove();
78  continue;
79  }
80 
81  if (item == null || item.getType() == Material.AIR) {
82  array[i] = newItem;
83  iterator.remove();
84  continue;
85  }
86 
87  int maxStackSize = item.getType().getMaxStackSize();
88 
89  // make sure item is not already full
90  if (item.getAmount() == maxStackSize)
91  break;
92 
93  if (!matcher.isMatch(item, newItem))
94  continue;
95 
96  // merge
97  int space = maxStackSize - item.getAmount();
98  int add = Math.min(newItem.getAmount(), space);
99 
100  item.setAmount(item.getAmount() + add);
101 
102  if (newItem.getAmount() >= add) {
103  iterator.remove();
104  }
105  else {
106  newItem.setAmount(newItem.getAmount() - add);
107  }
108 
109  // check no more room in item
110  if (add == space)
111  break;
112  }
113 
114  // check no more items to add
115  if (toAdd.size() == 0)
116  break;
117  }
118 
119  return toAdd;
120  }

798  public static List<ItemStack> removeAmount(Inventory inventory,
799  ItemStack itemStack,
800  ItemStackMatcher matcher, int qty) {
801  PreCon.notNull(inventory);
802  PreCon.notNull(itemStack);
803  PreCon.notNull(matcher);
804  PreCon.positiveNumber(qty);
805 
806  int size = inventory.getSize();
807 
808  List<ItemStack> results = new ArrayList<ItemStack>(size);
809 
810  int qtyLeft = qty;
811 
812  for (int i=0; i < size ; i++) {
813 
814  if (qtyLeft <= 0)
815  return results;
816 
817  ItemStack item = inventory.getItem(i);
818 
819  if (item == null || item.getType() == Material.AIR)
820  continue;
821 
822  if (matcher.isMatch(itemStack, item)) {
823 
824  ItemStack clone = item.clone();
825 
826  if (item.getAmount() > qtyLeft) {
827 
828  int newAmount = item.getAmount() - qtyLeft;
829  item.setAmount(newAmount);
830  clone.setAmount(qtyLeft);
831  results.add(clone);
832 
833  inventory.setItem(i, item);
834 
835  break;
836  }
837  else {
838 
839  qtyLeft -= item.getAmount();
840 
841  clone.setAmount(item.getAmount());
842  results.add(clone);
843 
844  inventory.setItem(i, ItemStackUtils.AIR);
845  }
846  }
847  }
848 
849  if (qtyLeft > 0 && inventory instanceof PlayerInventory) {
850  PlayerInventory playerInventory = (PlayerInventory)inventory;
851 
852  Set<MaterialProperty> properties = Materials.getProperties(itemStack.getType());
853 
854  if (properties.contains(MaterialProperty.ARMOR)) {
855 
856  if (properties.contains(MaterialProperty.HELMET) &&
857  matcher.isMatch(playerInventory.getHelmet(), itemStack)) {
858 
859  results.add(playerInventory.getHelmet());
860  playerInventory.setHelmet(null);
861  }
862  else if (properties.contains(MaterialProperty.CHESTPLATE) &&
863  matcher.isMatch(playerInventory.getChestplate(), itemStack)) {
864 
865  results.add(playerInventory.getChestplate());
866  playerInventory.setChestplate(null);
867  }
868  else if (properties.contains(MaterialProperty.LEGGINGS) &&
869  matcher.isMatch(playerInventory.getLeggings(), itemStack)) {
870 
871  results.add(playerInventory.getLeggings());
872  playerInventory.setLeggings(null);
873  }
874  else if (properties.contains(MaterialProperty.BOOTS) &&
875  matcher.isMatch(playerInventory.getBoots(), itemStack)) {
876 
877  results.add(playerInventory.getBoots());
878  playerInventory.setLeggings(null);
879  }
880  }
881  }
882 
883  return results;
884  }
