184  public void sendResult(Result<R> result) {
185  PreCon.notNull(result);
186 
187  _finalResult = result;
188 
189  if (!hasSubscribers())
190  return;
191 
192  if (result.isSuccess()) {
193  sendSuccess(result);
194  }
195  else if (result.isComplete()) {
196 
197  if (result.isCancelled()) {
198  sendCancel(result);
199  }
200  else {
201  sendError(result);
202  }
203  }
204  else {
205 
206  List<ISubscriber> list = new ArrayList<>(subscribers());
207 
208  for (ISubscriber subscriber : list) {
209  if (subscriber instanceof FutureResultSubscriber) {
210  //noinspection unchecked
211  ((FutureResultSubscriber<R>) subscriber).on(result);
212  }
213  }
214 
215  _updateAgents.update("onResult", result);
216  }
