50  public static void addEnchantments(
51  ItemStack stack, Collection<? extends IEnchantmentLevel> enchantments) {
52 
53  PreCon.notNull(stack);
54  PreCon.notNull(enchantments);
55 
56  ItemMeta meta = stack.getItemMeta();
57 
58  // check for enchantment storage items such as enchanted books
59  if (meta instanceof EnchantmentStorageMeta) {
60  EnchantmentStorageMeta storage = (EnchantmentStorageMeta)meta;

68  else {
69  for (IEnchantmentLevel enchant : enchantments) {
70  stack.addUnsafeEnchantment(enchant.getEnchantment(), enchant.getLevel());
71  }
