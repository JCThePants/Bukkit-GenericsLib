181 
182  @Override
183  public void run() {
184 
185  // make sure the current task is finished before
186  // starting the next one
187  if (_currentTask != null && !_currentTask.isEnded())
188  return;
189 
190  // check if all tasks are completed
191  synchronized (_tasks) {
192  if (_tasks.isEmpty()) {
193  complete();
194  _currentTask = null;
195  return;
196  }
197 
198  // get next item in queue
199  _currentTask = _tasks.removeFirst();
200  }
201 
202  // make sure the task project hasn't been cancelled
203  if (_currentTask.isCancelled()) {
204 
205  _currentTask = null;
206  run();
207  }
208  else {
209 
210  switch (_currentTask.getConcurrency()) {
211  case MAIN_THREAD:
212  if (!Bukkit.isPrimaryThread()) {
213 
214  Scheduler.runTaskSync(_currentTask.getPlugin(), new Runnable() {
215 
216  @Override
217  public void run() {
218  _currentTask.run();
219  }
220  });
221  break;
222  }
223  // fall through
224 
225  case CURRENT_THREAD:
226  _currentTask.run();
227  break;
228 
229  case ASYNC:
230  if (Bukkit.isPrimaryThread()) {
231  Scheduler.runTaskLaterAsync(_currentTask.getPlugin(), 1, new Runnable() {
232  @Override
233  public void run() {
234  _currentTask.run();
235  }
236  });
237  }
238  else {
239  _currentTask.run();
240  }
241  break;
242  }
243  }
