100  /*
101  * Parse the next item stack in the string.
102  * Returns null if there are no more item stacks.
103  */
104  @Nullable
105  private ItemStack parseItemStack() throws InvalidItemStackStringException {
106 
107  _buffer.setLength(0);
108 
109  if (current() == ',')
110  next();
111 
112  if (!canParse())
113  return null;
114 
115  skipEmpty();
116 
117  ItemStack stack;
118  Material material;
119  short data = 0;
120  int amount = 1;
121  Potion potion;
122  List<ItemMetaValue> metaObjects = new ArrayList<>(15);
123 
124  String rawMaterial = parseMaterial();
125  if (rawMaterial.isEmpty())
126  return null;
127 
128  material = Converters.MATERIAL.convert(rawMaterial);
129  if (material == null) {
130  throw new InvalidItemStackStringException(_itemString);
131  }
132 
133  if (current() == ':') {
134  next();
135  data = parseData();
136  }
137 
138  if (current() == ';') {
139  next();
140  amount = parseAmount();
141  }
142 
143  if (current() == '{') {
144  next();
145  parseMeta(metaObjects);
146  }
147 
148  // handle special cases
149  if (material == Material.POTION && (potion = Converters.POTION.convert(data)) != null) {
150  stack = potion.toItemStack(amount);
151  }
152  else {
153  // set stack data
154  stack = new ItemStack(material, amount);
155  stack.setDurability(data);
156  }
157 
158  for (ItemMetaValue meta : metaObjects) {
159 
160  IItemMetaHandler handler = Nucleus.getItemMetaHandlers().getHandler(meta.getName());
161  if (handler == null)
162  continue;
163 
164  handler.apply(stack, meta);
165  }
166 
167  return stack;
168  }
