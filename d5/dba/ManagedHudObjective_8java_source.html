190  private void fillLinesTo(int index) {
191 
192  boolean isUpdated = false;
193 
194  for (int i=_lines.size(); i <= index; i++) {
195 
196  Score score = _objective.getScore(getInvisiblePrefix(i));
197  score.setScore(0);
198 
199  _lines.add(new Line(BLANK_TEXT, score));

203 
204  if (!isUpdated)

208  Line line = _lines.get(i);

251  void update() {
252 
253  int currentRefreshRate = text.getRefreshRate();
254 
255  if (currentRefreshRate == refreshRate && nextUpdate > System.currentTimeMillis())
256  return;
257 
258  refreshRate = currentRefreshRate;
259 
260  int lineIndex = score.getScore();
261 
262  String t = getInvisiblePrefix(reverseIndex(lineIndex)) + text.nextText();
263  if (previousText != null && previousText.equals(t))
264  return;
265 
266  previousText = t;
267 
268  if (t.length() > 40)
269  t = TextUtils.truncate(t, 40);
270 
271  score.getScoreboard().resetScores(score.getEntry());
272 
273  score = _objective.getScore(t);
274  score.setScore(lineIndex);
275 
276  nextUpdate = System.currentTimeMillis() + (refreshRate > 0
277  ? refreshRate * 50
278  : 1000);
279  }
