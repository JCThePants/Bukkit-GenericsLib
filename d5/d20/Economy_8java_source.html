128  public static IFutureResult<IEconomyTransaction> transfer(

130 
131  PreCon.notNull(giverPlayerId);

136  IAccount fromAccount = getProvider().getAccount(giverPlayerId);

178  public static IFutureResult<IEconomyTransaction> transfer(
179  IAccount fromAccount, IAccount toAccount, double amount, ICurrency currency) {
180 
181  PreCon.notNull(fromAccount);
182  PreCon.notNull(toAccount);
183  PreCon.positiveNumber(amount);
184  PreCon.notNull(currency);
185 
186  IEconomyTransaction transaction = createTransaction();
187 
188  if (!transaction.withdraw(fromAccount, amount, currency))
189  return FutureResultAgent.errorResult(null, "Insufficient funds");
190 
191  if (!transaction.deposit(toAccount, amount, currency))
192  return FutureResultAgent.errorResult(null, "Unable to deposit.");
193 
194  return transaction.commit();
195  }

425  public static IFutureResult<IEconomyTransaction> depositOrWithdraw(
426  IAccount account, double amount, ICurrency currency) {
427 
428  PreCon.notNull(account);
429  PreCon.notNull(currency);
430 
431  IEconomyTransaction transaction = getProvider().createTransaction();
432 
433  if (amount >= 0) {
434  if (!transaction.deposit(account, amount, currency))
435  return FutureResultAgent.errorResult(null, "Failed to deposit.");
436 
437  } else if (amount < 0) {
438 
439  if (!transaction.withdraw(account, -amount, currency))
440  return FutureResultAgent.errorResult(null, "Insufficient funds.");
441  }
442 
443  return transaction.commit();
444  }
