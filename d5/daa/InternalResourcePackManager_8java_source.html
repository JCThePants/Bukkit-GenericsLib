324  private void handleStatusEvent(Player player, PlayerResourcePackStatusEvent.Status bukkitStatus) {
325  PlayerResourcePacks packs = get(player);
326  if (packs == null)
327  return;
328 
329  final ResourcePackStatus status = ResourcePackStatus.fromBukkit(bukkitStatus);
330  ResourcePackStatus current = packs.getStatus();
331 
332  if ((status == ResourcePackStatus.ACCEPTED
333  || status == ResourcePackStatus.DECLINED)
334  && current != ResourcePackStatus.PENDING
335  && current != ResourcePackStatus.NO_RESOURCE) {
336  return;
337  }
338 
339  if ((status == ResourcePackStatus.FAILED
340  || status == ResourcePackStatus.SUCCESS)
341  && (current != ResourcePackStatus.ACCEPTED
342  && current != ResourcePackStatus.DECLINED)) {
343  return;
344  }
345 
346  if (status == ResourcePackStatus.SUCCESS
347  && current != ResourcePackStatus.ACCEPTED) {
348  return;
349  }
350 
351  packs.setStatus(status);
352  }

354  private void handleRequiredEvent(MissingRequiredResourcePackEvent event) {
355 
356  Player player = event.getPlayer();
357  CharSequence message = event.getMessage();
358 
359  switch (event.getAction()) {
360  case RELOCATE:
361  Teleporter.teleport(player, event.getRelocation());
362  // fall through
363  case IGNORE:
364  if (message != null) {
365  NucMsg.tell(player, message);
366  }
367  break;
368  case KICK:
369  if (message == null) {
370  message = NucLang.get(_RESOURCE_PACK_REQUIRED, event.getWorld().getName());
371  }
372  player.kickPlayer(message.toString());
373  break;
374  }
375 
376  META.set(player, HANDLING_REQUIRED_META_NAME, null);
377  }
