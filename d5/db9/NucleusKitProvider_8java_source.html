73  @Override
74  public IKitContext pluginContext(Plugin plugin) {
75  PreCon.notNull(plugin);
76 
77  if (plugin.equals(_globalContext.getPlugin()))
78  return _globalContext;
79 
80  NucleusKitContext context = _contexts.get(plugin);
81  if (context == null) {
82 
83  IDataNode dataNode = DataStorage.get(plugin, new DataPath("nucleus.kits"));
84  dataNode.load();
85 
86  context = new NucleusKitContext(plugin, dataNode);
87  _contexts.put(plugin, context);
88  }
89 
90  return context;
91  }
