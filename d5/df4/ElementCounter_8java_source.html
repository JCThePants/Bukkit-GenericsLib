354  private int modifyCount(E element, int amount) {
355  PreCon.notNull(element);
356 
357  ElementCount<E> counter;
358 
359  counter = _current != null && _current.equals(element)
360  ? _currentCounter
361  : _countMap.get(element);
362 
363  // Check if item is in counter
364  if (counter == null) {
365  if (_policy == RemovalPolicy.REMOVE && amount <= 0) {
366  return 0;
367  }
368  else {
369  counter = new ElementCount<>(_policy, element, amount);
370  _countMap.put(element, counter);
371  }
372  }
373  // modify count
374  else {
375  counter.increment(amount);
376  }
377 
378  // check if the item needs to be removed
379  if (_policy == RemovalPolicy.REMOVE && counter.count <= 0) {
380  _current = null;
381  _currentCounter = null;
382  _countMap.remove(element);
383  return 0;
384  }
385 
386  // place value in cache
387  _current = counter.element;
388  _currentCounter = counter;
389 
390  return counter.count;
391  }
