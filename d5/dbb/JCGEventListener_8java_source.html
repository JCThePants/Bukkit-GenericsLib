149  @EventHandler(priority=EventPriority.NORMAL, ignoreCancelled = true)
150  private void onPlayerInteract(PlayerInteractEvent event) {
151 
152  if (!event.hasBlock()) {
153  return;
154  }
155 
156  Block clicked = event.getClickedBlock();
157 
158  // Signs
159  if (clicked != null && (clicked.getType() == Material.WALL_SIGN
160  || clicked.getType() == Material.SIGN_POST)) {
161 
162  Action action = event.getAction();
163  BlockState signState = clicked.getState();
164 
165  if (action != Action.LEFT_CLICK_BLOCK && action != Action.RIGHT_CLICK_BLOCK)
166  return;
167 
168  if (!(signState instanceof Sign))
169  return;
170 
171  Sign sign = (Sign)signState;
172  SignInteractEvent signInteractEvent = new SignInteractEvent(event, sign);
173  Nucleus.getEventManager().callBukkit(this, signInteractEvent);
174  }
175  }

177  @EventHandler(priority=EventPriority.NORMAL, ignoreCancelled = true)
178  private void onInventoryClick(InventoryClickEvent event) {
179 
180  HumanEntity entity = event.getWhoClicked();
181 
182  if (!(entity instanceof Player))
183  return;
184 
185  final Player p = (Player)entity;
186 
187  Inventory inventory = event.getInventory();
188  if (!(inventory instanceof AnvilInventory))
189  return;
190 
191  final AnvilInventory anvilInventory = (AnvilInventory)inventory;
192  InventoryView view = event.getView();
193  int rawSlot = event.getRawSlot();
194 
195  if (rawSlot != view.convertSlot(rawSlot) || rawSlot != 2)
196  return;
197 
198  ItemStack resultItem = anvilInventory.getItem(2);
199  if (resultItem == null)
200  return;
201 
202  ItemStack slot1 = anvilInventory.getItem(0);
203  ItemStack slot2 = anvilInventory.getItem(1);
204 
205  ItemStack slot1Clone = slot1 == null ? null : slot1.clone();
206  ItemStack slot2Clone = slot2 == null ? null : slot2.clone();
207 
208  // check for rename
209  String originalName = slot1 != null
210  ? ItemStackUtils.getDisplayName(slot1, DisplayNameOption.OPTIONAL)
211  : null;
212 
213  String newName = ItemStackUtils.getDisplayName(resultItem, DisplayNameOption.OPTIONAL);
214 
215  if (newName != null && !newName.equals(originalName)) {
216 
217  AnvilRenameItemEvent renameEvent = new AnvilRenameItemEvent(
218  p, anvilInventory, resultItem, newName, originalName);
219 
220  Nucleus.getEventManager().callBukkit(this, renameEvent);
221 
222  if (renameEvent.isCancelled()) {
223  cancelAnvilEvent(p, anvilInventory, slot1Clone, slot2Clone);
224  return;
225  }
226 
227  ItemStackUtils.setDisplayName(resultItem, renameEvent.getNewName());
228  }
229 
230  if (slot2 == null || slot2.getType() == Material.AIR)
231  return;
232 
233  if (slot2.getType() == Material.ENCHANTED_BOOK) {
234 
235  AnvilEnchantItemEvent enchantEvent = new AnvilEnchantItemEvent(p, anvilInventory);
236  Nucleus.getEventManager().callBukkit(this, enchantEvent);
237 
238  if (enchantEvent.isCancelled()) {
239  cancelAnvilEvent(p, anvilInventory, slot1Clone, slot2Clone);
240  }
241  }
242  else {
243 
244  AnvilRepairItemEvent repairEvent = new AnvilRepairItemEvent(p, anvilInventory, resultItem);
245  Nucleus.getEventManager().callBukkit(this, repairEvent);
246 
247  if (repairEvent.isCancelled()) {
248  cancelAnvilEvent(p, anvilInventory, slot1Clone, slot2Clone);
249  }
250  }
251  }

253  // detect cancelled thrown item and re-insert into inventory
254  // so that it does not become invisible in the players inventory.
255  @EventHandler(priority = EventPriority.MONITOR, ignoreCancelled = false)
256  private void onItemThrow(PlayerInteractEvent event) {
257 
258  if (!event.isCancelled())
259  return;
260 
261  Player player = event.getPlayer();
262 
263  if (event.getAction() != Action.RIGHT_CLICK_BLOCK &&
264  event.getAction() != Action.RIGHT_CLICK_AIR) {
265  return;
266  }
267 
268  ItemStack thrown = player.getItemInHand();
269  Material material = thrown.getType();
270 
271  if (!Materials.isThrowable(material))
272  return;
273 
274  if (material == Material.POTION && !PotionUtils.isSplash(thrown.getDurability()))
275  return;
276 
277  // set same item in hand, causes packets to be sent to client
278  // "reminding" it that there is still an item in the players hand.
279  player.setItemInHand(thrown);
280  }
281 
282  private void cancelAnvilEvent(final Player player, final Inventory inventory,
283  final ItemStack original1, final ItemStack original2) {
284 
285  player.setItemOnCursor(null);
286 
287  Scheduler.runTaskLater(Nucleus.getPlugin(), new Runnable() {
288  @Override
289  public void run() {
290  inventory.setItem(0, original1);
291  inventory.setItem(1, original2);
292  inventory.setItem(2, null);
293  player.setItemOnCursor(null);
294  }
295  });
296  }
