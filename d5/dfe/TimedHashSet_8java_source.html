521  private void cleanup() {
522 
523  if (_expireMap.isEmpty())
524  return;
525 
526  // prevent cleanup from running too often
527  if (_nextCleanup > System.currentTimeMillis())
528  return;
529 
530  _nextCleanup = System.currentTimeMillis() + MIN_CLEANUP_INTERVAL_MS;
531 
532  _cleanupList.addAll(_expireMap.entrySet());
533 
534  for (Entry<E, ExpireInfo> entry : _cleanupList) {
535  if (!entry.getValue().isExpired())
536  continue;
537 
538  _expireMap.remove(entry.getKey());
539  onLifespanEnd(entry.getKey());
540  _expirePool.recycle(entry.getValue());
541  }
542 
543  _cleanupList.clear();
544  }

546  private void startJanitor() {
547  if (_janitor != null)
548  return;
549 
550  _janitor = Scheduler.runTaskRepeatAsync(Nucleus.getPlugin(),
551  JANITOR_INITIAL_DELAY_TICKS, JANITOR_INTERVAL_TICKS, new Runnable() {

554 
555  @Override
556  public void run() {
557 
558  synchronized (_instances) {
559  sets.addAll(_instances.keySet());
560  }
561 
562  for (TimedHashSet set : sets) {
563 
564  // remove instance if owning plugin is not enabled
565  if (!set.getPlugin().isEnabled()) {
566 
567  synchronized (_instances) {
568  _instances.remove(set);
569  }
570  continue;
571  }
572 
573  // cleanup instance
574  synchronized (set._sync) {
575  set.cleanup();
576  }
577  }

579  sets.clear();
580  }
581  });

598  private final class Itr implements Iterator<E> {
599 
600  Iterator<Entry<E, ExpireInfo>> iterator
601  = _expireMap.entrySet().iterator();

603  Entry<E, ExpireInfo> peek;
604  boolean invokedHasNext;
605  boolean invokedNext;

607  @Override
608  public boolean hasNext() {
609 
610  IteratorWrapper.assertIteratorLock(_sync);
611 
612  invokedHasNext = true;
613 
614  if (!iterator.hasNext())
615  return false;
616 
617  while (iterator.hasNext()) {
618  peek = iterator.next();
619 
620  if (peek.getValue().isExpired()) {
621  iterator.remove();
622  }
623  else {
624  return true;
625  }
626  }
627 
628  return false;
629 
630  }

632  @Override
633  public E next() {
634 
635  IteratorWrapper.assertIteratorLock(_sync);
636 
637  if (!invokedHasNext)
638  throw new IllegalStateException("Cannot invoke 'next' before invoking 'hasNext'.");
639 
640  invokedHasNext = false;
641  invokedNext = true;
642 
643  if (peek == null)
644  hasNext();
645 
646  if (peek == null)
647  throw new NoSuchElementException();
648 
649  Entry<E, ExpireInfo> n = peek;
650  peek = null;
651  return n.getKey();
652  }

656 
657  IteratorWrapper.assertIteratorLock(_sync);
658 
659  if (!invokedNext)
660  throw new IllegalStateException("Cannot invoke 'remove' before invoking 'next'");

663  }
