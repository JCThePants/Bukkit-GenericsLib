264 
265  if (player.playerConnection == null)
266  return null;
267 
268  if (player.activeContainer != player.defaultContainer) {
269  // fire INVENTORY_CLOSE if one already open
270  player.playerConnection.a(new PacketPlayInCloseWindow(player.activeContainer.windowId));

273  String windowType = CraftContainer.getNotchInventoryType(InventoryType.ANVIL);
274 
275  Container container = new ContainerAnvil(player.inventory, player.world, position, player);
276  container = CraftEventFactory.callInventoryOpenEvent(player, container);
277  if(container == null)
278  return null;

281

283  int size = 0;

285  player.playerConnection.sendPacket(new PacketPlayOutOpenWindow(
286  container.windowId, windowType, new ChatComponentText(title), size));
287 
288  player.activeContainer = container;
289  player.activeContainer.addSlotListener(player);
290  container.checkReachable = false;
