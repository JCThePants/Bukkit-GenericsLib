88  Set<EntityType> results = EnumSet.noneOf(EntityType.class);
89 
90  for (EntityTypeProperty property : properties) {

93  Iterator<EntityType> iterator = results.iterator();

95  while (iterator.hasNext()) {

331  public static void register(EntityType type, EntityTypeProperty... properties) {
332  PreCon.notNull(type);
333 
334  if (properties == null || properties.length == 0)
335  return;
336 
337  // make sure a default property type is not being registered.
338  for (EntityTypeProperty property : properties) {
339  if (property.isDefaultProperty()) {
340  throw new IllegalArgumentException("Cannot register a default property type.");
341  }
342  }
343 
344  add(type, properties);
345  }

347  private static void add(EntityType type, EntityTypeProperty... properties) {
348  _properties.putAll(type, Arrays.asList(properties));
