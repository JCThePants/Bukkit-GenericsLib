89  public static List<IScript> loadScripts(Plugin plugin,
90  ScriptEngineManager engineManager,
91  File scriptFolder,
92  @Nullable File exclude,
93  DirectoryTraversal traversal,
94  IScriptFactory scriptFactory) {
95  PreCon.notNull(plugin);
96  PreCon.notNull(scriptFolder);
97  PreCon.isValid(scriptFolder.isDirectory());
98 
99  if (!scriptFolder.exists())
100  return new ArrayList<>(0);
101 
102  List<File> scriptFiles = FileUtils.getFiles(scriptFolder, traversal);
103  List<IScript> result = new ArrayList<>(scriptFiles.size());
104 
105  for (File file : scriptFiles) {
106 
107  if (exclude != null && file.getAbsolutePath().startsWith(exclude.getAbsolutePath())) {
108  continue;
109  }
110 
111  String type = getScriptType(file);
112  if (type == null || type.isEmpty())
113  continue;
114 
115  if (engineManager.getEngineByExtension(type) == null)
116  continue;
117 
118  String name = getScriptName(scriptFolder, file);
119 
120  try {
121 
122  BufferedReader reader = new BufferedReader(new FileReader(file));
123 
124  StringBuilder buffer = new StringBuilder(2000);
125  String line;
126 
127  while ((line = reader.readLine()) != null) {
128  buffer.append(line);
129  buffer.append('\n');
130  }
131 
132  reader.close();
133 
134  IScript script = scriptFactory.create(name, file, type, buffer.toString());
135  if (script != null)
136  result.add(script);
137 
138  } catch (IOException e) {
139  e.printStackTrace();
140  }
141  }
142 
143  return result;
144  }

240  public static Result<Object> eval(ScriptEngine engine, ScriptContext context, IScript script) {
241 
242  File file = script.getFile();
243 
244  String filename = file != null ? file.getName() : "<unknown>";
245  String engineName = engine.getFactory().getEngineName();
246  boolean isNashorn = engineName.contains("Nashorn");
247 
248  engine.put(ScriptEngine.FILENAME, filename);
249  context.setAttribute(ScriptEngine.FILENAME, filename, ScriptContext.ENGINE_SCOPE);
250 
251  Object result;
252 
253  try {
254  if (isNashorn) {
255 
256  context.setAttribute("script", script.getScript(), ScriptContext.ENGINE_SCOPE);
257  context.setAttribute("scriptName", filename, ScriptContext.ENGINE_SCOPE);
258 
259  // evaluate script into Nashorn
260  result = engine.eval("load({ script: script, name: scriptName})", context);
261  }
262  else {
263  // evaluate script
264  result = engine.eval(script.getScript(), context);
265  }
266 
267  return new Result<Object>(true, result);
268 
269  } catch (Throwable e) {
270  e.printStackTrace();
271  return new Result<Object>(false);
272  }
