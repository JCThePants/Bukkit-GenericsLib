147  @Override
148  public PathableResult isPathable(AStarNode from, AStarNode to) {
149  PreCon.notNull(from);
150  PreCon.notNull(to);
151  PreCon.isValid(from.isAdjacent(to), "'to' node should be adjacent to 'from' node");
152 
153  IAStarNodeContainer container = from.getContext().getContainer();
154 
155  // check if candidate is already closed
156  if (container.isClosed(to))
157  return PathableResult.INVALID_POINT;
158 
159  ICoords3Di parent = from.getCoords();
160  ICoords3Di candidate = to.getCoords();
161 
162  int x = candidate.getX() - parent.getX();
163  int y = candidate.getY() - parent.getY();
164  int z = candidate.getZ() - parent.getZ();
165 
166  Material material = _world
167  .getBlockAt(candidate.getX(), candidate.getY(), candidate.getZ())
168  .getType();
169 
170  // check candidate to see if its valid for the entity to stand on
171  if (!Materials.isSurface(material) ||
172  !hasRoomForEntity(to, getDoorPathMode())) {
173 
174  // invalidate column if material is NOT transparent
175  if (!Materials.isTransparent(material)) {
176  return PathableResult.INVALID_COLUMN;
177  }
178 
179  return PathableResult.INVALID_POINT;
180  }
181 
182  // Check for diagonal obstruction
183  if (x != 0 && z != 0 && y >= 0) {
184  AStarNode diagX = from.getRelative(x, y, 0);
185  AStarNode diagZ = from.getRelative(0, y, z);
186 
187  boolean isXValid = hasRoomForEntity(diagX, getDoorPathMode());
188  boolean isZValid = hasRoomForEntity(diagZ, getDoorPathMode());
189 
190  // prevent walking through diagonal walls
191  if(y == 0 && !isXValid && !isZValid ) {
192  return PathableResult.INVALID_COLUMN;
193  }
194 
195  // prevent walking through corners
196  if (!isXValid || !isZValid) {
197  return PathableResult.INVALID_COLUMN;
198  }
199  }
200 
201  return PathableResult.VALID;
202  }

207  protected boolean hasRoomForEntity(AStarNode node, DoorPathMode doorMode) {
208 
209  ICoords3Di coords = node.getCoords();
210  Block block = _world
211  .getBlockAt(coords.getX(), coords.getY(), coords.getZ());
212 
213  int height = (int)Math.ceil(getEntityHeight());
214 
215  // check head room
216  for (int i=0; i < height; i++) {
217 
218  Block above = block.getRelative(0, i + 1, 0);
219 
220  if (doorMode != DoorPathMode.IGNORE_OPEN) {
221 
222  BlockState state = above.getState();
223 
224  // check if block is an open doorway
225  if (state.getData() instanceof Openable) {
226 
227  Openable openable = (Openable) state.getData();
228 
229  if (doorMode == DoorPathMode.IGNORE_CLOSED)
230  continue;
231 
232  Block bottomDoor = block.getRelative(0, i, 0);
233  BlockState bottomState = bottomDoor.getState();
234 
235  // check the lower door block instead
236  if (bottomState.getData() instanceof Openable) {
237  openable = (Openable) bottomState.getData();
238  }
239 
240  if (openable.isOpen()) {
241  continue;
242  }
243  }
244  }
245 
246  // make sure block is transparent
247  if (!Materials.isTransparent(above.getType())) {
248  return false;
249  }
250  }
251 
252  return true;
253  }
