73  public void open(@Nullable AStarNode parent, AStarNode node) {
74  PreCon.notNull(node);
75 
76  if (parent != null && !parent.getContext().equals(node.getContext()))
77  throw new IllegalArgumentException("parent and node arguments are from different contexts.");
78 
79  IAStarScore score = node.getContext().getAstar().getExaminer().getScore(parent, node);
80 
81  AStarNode open = _open.remove(node);
82  if (open != null) {
83 
84  IAStarScore openScore = open.getScore();
85  float newG = score.getG();
86 
87  if (openScore == null || newG < openScore.getG()) {
88  node.setParent(parent, score);
89  _open.put(node, node);
90  }
91  else {
92  _open.put(open, open);
93  }
94  }
95  else {
96  node.setParent(parent, score);
97  _open.put(node, node);
98  }
99  }
