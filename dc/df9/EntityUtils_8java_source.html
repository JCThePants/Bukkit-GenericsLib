300  @Nullable
301  public static Entity getClosestEntity(Location sourceLocation, double radius,
302  @Nullable final IValidator<Entity> validator) {
303  PreCon.notNull(sourceLocation);
304  PreCon.positiveNumber(radius);
305 
306  if (sourceLocation.getWorld() == null)
307  return null;
308 
309  int xRadius = (int)Math.ceil(radius / 16);
310  int zRadius = xRadius;
311 
312  int xSource = sourceLocation.getBlockX();
313  if (xSource < radius || xSource > 16 - radius)
314  xRadius++;
315 
316  int zSource = sourceLocation.getBlockZ();
317  if (zSource < radius || zSource > 16 - radius)
318  zRadius++;
319 
320  Entity closest = null;
321  double closestDistanceSq = 0;
322 
323  for (int x = 0; x <= xRadius; x++) {
324  for (int z = 0; z <= zRadius; z++) {
325 
326  Chunk chunk = sourceLocation.getWorld().getChunkAt(x, z);
327  Entity[] entities = chunk.getEntities();
328 
329  for (Entity entity : entities) {
330 
331  if (validator != null && !validator.isValid(entity))
332  continue;
333 
334  double distance = entity.getLocation(NEARBY_ENTITY_LOCATION)
335  .distanceSquared(sourceLocation);
336  if (distance > radius * radius)
337  continue;
338 
339  if (closest == null ||
340  distance < closestDistanceSq) {
341  closestDistanceSq = distance;
342  closest = entity;
343  }
344  }
345  }
346  }
347 
348  return closest;
349  }

634  public static List<Entity> getNearbyEntities(Location location,
635  double radiusX, double radiusY, double radiusZ,
636  @Nullable IValidator<Entity> validator) {
637  PreCon.notNull(location);
638  PreCon.positiveNumber(radiusX);
639  PreCon.positiveNumber(radiusY);
640  PreCon.positiveNumber(radiusZ);
641 
642  List<Entity> results = new ArrayList<>(15);
643 
644  World world = location.getWorld();
645  if (world == null)
646  return results;
647 
648  int xStart = getStartChunk(location.getX(), radiusX);
649  int xEnd = getEndChunk(location.getX(), radiusX);
650  int zStart = getStartChunk(location.getZ(), radiusZ);
651  int zEnd = getEndChunk(location.getZ(), radiusZ);
652 
653  for (int x = xStart; x <= xEnd; x++) {
654  for (int z = zStart; z <= zEnd; z++) {
655 
656  Chunk chunk = world.getChunkAt(x, z);
657  if (!chunk.isLoaded())
658  chunk.load();
659 
660  Entity[] entities = chunk.getEntities();
661 
662  for (Entity entity : entities) {
663 
664  if (!entity.isValid())
665  continue;
666 
667  if (!entity.getWorld().equals(world))
668  continue;
669 
670  Location entityLocation = getEntityLocation(entity, NEARBY_ENTITY_LOCATION);
671 
672  if (!LocationUtils.isInRange(location, entityLocation, radiusX, radiusY, radiusZ))
673  continue;
674 
675  if (validator != null && !validator.isValid(entity))
676  continue;
677 
678  results.add(entity);
679  }
680  }
681  }
