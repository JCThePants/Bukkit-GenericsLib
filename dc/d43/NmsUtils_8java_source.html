91  public static boolean enforceNmsVersion(Plugin plugin, String... compatibleVersions) {
92 
93  if (!isVersionCompatible(compatibleVersions)) {
94 
95  if (plugin instanceof NucleusPlugin) {
96 
97  // allow server administrator the option to disable NMS version enforcement
98  boolean enforceNmsVersion = ((NucleusPlugin) plugin).getDataNode()
99  .getBoolean("enforce-nms-version", true);
100 
101  if (!enforceNmsVersion) {
102  NucMsg.warning(plugin, "Plugin {0} is not compatible with NMS version {1}.",
103  plugin.getName(), getNmsVersion());
104 
105  NucMsg.warning(plugin, "enforce-nms-version has been set to " +
106  "false in the plugins config.");
107  return true;
108  }
109  }

113 
114  NucMsg.warning(plugin, "Disabling {0} because it's not compatible with NMS version {1}",
115  plugin.getName(), getNmsVersion());

117  }

119  return true;

243  // load the the craft package version from NucleusFramework's config
244  // or detect the version.
245  private static void loadPackageVersion() {
246  _version = Nucleus.getPlugin().getDataNode().getString("package-version");
247  if (_version == null) {
248 
249  Class<? extends Server> serverClass = Bukkit.getServer().getClass();
250 
251  Matcher versionMatcher = PATTERN_VERSION.matcher(serverClass.getName());
252 
253  if (versionMatcher.matches()) {
254  _version = versionMatcher.group(1);
255 
256  NucMsg.info("Reflection: CraftBukkit version found: {0}", _version);
257  }
258  }
259  else {
260  NucMsg.info("Reflection: Using craft version from NucleusFramework config: {0}", _version);
261  }
262 
263  if (_version == null) {
264  NucMsg.severe("Failed to find CraftBukkit version for reflection purposes.");
265  }
266  }
