150  @Override
151  public boolean tellNoSpam(CommandSender sender, Integer ticks, LineWrapping lineWrapping,
152  Object message, Object... params) {
153  PreCon.notNull(sender);
154  PreCon.positiveNumber(ticks);
155  PreCon.notNull(lineWrapping);
156  PreCon.notNull(message);
157  PreCon.notNull(params);
158 
159  if (!(sender instanceof Player)) {
160  // no spam not applied to console
161  return tell(sender, message, params);
162  }
163 
164  Player player = (Player)sender;
165 
166  IChatMessage msg = TextUtils.format(message, params);
167  String str = msg.toString();
168 
169  TimedHashSet<String> recent = _noSpamCache.get(player.getUniqueId());
170  if (recent == null) {
171  recent = new TimedHashSet<String>(_plugin, 20, 140);
172  _noSpamCache.put(player.getUniqueId(), recent);
173  }
174 
175  if (recent.contains(str, ticks, TimeScale.TICKS))
176  return false;
177 
178  recent.add(str, ticks, TimeScale.TICKS);
179 
180  return tell(player, lineWrapping, msg);
181  }
