238  @Override
239  public void run() {
240 
241  if (isEnded()) {
242  cancelTask();
243  return;
244  }
245 
246  synchronized (_sync) {
247 
248  boolean isStart = true;
249  int completed = 0;
250 
251  for (int y = isStart ? _yCurrent : _yStart; y <= _yEnd; y++) {
252 
253  for (int x = isStart ? _xCurrent : _xStart; x <= _xEnd; x++) {
254 
255  for (int z = isStart ? _zCurrent : _zStart; z <= _zEnd; z++) {
256 
257  // check for end of segment
258  if (_segmentSize > 0 && completed >= _segmentSize) {
259  _segmentsCompleted ++;
260 
261  _xCurrent = x;
262  _yCurrent = y;
263  _zCurrent = z;
264 
265  onSegmentEnd(x, y, z);
266 
267  // end segment
268  return;
269  }
270 
271  // check if this is the start of the segment
272  if (isStart) {
273  onSegmentStart(x, y, z);
274  }
275 
276 
277  isStart = false;
278 
279  onIterateItem(x, y, z);
280 
281  if (!isRunning()) {
282  cancelTask();
283  return;
284  }
285 
286  _iterations++;
287  completed++;
288  }
289  }
290  }
