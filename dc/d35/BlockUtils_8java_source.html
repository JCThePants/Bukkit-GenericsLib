103  public static <T extends Collection<Block>> T getAdjacent(Block block, T output,
104  MaterialProperty... properties) {
105  PreCon.notNull(block);
106 
107  if (output instanceof ArrayList)
108  ((ArrayList) output).ensureCapacity(output.size() + 6);
109 
110  for (BlockFace face : DIRECTIONS_3D) {
111  if (block.getY() == 254 && face == BlockFace.UP)
112  continue;
113 
114  if (block.getY() == 0 && face == BlockFace.DOWN)
115  continue;
116 
117  Block adjacent = block.getRelative(face);
118 
119  if (properties.length > 0) {
120  boolean isValid = true;
121  for (MaterialProperty property : properties) {
122  if (!Materials.hasProperty(adjacent.getType(), property)) {
123  isValid = false;
124  break;
125  }
126  }
127 
128  if (!isValid)
129  continue;
130  }
131 
132  output.add(adjacent);
133  }
134  return output;
135  }

344  public static void dropRemoveBlock(final Location location, final int removeDelayTicks) {
345  PreCon.notNull(location);
346 
347  final BlockState startBlock = location.getBlock().getState();
348 
349  location.getBlock().setType(Material.AIR);
350 
351  // schedule task so the block has a chance to turn into air
352  Scheduler.runTaskLater(Nucleus.getPlugin(), new Runnable() {
353 
354  @Override
355  public void run() {
356 
357  // spawn a falling block
358  final FallingBlock fallBlock = location.getWorld().spawnFallingBlock(
359  location, startBlock.getType(), startBlock.getData().getData());
360 
361  // schedule the removal of the block
362  Scheduler.runTaskLater(Nucleus.getPlugin(), removeDelayTicks, new Runnable() {
363 
364  @Override
365  public void run() {
366  if (fallBlock.isOnGround() || fallBlock.isDead()) {
367 
368  // find the fallen block
369  Location landedLoc = LocationUtils.findSurfaceBelow(location);
370  if (landedLoc == null)
371  return;
372 
373  if (fallBlock.getFallDistance() == 0.0) {
374 
375  Block landedBlock = landedLoc.getBlock();
376  if (landedBlock.getType() != startBlock.getType())
377  return;
378 
379  landedBlock.setType(Material.AIR);
380  }
381  } else {
382  // remove the falling block if it has not
383  // become a block yet.
384  fallBlock.remove();
385  }
386 
387  }
388 
389  });
390  }
391  });
392  }
