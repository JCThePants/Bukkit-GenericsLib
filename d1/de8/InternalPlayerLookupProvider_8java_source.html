157  @Override
158  public IFutureResult<Collection<UUID>> searchNames(String searchText, int maxResults) {
159 
160  Deque<PlayerData> unsorted = new ArrayDeque<>(15);
161  String lower = searchText.toLowerCase();
162 
163  IDataNode playersNode = getPlayerData();
164  for (IDataNode node : playersNode) {
165  String name = node.getString("name");
166  if (name == null)
167  continue;
168 
169  if (name.toLowerCase().contains(lower))
170  unsorted.add(new PlayerData(name, TextUtils.parseUUID(node.getName())));
171  }
172 
173  Collection<PlayerData> sorted = CollectionUtils.textSearch(
174  unsorted, searchText, new CollectionUtils.ISearchTextGetter<PlayerData>() {
175  @Override
176  public String getText(PlayerData element) {
177  return element.name;
178  }
179  });
180 
181  List<UUID> result = new ArrayList<>(Math.min(sorted.size(), maxResults));
182 
183  int count = 0;
184  for (PlayerData data : sorted) {
185  if (count >= maxResults)
186  break;
187  result.add(data.id);
188  count++;
189  }
190 
191  return new FutureResultAgent<Collection<UUID>>().success(result);
192  }

197  private void setPlayerName(UUID playerId, String name) {
198  PreCon.notNull(playerId);
199  PreCon.notNullOrEmpty(name);
200 
201  Date loginDate = new Date();
202 
203  IDataNode data = getPlayerData(playerId);
204 
205  synchronized (_dataSync) {
206  if (data == null) {
207  data = getPlayerData().getNode(playerId.toString());
208 
209  data.set("first", loginDate);
210  }
211 
212  int count = data.getInteger("count", 0);
213 
214  data.set("last", loginDate);
215  data.set("count", count + 1);
216  }
217 
218  String currentName = getPlayerName(playerId);
219 
220  if (name.equals(currentName))
221  return;
222 
223  synchronized (_dataSync) {
224  data.set("name", name);
225  }
226 
227  data.save();
228  }

233  if (_nameData == null) {
