496  protected void setCoords(Location p1, Location p2) {
497  PreCon.notNull(p1);
498  PreCon.notNull(p2);
499 
500  World p1World = p1.getWorld();
501  World p2World = p2.getWorld();
502 
503  if (p1World != null && !p1World.equals(p2World) ||
504  p2World != null && !p2World.equals(p1World)) {
505  throw new IllegalArgumentException("Both region points must be from the same world.");
506  }
507 
508  double lowerX = Math.min(p1.getX(), p2.getX());
509  double lowerY = Math.min(p1.getY(), p2.getY());
510  double lowerZ = Math.min(p1.getZ(), p2.getZ());
511 
512  double upperX = Math.max(p1.getX(), p2.getX());
513  double upperY = Math.max(p1.getY(), p2.getY());
514  double upperZ = Math.max(p1.getZ(), p2.getZ());
515 
516  String worldName = null;
517 
518  if (p1 instanceof SyncLocation) {
519  worldName = ((SyncLocation) p1).getWorldName();
520  }
521  else if (p1.getWorld() != null) {
522  worldName = p1.getWorld().getName();
523  }
524 
525  _p1 = new SyncLocation(worldName, p1.getX(), p1.getY(), p1.getZ(), 0F, 0F);
526  _p2 = new SyncLocation(worldName, p2.getX(), p2.getY(), p2.getZ(), 0F, 0F);
527  _lowerPoint = new SyncLocation(worldName, lowerX, lowerY, lowerZ, 0F, 0F);
528  _upperPoint = new SyncLocation(worldName, upperX, upperY, upperZ, 0F, 0F);
529  updateMath();
530  }

532  /*
533  * Update region math variables
534  */
535  protected void updateMath() {
536 
537  if (_p1 == null || _p2 == null)
538  return;
539 
540  synchronized (_sync) {
541 
542  _startX = Math.min(_p1.getBlockX(), _p2.getBlockX());
543  _startY = Math.min(_p1.getBlockY(), _p2.getBlockY());
544  _startZ = Math.min(_p1.getBlockZ(), _p2.getBlockZ());
545 
546  _endX = Math.max(_p1.getBlockX(), _p2.getBlockX());
547  _endY = Math.max(_p1.getBlockY(), _p2.getBlockY());
548  _endZ = Math.max(_p1.getBlockZ(), _p2.getBlockZ());
549 
550  _xWidth = (int)Math.abs(_p1.getX() - _p2.getX());
551  _zWidth = (int)Math.abs(_p1.getZ() - _p2.getZ());
552  _yHeight = (int)Math.abs(_p1.getY() - _p2.getY());
553 
554  _xBlockWidth = _xWidth + 1;
555  _zBlockWidth = _zWidth + 1;
556  _yBlockHeight = _yHeight + 1;
557 
558  _volume = _xWidth * _zWidth * _yHeight;
559 
560  if (getWorld() != null) {
561  double xCenter = (double) _startX + (_xBlockWidth / 2.0D);
562  double yCenter = (double) _startY + (_yBlockHeight / 2.0D);
563  double zCenter = (double) _startZ + (_zBlockWidth / 2.0D);
564 
565  _center = new Location(getWorld(), xCenter, yCenter, zCenter);
566  }
567 
568  _chunkX = Math.min(_p1.getBlockX(), _p2.getBlockX()) == _p1.getBlockX()
569  ? (int)Math.floor(_p1.getBlockX() / 16.0D)
570  : (int)Math.floor(_p2.getBlockX() / 16.0D);
571 
572  _chunkZ = Math.min(_p1.getBlockZ(), _p2.getBlockZ()) == _p1.getBlockZ()
573  ? (int)Math.floor(_p1.getBlockZ() / 16.0D)
574  : (int)Math.floor(_p2.getBlockZ() / 16.0D);
575 
576  int chunkEndX = Math.max(_p1.getBlockX(), _p2.getBlockX()) == _p1.getBlockX()
577  ? (int)Math.floor(_p1.getBlockX() / 16.0D)
578  : (int)Math.floor(_p2.getBlockX() / 16.0D);
579 
580  int chunkEndZ = Math.max(_p1.getBlockZ(), _p2.getBlockZ()) == _p1.getBlockZ()
581  ? (int)Math.floor(_p1.getBlockZ() / 16.0D)
582  : (int)Math.floor(_p2.getBlockZ() / 16.0D);
583 
584  _chunkXWidth = chunkEndX - _chunkX + 1;
585  _chunkZWidth = chunkEndZ - _chunkZ + 1;
586 
587  _chunks = null;
588 
589  _flatness = RegionShape.CUBOID;
590 
591  if (getXBlockWidth() == 1) { // west/east
592 
593  if (getZBlockWidth() == 1) { // north/south
594 
595  _flatness = getYBlockHeight() == 1
596  ? RegionShape.POINT
597  : RegionShape.VERTICAL_LINE;
598  }
599  else {
600  _flatness = getYBlockHeight() == 1
601  ? RegionShape.NORTH_SOUTH_LINE
602  : RegionShape.FLAT_WEST_EAST;
603  }
604  }
605  else if (getZBlockWidth() == 1) { // north/south
606  _flatness = getYBlockHeight() == 1
607  ? RegionShape.WEST_EAST_LINE
608  : RegionShape.FLAT_NORTH_SOUTH;
609  }
610  else if (getYBlockHeight() == 1) {
611  _flatness = RegionShape.FLAT_HORIZONTAL;
612  }
613  }
614  }
