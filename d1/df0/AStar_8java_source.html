128  public AStarResult search(ICoords3Di start, ICoords3Di destination, IAStarNodeContainer container) {
129  PreCon.notNull(start);
130  PreCon.notNull(destination);
131 
132  if (Coords3Di.distanceSquared(start, destination) > getRangeSquared()) {
133  return new AStarResult(AStarResultStatus.RANGE_EXCEEDED);
134  }
135 
136  container.reset();
137 
138  AStarContext context = new AStarContext(this, container, start, destination);
139  AStarNode startNode = new AStarNode(context, start);
140 
141  container.open(null, startNode);
142 
143  openAdjacent(startNode);
144 
145  AStarNode current = null;
146 
147  // iterate until destination is found
148  // or unable to continue
149 
150  long iterations = 0;
151  long maxIterations = getMaxIterations();
152 
153  while (getExaminer().canSearch(context)) {
154 
155  // get and close best candidate for next node to path to.
156  current = container.closeBest();
157  if (current == null || getExaminer().isDestination(current))
158  break;
159 
160  // open valid adjacent nodes
161  openAdjacent(current);
162 
163  iterations ++;
164 
165  // do not exceed max iterations
166  if (maxIterations > 0 && iterations >= maxIterations) {
167  return new AStarResult(AStarResultStatus.ITERATIONS_EXCEEDED);
168  }
169  }
170 
171  return new AStarResult(context, current);
172  }

179  protected void openAdjacent(AStarNode node) {
180 
181  // column validations, work from top down, skip columns that are false
182  boolean[][] columns = new boolean[][] {
183  { true, true, true },
184  { true, false, true },
185  { true, true, true }
186  };
187 
188  int dropHeight = -getMaxDropHeight();
189 
190  for (byte y = 1; y >= dropHeight; y--) {
191  for (byte x = -1; x <= 1; x++) {
192  for (byte z = -1; z <= 1; z++) {
193 
194  openCandidate(node, x, y, z, columns);
195  }
196  }
197  }

209  protected void openCandidate(AStarNode parent, int x, int y, int z, boolean[][] columns) {
210 
211  if (!columns[x + 1][z + 1])
212  return;
213 
214  // get instance of candidate node
215  AStarNode candidate = parent.getRelative(x, y, z);
216  AStarContext context = parent.getContext();
217 
218  // check range
219  if (Coords3Di.distanceSquared(candidate.getCoords(), context.getStartCoords()) > getRangeSquared()) {
220  columns[x + 1][z + 1] = false;
221  return;
222  }
223 
224  PathableResult result = getExaminer().isPathable(parent, candidate);
225 
226  switch (result) {
227  case VALID:
228  context.getContainer().open(parent, candidate);
229  // fall through, don't check columns where a valid node was already found.
230  case INVALID_COLUMN:
231  columns[x + 1][z + 1] = false;
232  // fall through
233  case INVALID_POINT:
234  break;
235  }
236  }
