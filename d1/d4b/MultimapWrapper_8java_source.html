398  private Collection<V> replaceValuesSource(K k, Iterable<? extends V> iterable) {
399  Collection<V> removed = map().removeAll(k);
400  ElementCounter<V> counter;
401 
402  counter = new ElementCounter<>(RemovalPolicy.KEEP_COUNTING, removed);
403 
404  map().putAll(k, iterable);
405 
406  counter.subtractAll(map().get(k));
407 
408  for (ElementCount<V> elmCount : counter) {
409 
410  V agent = elmCount.getElement();
411 
412  for (int i=0; i < Math.abs(elmCount.getCount()); i++) {
413  if (elmCount.getCount() > 0) {
414  onRemove(k, agent);
415  }
416  else {
417  onPut(k, agent);
418  }
419  }
420  }
421 
422  return removed;
423  }
