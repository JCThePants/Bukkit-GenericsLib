92  @Override
93  public void execute(final CommandSender sender, ICommandArguments args) throws CommandException {
94 
95  CommandException.checkNotConsole(getPlugin(), this, sender);
96 
97  Player player = (Player)sender;
98 
99  final String receiverName = args.getString("playerName");
100  final double amount = args.getDouble("amount");
101  final String myBankName = args.isDefaultValue("myBank") ? "" : args.getString("mybank");
102  final String bankName = args.getString("bank");
103 
104  UUID receiverId = PlayerUtils.getPlayerId(receiverName);
105  if (receiverId == null)
106  throw new CommandException(NucLang.get(_PLAYER_NOT_FOUND, receiverName));
107 
108  // Get command senders account
109  IAccount myAccount;
110  if (myBankName.isEmpty()) {
111 
112  myAccount = Economy.getAccount(player.getUniqueId());
113  if (myAccount == null)
114  throw new CommandException(NucLang.get(_NO_SEND_ACCOUNT));
115 
116  } else {
117 
118  IBank myBank = Economy.getBank(myBankName);
119  if (myBank == null)
120  throw new CommandException(NucLang.get(_BANK_NOT_FOUND, myBankName));
121 
122  myAccount = myBank.getAccount(player.getUniqueId());
123 
124  if (myAccount == null)
125  throw new CommandException(NucLang.get(_NO_SEND_ACCOUNT_AT_BANK, myBank.getName()));
126  }
127 
128  // Get receivers account
129  IAccount receiverAccount;
130  if (bankName.isEmpty()) {
131 
132  receiverAccount = Economy.getAccount(receiverId);
133  if (receiverAccount == null)
134  throw new CommandException(NucLang.get(_NO_RECEIVE_ACCOUNT, receiverName));
135 
136  }
137  else {
138 
139  IBank bank = Economy.getBank(bankName);
140  if (bank == null)
141  throw new CommandException(NucLang.get(_BANK_NOT_FOUND, myBankName));
142 
143  receiverAccount = bank.getAccount(receiverId);
144  if (receiverAccount == null)
145  throw new CommandException(NucLang.get(_NO_RECEIVE_ACCOUNT_AT_BANK, receiverName, bank.getName()));
146  }
147 
148  // check command senders balance
149  double balance = myAccount.getBalance();
150  if (balance < amount)
151  throw new CommandException(NucLang.get(_NOT_ENOUGH_MONEY, balance));
152 
153  final IAccount finalMyAccount = myAccount;
154 
155  // transfer money
156  Economy.transfer(myAccount, receiverAccount, amount)
157  .onError(new FutureResultSubscriber<IEconomyTransaction>() {
158  @Override
159  public void on(Result<IEconomyTransaction> result) {
160  tellError(sender, NucLang.get(_FAILED));
161  }
162  })
163  .onSuccess(new FutureResultSubscriber<IEconomyTransaction>() {
164  @Override
165  public void on(Result<IEconomyTransaction> result) {
166  tellSuccess(sender, NucLang.get(_SUCCESS, Economy.getCurrency().format(amount),
167  receiverName, finalMyAccount.getBalance()));
168  }
169  });
170  }
171 }
