163  public static <T extends Collection<IChunkCoords>> T getChunksInRadius(Location location,
164  int blockRadius, T output) {
165  PreCon.notNull(location, "location");
166  PreCon.notNull(location.getWorld(), "location world");
167  PreCon.positiveNumber(blockRadius, "blockRadius");
168  PreCon.notNull(output, "output");
169 
170  int chunkRadius = (blockRadius >> 4) + 1;
171  int blockRadiusSquared = blockRadius * blockRadius;
172 
173  int chunkX = location.getBlockX() >> 4;
174  int chunkZ = location.getBlockZ() >> 4;
175 
176  CacheCoords cache = CACHE_COORDS.get();
177  MutableChunkCoords chunkCoords = new MutableChunkCoords();
178 
179  for (int x=-chunkRadius; x <= chunkRadius; x++) {
180  for (int z=-chunkRadius; z <= chunkRadius; z++) {
181 
182  setChunkCoords(chunkCoords, location.getWorld(), chunkX + x, chunkZ + z);
183 
184  if (x == 0 && z == 0) {
185  output.add(chunkCoords);
186  chunkCoords = new MutableChunkCoords();
187  continue;
188  }
189 
190  double distance = getChunkDistanceSquared(location, chunkCoords, cache);
191  if (distance <= blockRadiusSquared) {
192  output.add(chunkCoords);
193  chunkCoords = new MutableChunkCoords();
194  }
195  }
196  }
197 
198  return output;
199  }

326  private static int getChunkDistanceSquared(Location source, ICoords2Di chunkCoords,
327  CacheCoords cache) {
328 
329  int sourceChunkX = source.getBlockX() >> 4;
330  int sourceChunkZ = source.getBlockZ() >> 4;
331  int chunkX = chunkCoords.getX();
332  int chunkZ = chunkCoords.getZ();
333 
334  if (chunkX == sourceChunkX && chunkZ == sourceChunkZ) {
335  return 0;
336  }
337 
338  int faceX = sourceChunkX > chunkX ? 15 : sourceChunkX == chunkX ? -1 : 0;
339  int faceZ = sourceChunkZ > chunkZ ? 15 : sourceChunkZ == chunkZ ? -1 : 0;
340 
341  int arraySize = 0;
342 
343  if (faceX != -1) {
344  for (int z = 0; z < 16; z++, arraySize++) {
345  cache.chunksX[arraySize] = faceX;
346  cache.chunksZ[arraySize] = z;
347  }
348  }
349 
350  if (faceZ != -1) {
351  for (int x = 0; x < 16; x++, arraySize++) {
352  cache.chunksX[arraySize] = x;
353  cache.chunksZ[arraySize] = faceZ;
354  }
355  }
356 
357  int closest = -1;
358  for (int i=0; i < arraySize; i++) {
359  int x = cache.chunksX[i];
360  int z = cache.chunksZ[i];
361 
362  int deltaX = (chunkX + x) - source.getBlockX();
363  int deltaZ = (chunkZ + z) - source.getBlockZ();
364  int distance = deltaX * deltaX + deltaZ * deltaZ;
365 
366  if (closest == -1 || distance < closest) {
367  closest = distance;
368  }
369  }
370 
371  return closest;
372  }

402  if (faceZ != -1) {
403  for (int x = 0; x < 16; x++, arraySize++) {
404  cache.chunksX[arraySize] = x;
405  cache.chunksZ[arraySize] = faceZ;
406  }
407  }
408 
409  int closest = -1;
410  int closestX = 0;
411  int closestZ = 0;
412  for (int i=0; i < arraySize; i++) {
413  int x = cache.chunksX[i];
414  int z = cache.chunksZ[i];
415
