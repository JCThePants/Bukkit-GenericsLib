78  @Override
79  protected void lazyLoadFriends(UUID playerId) {
80 
81  synchronized (_sync) {
82  if (getFriendsMap().containsKey(playerId))
83  return;
84  }
85 
86  IDataNode dataNode = DataStorage.get(Nucleus.getPlugin(),
87  getProvider().getDataPath(
88  getName() + '.'
89  + (Bukkit.getOnlineMode() ? "" : "-offline.")
90  + playerId.toString()));
91  dataNode.load();
92 
93  synchronized (_sync) {
94  Map<UUID, IFriend> friends = new HashMap<>(dataNode.size());
95 
96  for (IDataNode friendNode : dataNode) {
97  UUID friendId = TextUtils.parseUUID(friendNode.getName());
98  if (friendId == null) {
99  NucMsg.debug("Failed to load friend node because it could not be parsed " +
100  "to a UUID: {0}", friendNode.getName());
101  continue;
102  }
103 
104  NucleusFriend friend = new NucleusFriend(this, playerId, friendId, friendNode);
105  friends.put(friendId, friend);
106  }
107 
108  getFriendsMap().put(playerId, new FriendInfo(playerId, dataNode, friends));
109  }
110  }
