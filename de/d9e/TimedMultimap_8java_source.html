469  private boolean isExpiredRemove(Object key) {
470  //noinspection SuspiciousMethodCalls
471  ExpireInfo<K> info = _expireMap.get(key);
472  if (info == null)
473  return true;
474 
475  if (info.isExpired()) {
476 
477  //noinspection SuspiciousMethodCalls
478  Collection<V> collection = _map.removeAll(key);
479 
480  //noinspection SuspiciousMethodCalls
481  _expireMap.remove(key);
482 
483  onLifespanEnd(info.key, collection);
484 
485  info.recycle();
486 
487  return true;
488  }
489 
490  return false;
491  }

493  private void cleanup() {
494 
495  if (_expireMap.isEmpty())
496  return;
497 
498  // prevent cleanup from running too often
499  if (_nextCleanup > System.currentTimeMillis())
500  return;
501 
502  _nextCleanup = System.currentTimeMillis() + MIN_CLEANUP_INTERVAL_MS;
503 
504  _cleanupList.addAll(_expireMap.entrySet());
505 
506  // iterate over entry set for items to remove
507  for (Map.Entry<K, ExpireInfo<K>> entry : _cleanupList) {
508 
509  // check if entry is expired
510  if (!entry.getValue().isExpired())
511  continue;
512 
513  _expireMap.remove(entry.getKey());
514 
515  Collection<V> removed = _map.removeAll(entry.getKey());
516 
517  // notify subscribers
518  onLifespanEnd(entry.getKey(), removed);
519 
520  entry.getValue().recycle();
521  }
522 
523  _cleanupList.clear();
524  }

550  private void startJanitor() {
551 
552  if (_janitor != null)
553  return;
554 
555  synchronized (_instances) {
556 
557  if (_janitor != null)
558  return;
559 
560  _janitor = Scheduler.runTaskRepeatAsync(
561  Nucleus.getPlugin(), JANITOR_INITIAL_DELAY_TICKS, JANITOR_INTERVAL_TICKS, new Runnable() {
562 
563  List<TimedMultimap> maps = new ArrayList<>(20);
564 
565  @Override
566  public void run() {
567 
568  synchronized (_instances) {
569  maps.addAll(_instances.keySet());
570  }
571 
572  for (TimedMultimap map : maps) {
573 
574  if (!map._plugin.isEnabled()) {
575  synchronized (_instances) {
576  _instances.remove(map);
577  }
578  continue;
579  }
580 
581  synchronized (map._sync) {
582  map.cleanup();
583  }
584  }
585 
586  maps.clear();
587  }
588  });
589  }

603  ExpireInfo<K> set(K key, long lifespan, TimeScale timeScale) {
604  this.key = key;
605  this.expires = System.currentTimeMillis() + (lifespan * timeScale.getTimeFactor());
606  this.isRecycled = false;
607  return this;
608  }
609 
610  void recycle() {
611  if (this.isRecycled)
612  return;
613 
614  this.isRecycled = true;
615  this.key = null;
616  this.parent._expirePool.recycle(this);
617  }
618 
619  boolean isExpired() {
620  return System.currentTimeMillis() >= expires;
621  }
622  }
