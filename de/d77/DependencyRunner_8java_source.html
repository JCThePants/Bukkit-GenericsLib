210  /*
211  * Periodically checks dependencies and runs dependant runnables whose
212  * dependency conditions have been met. Cancels self when timeout is reached.
213  */
214  private class Watcher extends TaskHandler {
215 
216  @Override
217  public void run() {
218  Deque<T> runners = new ArrayDeque<>(_runnables);
219 
220  while (!runners.isEmpty()) {
221 
222  T runnable = runners.remove();
223 
224  try {
225  DependencyStatus status = runnable.getDependencyStatus();
226  if (status == DependencyStatus.READY) {
227 
228 
229  runnable.run();
230  _runnables.remove(runnable);
231 
232  }
233  }
234  catch (Exception e) {
235  e.printStackTrace();
236  }
237  }
238 
239  if (_runnables.isEmpty() || new Date().compareTo(_endTime) >= 0)
240  cancelTask();
241  }
242 
243  /*
244  * Invoked when the watcher is finished.
245  */
246  @Override
247  protected void onCancel() {
248 
249  // run if only optional dependencies are not ready
250  Deque<T> runners = new ArrayDeque<>(_runnables);
251 
252  while (!runners.isEmpty()) {
253 
254  T runnable = runners.remove();
255 
256  try {
257  DependencyStatus status = runnable.getDependencyStatus();
258  if (status == DependencyStatus.READY ||
259  status == DependencyStatus.REQUIRED_READY) {
260 
261  runnable.run();
262  _runnables.remove(runnable);
263  }
264  }
265  catch (Exception e) {
266  e.printStackTrace();
267  }
268  }
269 
270  while (!_onFinished.isEmpty()) {
271  IFinishHandler<T> onFinished = _onFinished.remove();
272  onFinished.onFinish(new ArrayList<T>(_runnables));
273  }
274  }
275  }
