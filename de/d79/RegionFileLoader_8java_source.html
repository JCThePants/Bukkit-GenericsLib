85  @Override
86  public IFuture load(LoadType loadType, final LoadSpeed speed, IRegionFileData data) throws IOException {
87  PreCon.notNull(loadType);
88  PreCon.notNull(data);
89 
90  final IRegion region = getRegion();
91 
92  Collection<IChunkCoords> chunks = region.getChunkCoords();
93  final QueueProject restoreProject = new QueueProject(region.getPlugin());
94 
95  if (chunks.size() == 0)
96  return restoreProject.cancel("No chunks to read.");
97 
98  Boolean isReading = region.getMeta().get(META_IS_READING);
99 
100  if (isReading != null && isReading)
101  return restoreProject.cancel("Region is still being read.");
102 
103  if (!canRead())
104  return restoreProject.cancel("Region files not found.");
105 
106  region.getMeta().setKey(META_IS_READING, true);
107 
108  for (IChunkCoords chunk : chunks) {
109 
110  QueueProject chunkProject = new QueueProject(region.getPlugin());
111  RegionChunkFileLoader loader = new RegionChunkFileLoader(region, chunk);
112 
113  // add load task to chunk project
114  loader.loadInProject(
115  getChunkFile(region, chunk.getX(), chunk.getZ(), false),
116  chunkProject, loadType, data);
117 
118  restoreProject.addTask(chunkProject);
119  }
120 
121  final FutureAgent agent = new FutureAgent();
122 
123  runProject(restoreProject, speed);
124 
125  restoreProject.getResult().onStatus(new FutureSubscriber() {
126  @Override
127  public void on(FutureStatus status, @Nullable CharSequence message) {
128 
129  switch (status) {
130  case ERROR:
131  agent.error(message);
132  break;
133  case CANCEL:
134  agent.cancel(message);
135  break;
136  case SUCCESS:
137  if (restoreProject.getTasks().isEmpty()) {
138  agent.success(message);
139  } else {
140  runProject(restoreProject, speed);
141  }
142  break;
143  }
144  }
145  });
146 
147  return agent.getFuture().onStatus(new FutureSubscriber() {
148  @Override
149  public void on(FutureStatus status, @Nullable CharSequence message) {
150  region.getMeta().setKey(META_IS_READING, null);
151  }
152  });
153  }

162  break;
