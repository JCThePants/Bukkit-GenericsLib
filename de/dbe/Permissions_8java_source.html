97  public static IPermission register(final String permissionName,
98  final PermissionDefault value, boolean registerParents) {
99  PreCon.notNullOrEmpty(permissionName);
100  PreCon.notNull(value);
101 
102  if (permissionName.endsWith(".*") || !registerParents)
103  return getProvider().register(permissionName, value);
104 
105  final IPermission result = getProvider().register(permissionName, value);
106 
107  runBatchOperation(new Runnable() {
108  @Override
109  public void run() {
110 
111  String[] components = TextUtils.PATTERN_DOT.split(permissionName);
112 
113  StringBuilder buffer = new StringBuilder(permissionName.length() + 2);
114  List<IPermission> permParents = new ArrayList<>(components.length);
115 
116  for (int i=0; i < components.length; i++) {
117 
118  buffer.append(components[i]);
119 
120  if (i < components.length - 1) {
121  String name = buffer.toString();
122  IPermission permission = get(name + ".*");
123  if (permission == null) {
124  permission = register(name + ".*", PermissionDefault.FALSE, false);
125  }
126 
127  for (IPermission permParent : permParents) {
128  permission.addParent(permParent, true);
129  }
130  permParents.add(permission);
131 
132  buffer.append('.');
133  }
134  }
135 
136  for (IPermission permParent : permParents) {
137  result.addParent(permParent, true);
138  }
139 
140 
141  IPermission allPermission = Permissions.get(permissionName + ".*");
142  if (allPermission == null) {
143  allPermission = Permissions.register(permissionName + ".*", PermissionDefault.FALSE);
144  }
145  result.addParent(allPermission, true);
146  }
147  });
148 
149  return result;
150  }

517  public static void fixPermissionGroups(Plugin plugin, OfflinePlayer player,
518  Collection<IPermissionGroup> groups) {
519  PreCon.notNull(plugin);
520  PreCon.notNull(player);
521  PreCon.notNull(groups);
522 
523  if (!hasGroupSupport())
524  return;
525 
526  if (!(player instanceof Player))
527  return;
528 
529  for (IPermissionGroup group : groups) {
530  boolean canAssign = group.canAssign(player);
531 
532  if (canAssign) {
533  addGroup(plugin, player, group.getName());
534  } else {
535  removeGroup(plugin, player, group.getName());
536  }
537  }
