95  synchronized (_sync) {

111  PreCon.notNull(player);
112  PreCon.notNull(selection);
113 
114  if (!selection.isDefined())
115  return false;
116 
117  assert selection.getP1() != null;
118  assert selection.getP2() != null;
119 
120  synchronized (_sync) {
121  _p1Selections.put(player.getUniqueId(), selection.getP1().clone());
122  _p2Selections.put(player.getUniqueId(), selection.getP2().clone());
123  }
124 
125  return true;

130  @EventHandler(priority = EventPriority.HIGHEST, ignoreCancelled = false)
131  private void onPlayerInteract(PlayerInteractEvent event) {
132 
133  if (!event.hasBlock())
134  return;
135 
136  Player player = event.getPlayer();
137 
138  if (!hasPermission(player))
139  return;
140 
141  ItemStack item = player.getInventory().getItemInHand();
142  if (item == null)
143  return;
144 
145  if (item.getType() != Material.WOOD_AXE)
146  return;
147 
148  if (event.getAction() == Action.LEFT_CLICK_BLOCK) {
149  Location p1 = event.getClickedBlock().getLocation();
150  setP1(player, p1);
151  event.setCancelled(true);
152  }
153  else if (event.getAction() == Action.RIGHT_CLICK_BLOCK) {
154  Location p2 = event.getClickedBlock().getLocation();
155  setP2(player, p2);
156  event.setCancelled(true);
157  }
158  }
159 
160  @EventHandler
161  private void onCommand(PlayerCommandPreprocessEvent event) {
162 
163  Player player = event.getPlayer();
164 
165  if (!hasPermission(player))
166  return;
167 
168  String message = event.getMessage();
169 
170  switch (message) {
171  case "//wand":
172  player.getInventory().addItem(new ItemStack(Material.WOOD_AXE));
173  event.setCancelled(true);
174  break;
175  case "//pos1":
176  setP1(player, player.getLocation());
177  event.setCancelled(true);
178  break;
179  case "//pos2":
180  setP2(player, player.getLocation());
181  event.setCancelled(true);
182  break;
183  }
184  }
185  }
