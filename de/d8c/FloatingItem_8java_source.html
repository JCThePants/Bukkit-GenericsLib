235  @Override
236  public boolean spawn(Location location) {
237  PreCon.notNull(location);
238 
239  if (_isDisposed)
240  throw new RuntimeException("Cannot spawn a disposed item.");
241 
242  location = location.clone(); // clone: prevent external changes from affecting the location
243 
244  FloatingItemSpawnEvent event = new FloatingItemSpawnEvent(this);
245 
246  Nucleus.getEventManager().callBukkit(this, event);
247 
248  if (event.isCancelled())
249  return false;
250 
251  if (!despawn())
252  return false;
253 
254  _isSpawned = true;
255 
256  _currentLocation = location;
257 
258  // get corrected location
259  final Location spawnLocation = _isCentered
260  ? LocationUtils.getCenteredLocation(location, CENTERED_LOCATION).add(0, 0.5, 0) // add y 0.5 to prevent falling through surface block
261  : LocationUtils.add(location, 0, 0.5, 0);
262 
263  if (!location.getChunk().isLoaded()) {
264  _listener.registerPendingSpawn(this);
265  return true;
266  }
267 
268  // spawn item entity
269  Entity entity = location.getWorld().dropItem(spawnLocation, _item.clone());
270  _trackedEntity = EntityUtils.trackEntity(entity);
271  _entityId = entity.getUniqueId();
272  entity.setVelocity(new Vector(0, 0, 0));
273 
274  // register entity
275  _listener.register(this);
276 
277  // prevent stack merging
278  Item item = (Item)entity;
279  ItemMeta meta = item.getItemStack().getItemMeta();
280  meta.setDisplayName(_entityId.toString());
281  item.getItemStack().setItemMeta(meta);
282 
283  _dataNode.set("location", location);
284  _dataNode.set("is-spawned", true);
285  _dataNode.set("entity-id", _entityId);
286  _dataNode.save();
287 
288  _agents.update("onSpawn", entity);
289 
290  return true;
291  }

403  private void loadSettings() {
404 
405  _canPickup = _dataNode.getBoolean("can-pickup", _canPickup);
406  _isCentered = _dataNode.getBoolean("is-centered", _isCentered);
407  _respawnTimeSeconds = _dataNode.getInteger("respawn-time-seconds", _respawnTimeSeconds);
408  _isSpawned = _dataNode.getBoolean("is-spawned", _isSpawned);
409  _currentLocation = _dataNode.getLocation("location", _currentLocation);
410  _entityId = _dataNode.getUUID("entity-id", _entityId);
411 
412  if (_currentLocation != null && _entityId != null) {
413 
414  // Find the entity from the chunk of the stored location
415  Entity entity = EntityUtils.getEntityByUUID(_currentLocation.getChunk(), _entityId);
416 
417  if (entity != null) {
418  _trackedEntity = EntityUtils.trackEntity(entity);
419  }
420 
421  if (entity != null && !_isSpawned) {
422  despawn();
423  } else if (entity == null && _isSpawned) {
424  spawn(_currentLocation);
425  } else if (entity != null) {
426  _listener.register(this);
427  }
428 
429  }
430  else {
431  _isSpawned = false;
432  }
