98  public static ItemStack getItemStack(Object object) {
99 
100  ItemStack result = null;
101 
102  if (object instanceof ItemStack) {
103  result = (ItemStack) object;
104  }
105  else if (object instanceof String) {
106 
107  String str = (String)object;
108 
109  MaterialData data = NamedMaterialData.get(str);
110  if (data != null) {
111  result = data.toItemStack();
112  }
113  else {
114  try {
115  ItemStack[] itemStacks = parse(str);
116 
117  if (itemStacks != null && itemStacks.length > 0)
118  result = itemStacks[0];
119  } catch (InvalidItemStackStringException ignore) {}
120  }
121  }
122  else if (object instanceof Material) {
123  result = new ItemStack((Material)object);
124  }
125  else if (object instanceof MaterialData) {
126  result = ((MaterialData)object).toItemStack();
127  }
128 
129  if (result != null && result.getAmount() <= 0) {
130  result.setAmount(1);
131  }
132 
133  return result;
134  }

233  @Nullable
234  public static String getDisplayName(ItemStack stack, DisplayNameOption nameResult) {
235  PreCon.notNull(stack);
236 
237  ItemMeta meta = stack.getItemMeta();
238  if (meta == null || !meta.hasDisplayName()) {
239 
240  switch (nameResult) {
241  case REQUIRED:
242 
243  if (meta instanceof PotionMeta) {
244  Potion potion = new Potion(stack.getDurability());
245  return PotionNames.getSimple(potion);
246  }
247 
248  String materialName = NamedMaterialData.getAlternate(stack.getData());
249  if (materialName == null) {
250  materialName = stack.getType().name().toLowerCase();
251  }
252 
253  String spaced = PATTERN_REPLACE_UNDERSCORE.matcher(materialName).replaceAll(" ");
254  return TextUtils.titleCase(spaced);
255 
256  case OPTIONAL:
257  // fall through
258 
259  default:
260  return null;
261  }
262  }
263 
264  return meta.getDisplayName();
265  }
