92  PreCon.notNullOrEmpty(permissionName);
93 
94  IPermission perm = Permissions.get(permissionName);
95  if (perm == null)
96  return false;
97 
98  PermissionAttachment attachment = _transient.get(player.getUniqueId());
99  if (attachment == null) {
100  attachment = player.addAttachment(plugin);
101  _transient.put(player.getUniqueId(), attachment);
102  }

104  assert perm.getHandle() != null;

132  public boolean add(Plugin plugin, OfflinePlayer player, String permissionName) {
133  PreCon.notNull(plugin);
134  PreCon.notNull(player);

136 
137  String pid = player.getUniqueId().toString() + '.' + plugin.getName();

139  List<String> permissions = _dataNode.getStringList(pid, null);

190  @EventHandler(priority= EventPriority.LOW)
191  private void onPlayerJoin(PlayerJoinEvent event) {
192 
193  Player p = event.getPlayer();
194 
195  // give permissions
196 
197  // get players permission data node
198  IDataNode playerNode = _provider._dataNode.getNode(p.getUniqueId().toString());
199 
200  // get the names of the plugins that have set permissions on the player.
201  for (IDataNode node : playerNode) {
202 
203  // make sure the plugin is loaded.
204  Plugin plugin = Bukkit.getPluginManager().getPlugin(node.getName());
205  if (plugin == null)
206  continue;
207 
208  // get the list of permissions the player has.
209  List<String> permissions = playerNode.getStringList(plugin.getName(), null);
210 
211  if (permissions == null || permissions.isEmpty())
212  continue;
213 
214  // add permissions to player
215  for (String permission : permissions) {
216  // add permissions as transient to prevent re-saving them
217  _provider.addTransient(plugin, p, permission);
218  }
219  }
220  }
