171  public void updatePlayerLocation(Player player, LeaveRegionReason reason) {
172  PreCon.notNull(player);
173  PreCon.notNull(reason);
174 
175  // ignore NPC's
176  if (Npcs.isNpc(player))
177  return;
178 
179  UUID playerId = player.getUniqueId();
180 
181 
182  EventOrderedRegions<IRegion> regions = forgetPlayer(playerId);
183  if (regions == null)
184  return;
185 
186  synchronized (this) {
187  Iterator<IRegion> iterator = regions.iterator(PriorityType.LEAVE);
188  while (iterator.hasNext()) {
189  IRegion region = iterator.next();
190 
191  if (region.isEventListener())
192  region.getEventListener().onPlayerLeave(player, reason);
193  }
194  }
195 
196  if (reason == LeaveRegionReason.QUIT_SERVER) {
197 
198  synchronized (this) {
199  // re-pool location cache
200  PlayerLocationCache cache = _playerLocationCache.remove(player.getUniqueId());
201  if (cache != null) {
202  _pools.recycle(cache);
203  }
204  }
205  }
206  else {
207  clearPlayerLocations(player.getUniqueId());
208  }
209  }

266  /*
267  * Repeating task that pre-processes and filters player movement data before
268  * adding it to the async player watchers queue.
269  */
270  private final class QueueFiller implements Runnable {
271 
272  @Override
273  public void run() {
274 
275  // do not run while async watcher is running
276  if (_isAsyncWatcherRunning)
277  return;
278 
279  // call queued region events
280  if (!_eventCaller.queue.isEmpty()) {
281  _eventCaller.run();
282  }
283 
284  // get worlds where listener regions exist
285  List<World> worlds = new ArrayList<World>(_manager.getListenerWorlds().getElements());
286 
287  // get players in worlds with regions
288  for (World world : worlds) {
289 
290  if (world == null)
291  continue;
292 
293  // get players currently in world
294  List<Player> players = world.getPlayers();
295  if (players == null || players.isEmpty())
296  continue;
297 
298  // process and add players to wp list
299  for (Player player : players) {
300 
301  // do not process NPC's
302  if (Npcs.isNpc(player))
303  continue;
304 
305  // get locations that the player was recorded in between watcher cycles
306  PlayerLocationCache locations = getPlayerLocations(player.getUniqueId());
307 
308  synchronized (InternalPlayerWatcher.this) {
309  // skip if there are no locations recorded
310  if (locations.getCheckedOut().size() == 0)
311  continue;
312 
313  WorldPlayer worldPlayer = new WorldPlayer(player, locations.getCheckedOut());
314  _watcherAsync.queue.add(worldPlayer);
315  }
316  }
317  }
318 
319  // end if there are no players to process
320  if (_watcherAsync.queue.isEmpty())
321  return;
322 
323  // let the async watcher run
324  _isAsyncWatcherRunning = true;
325  }
326  }

328  /*
329  * Async portion of the player watcher
330  */
331  private class PlayerWatcherAsync implements Runnable {
332 
333  final Queue<WorldPlayer> queue = new ArrayDeque<>(Bukkit.getMaxPlayers());
334  // Temporary queue to hold regions a player has possibly entered.
335  // Because "enter" is processed before "leave", the new "enter" event
336  // will not be processed unless regions in question are processed after
337  // the "leave" event has removed the region from the players region cache.
338  // The enter queue is used to hold regions that need to be checked again
339  // after "leave" events have been processed.
340  final Queue<IRegion> enter = new ArrayDeque<>(20);
341 
342  @Override
343  public void run() {
344 
345  if (!_isAsyncWatcherRunning)
346  return;
347 
348  // iterate players
349  while (!queue.isEmpty()) {
350 
351  WorldPlayer worldPlayer = queue.remove();
352 
353  UUID playerId = worldPlayer.player.getUniqueId();
354 
355  // get regions the player is in (cached from previous check)
356  EventOrderedRegions<IRegion> cachedRegions;
357 
358  synchronized (InternalPlayerWatcher.this) {
359  cachedRegions = _playerRegionCache.get(playerId);
360  }
361 
362  if (cachedRegions == null) {
363  cachedRegions = new EventOrderedRegions<>(7);
364 
365  synchronized (InternalPlayerWatcher.this) {
366  _playerRegionCache.put(playerId, cachedRegions);
367  }
368  }
369 
370  // iterate locations the player has been since the last check
371  for (CachedLocation location : worldPlayer.locations) {
372 
373  if (location.getReason() == RegionEventReason.JOIN_SERVER) {
374  // do not notify regions until player has moved.
375  // Allows time for player to load resource packs.
376  _joined.add(playerId);
377  continue;
378  }
379 
380  boolean isJoining = _joined.contains(playerId);
381 
382  if (isJoining && location.getReason() != RegionEventReason.MOVE) {
383  // ignore all other reasons until joined player moves
384  continue;
385  }
386 
387  // get regions the player location is in
388  List<IRegion> locationRegions = _manager.getListenerRegions(location, PriorityType.ENTER);
389 
390  RegionEventReason reason = null;
391 
392  // check ENTER regions
393  if (!locationRegions.isEmpty()) {
394 
395  reason = isJoining && _joined.remove(playerId)
396  ? RegionEventReason.JOIN_SERVER
397  : location.getReason();
398 
399  for (IRegion region : locationRegions) {
400  synchronized (InternalPlayerWatcher.this) {
401  // check if player was not previously in region
402  if (cachedRegions.contains(region)) {
403  // add to "enter" queue to verify later
404  enter.offer(region);
405  } else {
406  cachedRegions.add(region);
407  _eventCaller.queue.addLast(new EventInfo(
408  region, worldPlayer.player, reason, true
409  ));
410  }
411  }
412  }
413  }
414 
415  // check LEAVE regions
416  synchronized (InternalPlayerWatcher.this) {
417 
418  if (!cachedRegions.isEmpty()) {
419 
420  // get iterator for regions the player is currently in.
421  Iterator<IRegion> iterator = cachedRegions.iterator(PriorityType.LEAVE);
422 
423  while (iterator.hasNext()) {
424  IRegion region = iterator.next();
425 
426  // check if player was previously in region
427  if (!locationRegions.contains(region) || reason == RegionEventReason.DEAD) {
428  //remove from players cached regions
429  iterator.remove();
430  _eventCaller.queue.addLast(new EventInfo(
431  region, worldPlayer.player, location.getReason(), false
432  ));
433  }
434  }
435  }
436  }
437 
438  // verify "enter" queue
439  while (reason != null && !enter.isEmpty()) {
440  IRegion region = enter.poll();
441 
442  synchronized (InternalPlayerWatcher.this) {
443  // check if player was not previously in region
444  if (!cachedRegions.contains(region)) {
445  cachedRegions.add(region);
446  _eventCaller.queue.addLast(new EventInfo(
447  region, worldPlayer.player, reason, true
448  ));
449  }
450  }
451  }
452  }
453 
454  // recycle player locations so they can be reused
455  worldPlayer.locations.recycle();
456 
457  } // END while(queue.isEmpty)
458 
459  _isAsyncWatcherRunning = false;
460 
461  } // END run()
462  }

500  private static class EventCaller implements Runnable {
501 
502  final Deque<EventInfo> queue = new ArrayDeque<EventInfo>(Bukkit.getMaxPlayers());
503 
504  @Override
505  public void run() {
506 
507  while (!queue.isEmpty()) {
508 
509  EventInfo info = queue.removeFirst();
510  if (info.region.isDisposed())
511  continue;
512 
513  IRegionEventListener listener;
514 
515  try {
516  listener = info.region.getEventListener();
517  if (listener == null) {
518  // print null pointer exception message to console.
519  throw new NullPointerException("Region event listener cannot be null.");
520  }
521  } catch (Throwable e) {
522  e.printStackTrace();
523  continue;
524  }
525 
526  try {
527  if (info.isEntering)
528  listener.onPlayerEnter(info.player, info.reason.getEnterReason());
529  else
530  listener.onPlayerLeave(info.player, info.reason.getLeaveReason());
531  }
532  catch (Throwable e) {
533  e.printStackTrace();
534  }
535  }
536  }
