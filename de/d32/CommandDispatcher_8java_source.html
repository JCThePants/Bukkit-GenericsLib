115  @Override
116  public boolean onCommand(CommandSender sender, Command cmd, String rootName, String[] rootArguments) {
117 
118  RegisteredCommand rootCommand = _rootCommands.getCommand(cmd.getName());
119  if (rootCommand == null)
120  rootCommand = _defaultRoot;
121 
122  rootCommand.getInfo().setCurrentAlias(rootName);
123 
124  CommandParser parser = new CommandParser(rootCommand);
125  ParsedCommand parsed = parser.parseCommand(rootCommand.getCommandCollection(), rootArguments);
126 
127  if (parsed == null) {
128  // command not found
129  NucMsg.tell(getPlugin(), sender, ERROR_TEMPLATE, NucLang.get(_COMMAND_NOT_FOUND, rootName));
130  return true; // finish
131  }
132 
133  RegisteredCommand command = parsed.getCommand();
134  String[] rawArguments = parsed.getArguments();
135 
136  // Check if the player has permissions to run the command
137  if (sender instanceof Player && !Permissions.has((Player)sender, command.getPermission().getName())) {
138  NucMsg.tell(getPlugin(), sender, ERROR_TEMPLATE, NucLang.get(_ACCESS_DENIED));
139  return true;
140  }
141 
142  // make sure the command is visible to the command sender
143  if (command.getCommand() instanceof IVisibleCommand
144  && !((IVisibleCommand) command.getCommand()).isVisible(sender)) {
145  NucMsg.tell(getPlugin(), sender, ERROR_TEMPLATE, NucLang.get(_UNAVAILABLE));
146  return true;
147  }
148 
149  // handle command help, display if the command argument is '?' or 'help'
150  if (isCommandHelp(rawArguments)) {
151 
152  int page = TextUtils.parseInt(
153  ArrayUtils.get(rawArguments, 1, null), 1);
154 
155  command.showHelp(sender, page);
156 
157  return true; // finished
158  }
159 
160  if (isDetailedHelp(rawArguments)) {
161 
162  int page = TextUtils.parseInt(
163  ArrayUtils.get(rawArguments, 1, null), 1);
164 
165  command.showDetailedHelp(sender, page);
166 
167  return true; // finished
168  }
169 
170  // Determine if the command can execute or if it requires sub commands
171  if (!(command.getCommand() instanceof IExecutableCommand)) {
172  NucMsg.tell(getPlugin(), sender,
173  ERROR_TEMPLATE, NucLang.get(_COMMAND_INCOMPLETE,
174  _usageGenerator.generate(command, UsageGenerator.INLINE_HELP)));
175  return true; // finished
176  }
177 
178  // Parse command arguments
179  Arguments arguments;
180 
181  try {
182  arguments = new Arguments(command, rawArguments);
183 
184  } catch (CommandException e) {
185  NucMsg.tell(getPlugin(), sender, ERROR_TEMPLATE, e.getMessage());
186  return true; // finished
187  }
188 
189  // execute the command
190  try {
191  command.execute(sender, arguments);
192  }
193  catch (CommandException e) {
194  NucMsg.tell(getPlugin(), sender, ERROR_TEMPLATE, e.getMessage());
195  }
196 
197  return true;
198  }

200  @Override
201  public List<String> onTabComplete(CommandSender sender, Command cmd, String s, String[] args) {
202 
203  if (args.length == 0)
204  return new ArrayList<>(0);
205 
206  RegisteredCommand rootCommand = _rootCommands.getCommand(cmd.getName());
207  if (rootCommand == null) {
208  rootCommand = _defaultRoot;
209  }
210 
211  CommandParser _parser = new CommandParser(rootCommand);
212  ParsedTabComplete parsed = _parser.parseTabComplete(rootCommand, sender, args);
213 
214  RegisteredCommand command = parsed.getCommand();
215  String[] arguments = parsed.getArguments();
216  List<String> matches = parsed.getMatches();
217 
218  if (command != null) {
219 
220  // give the command the opportunity to modify the list
221  command.onTabComplete(
222  sender,
223  arguments,
224  matches);
225  }
226 
227  // add the command help
228  if (command != null && matches.size() > 1 &&
229  (arguments.length == 0 ||
230  (arguments.length == 1 && arguments[0].isEmpty()))) {
231  parsed.getMatches().add("?");
232  }
233 
234  return parsed.getMatches();

237  @Override
238  public boolean registerCommand(Class<? extends ICommand> commandClass) {
239  PreCon.notNull(commandClass);
240 
241  String rootName = _rootCommands.addCommand(commandClass);
242  if (rootName == null) {
243  _msg.debug("Failed to register command '{0}' possibly because another command with the " +
244  "same name is already registered and no alternative command names were provided.",
245  commandClass.getName());
246  return false;
247  }
248 
249  RegisteredCommand command = _rootCommands.getCommand(rootName);
250  if (command == null)
251  throw new AssertionError();
252 
253  if (!_pluginCommands.contains(rootName)) {
254  _rootCommands.removeAll(command);
255  if (!_defaultRoot.registerCommand(commandClass))
256  return false;
257 
258  command = _defaultRoot.getCommandCollection().getCommand(commandClass);
259  if (command == null)
260  throw new AssertionError();
261  }
262 
263  command.setDispatcher(this, null);
264  return true;
265  }
