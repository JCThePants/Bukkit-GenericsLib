109  public IFuture loadInProject(File file, QueueProject project, LoadType loadType, IRegionFileData builder) {
110  PreCon.notNull(file);
111  PreCon.notNull(loadType);
112  PreCon.notNull(builder);
113 
114  if (isLoading())
115  return project.cancel("Region cannot load because it is already loading.");
116 
117  World world = _region.getWorld();
118 
119  if (world == null) {
120  return project.cancel("Failed to get world while loading region '{0}'.",
121  _region.getName());
122  }
123 
124  Chunk chunk = _region.getWorld().getChunkAt(_coords.getX(), _coords.getZ());
125  if (chunk == null) {
126  return project.cancel("Failed to get chunk ({0}, {1}) in world '{0}' while building region '{1}'.",
127  _coords.getX(), _coords.getZ(), world.getName());
128  }
129 
130  if (!chunk.isLoaded())
131  chunk.load();
132 
133  _isLoading = true;
134 
135  RegionChunkSection section = new RegionChunkSection(_region, _coords);
136  LoadChunkIterator iterator = new LoadChunkIterator(project, file, loadType, builder, 8192,
137  section.getStartChunkX(), section.getStartY(), section.getStartChunkZ(),
138  section.getEndChunkX(), section.getEndY(), section.getEndChunkZ(), chunk);
139 
140  project.addTask(iterator);
141 
142  return iterator.getResult();
143  }

195  @Override
196  protected void onIterateBegin() {
197 
198  try {
199 
200  reader = new BasicByteReader(new FileInputStream(file));
201 
202  // Read restore file version
203  int restoreFileVersion = reader.getInteger();
204 
205  // make sure the file version is correct
206  if (restoreFileVersion != RESTORE_FILE_VERSION &&
207  restoreFileVersion != COMPATIBLE_FILE_VERSION) {
208  cancel("Invalid region file. File version is not compatible.");
209  _isLoading = false;
210  return;
211  }
212 
213  // get name of the region associated with the restore file
214  reader.getString();
215 
216  // get the name of the world the region was saved from.
217  reader.getString();
218 
219 
220  if (restoreFileVersion == COMPATIBLE_FILE_VERSION) {
221  // get the coordinates of the region.
222  reader.getLocation();
223  reader.getLocation();
224  }
225  else {
226  // get the chunk section info
227  reader.deserialize(RegionChunkSection.class);
228  }
229 
230  // get the volume of the region
231  long volume = reader.getLong();
232 
233  // Make sure the volume matches expected volume
234  if (volume != getVolume()) {
235  cancel("Invalid region file. Volume mismatch.");
236  _isLoading = false;
237  }
238  }
239  catch (IOException | InstantiationException e) {
240  handleException(e, "Failed to read header from file for chunk ({0}, {1}).",
241  snapshot.getX(), snapshot.getZ());
242  }
243  }
244 
248  @Override
249  protected void onIterateItem(int x, int y, int z) {
250 
251  Material type;
252  int data;
253  int light;
254  int skylight;
255 
256  try {
257 
258  String typeName = reader.getSmallString();
259  if (typeName == null) {
260  handleException(null, "Failed to read from file for chunk ({0}, {1}). " +
261  "Found a null block type in file.",
262  snapshot.getX(), snapshot.getZ());
263  return;
264  }
265 
266  type = EnumUtils.getEnum(typeName, Material.class);
267  if (type == null) {
268  handleException(null, "Failed to read from file for chunk ({0}, {1}). " +
269  "Found a block type in file that is not a valid type: {2}",
270  snapshot.getX(), snapshot.getZ(), typeName);
271  return;
272  }
273 
274  data = reader.getShort();
275 
276  int ls = reader.getByte();
277 
278  light = ls >> 4;
279  skylight = (ls & 0x0F);
280  } catch (IOException e) {
281  handleException(e, "Failed to read from file for chunk ({0}, {1}).",
282  snapshot.getX(), snapshot.getZ());
283  return;
284  }
285 
286  if (loadType == LoadType.ALL_BLOCKS || !isBlockMatch(x, y, z, type, data)) {
287  //_blockInfo.add(new ChunkBlockInfo(x, y, z, type, data, light, skylight));
288  this.builder.addBlock(
289  (snapshot.getX() * 16) + x, y, (snapshot.getZ() * 16) + z,
290  type, data, light, skylight);
291  }
292  }

299  protected void onPreComplete() {

301  try {
302  int totalEntities = reader.getInteger();

305

317  snapshot.getX(), snapshot.getZ());

319  }
320 
321  // Read entities

323  int totalEntities = reader.getInteger();
324

339  }
340

362  /*
363  * Determine if the block at the specified coordinates of the chunk snapshot
364  * matches the specified material type and data.
365  */
366  private boolean isBlockMatch(int x, int y, int z, Material type, int data) {
367  Material chunkType = Material.getMaterial(snapshot.getBlockTypeId(x, y, z));
368  int chunkData = snapshot.getBlockData(x, y, z);
369 
370  return chunkType == type && chunkData == data;
371  }
