133  public static int getLevel(double exp) {
134  PreCon.positiveNumber(exp);
135 
136  if (Double.compare(exp, 0.0D) == 0)

195  public static double getBarCapacity(int level) {
196  PreCon.positiveNumber(level);
197 
198  // algorithm found at NMS: EntityHuman.getExpToLevel()
199  if (level >= 30) {
200  return 112 + (level - 30) * 9;
201  }
202  else if (level >= 15) {
203  return 37 + (level - 15) * 5;
204  }
205  else {

210  private static int calculateExpInLevel(int level) {
211  int current = 0;

213 
214  while (current < level) {

222  private static int calculateLevel(int exp) {
223  int level = 0;
