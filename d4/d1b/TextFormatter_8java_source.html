192  private ITextFormatterResult format(ParseContext context, boolean isFormattingArgs) {
193 
194  if (context.template instanceof ITextFormatterResult
195  && context.params.length == 0) {
196  return (ITextFormatterResult) context.template;
197  }
198 
199  if (!context.shouldFormat()) {
200 
201  if (isFormattingArgs) {
202  context.buffer.append(context.template);
203  }
204  else {
205  context.result.append(new SimpleChatComponent(context.template));
206  context.result.finishResult(context.settings);
207  }
208 
209  return context.result;
210  }
211 
212  FormatResultBuffer buffer = context.buffer;
213  StringBuilder tagBuffer = context.tagBuffer;
214  TextFormatterSettings settings = context.settings;
215  TextFormatterResult result = context.result;
216  TextParser parser = context.parser;
217 
218  while (!parser.isFinished()) {
219  char ch = parser.next();
220  if (ch == 0)
221  break;
222 
223  // handle format codes
224  if (ch == TextFormat.CHAR
225  && TextFormat.isFormatChar(parser.peek(1))) {
226 
227  appendFormatCode(context);
228 
229  parser.skip(1);
230  continue;
231  }
232 
233  // check for tag opening
234  if (ch == '{') {
235 
236  // parse tag
237  String tag = parseTag(context);
238 
239  // update index position
240  parser.skip(tagBuffer.length());
241 
242  // template ended before tag was closed
243  if (tag == null) {
244  buffer.append('{');
245  buffer.append(tagBuffer);
246  }
247  // tag parsed
248  else {
249  parser.skip(1); // add 1 for closing brace
250  appendReplacement(context, tag);
251  }
252 
253  }
254  else if (ch == '\n' || ch == '\r') {
255  if (settings.getLineReturnPolicy() != FormatPolicy.REMOVE) {
256  buffer.newLine();
257  }
258  }
259  else if (ch == '\\' && parser.peek(1) != 0) {
260  processBackslash(context);
261  }
262  else {
263 
264  if (settings.isEscaped(ch))
265  buffer.append('\\');
266 
267  // append next character
268  buffer.append(ch);
269  }
270  }
271 
272  if (buffer.isModified()) {
273  buffer.reset();
274  }
275 
276  result.appendAll(buffer.results);
277  if (!isFormattingArgs) {
278  result.finishResult(settings);
279  }
280  return result;
281  }

303  private char parseUnicode(ParseContext context) {
304 
305  StringBuilder tagBuffer = context.tagBuffer;
306  TextParser parser = context.parser;
307 
308  tagBuffer.setLength(0);
309 
310  int readCount = 0;
311  while (parser.current() != 0) {
312  if (readCount == 4) {
313  break;
314  }
315  else {
316  char ch = parser.peek(readCount + 1);
317  if ("01234567890abcdefABCDEF".indexOf(ch) == -1)
318  return 0;
319  tagBuffer.append(ch);
320  }
321  readCount++;
322  }
323 
324  if (tagBuffer.length() == 4) {
325 
326  try {
327  return (char) Integer.parseInt(tagBuffer.toString(), 16);
328  } catch (NumberFormatException ignore) {
329  return 0;
330  }
331  }
332 
333  return 0;
334  }

339  private String parseTag(ParseContext context) {
340 
341  StringBuilder tagBuffer = context.tagBuffer;
342  TextParser parser = context.parser;
343 
344  tagBuffer.setLength(0);
345 
346  int i = 1;
347  while (parser.current() != 0) {
348 
349  char ch = context.parser.peek(i);
350  if (ch == 0)
351  return null;
352 
353  if (ch == '}') {
354  return tagBuffer.toString();
355  }
356  else {
357  tagBuffer.append(ch);
358  }
359  i++;
360  }
361 
362  return null;
363  }

368  private void appendReplacement(ParseContext context, String tag) {
369 
370  TextFormatterSettings settings = context.settings;
371  TextFormatterResult result = context.result;
372  FormatResultBuffer buffer = context.buffer;
373  StringBuilder tagBuffer = context.tagBuffer;
374  Object[] params = context.params;
375  Map<String, ITagFormatter> formatters = context.formatters;
376 
377  boolean isNumber = !tag.isEmpty();
378 
379  tagBuffer.setLength(0);
380 
381  // parse out tag from comment section
382  for (int i=0; i < tag.length(); i++) {
383 
384  char ch = tag.charAt(i);
385 
386  // done at comment character
387  if (ch == ':') {
388  break;
389  }
390  // append next tag character
391  else {
392  tagBuffer.append(ch);
393 
394  // check if the character is a number
395  if (isNumber && !Character.isDigit(ch)) {
396  isNumber = false;
397  }
398  }
399  }
400 
401  String parsedTag = tagBuffer.toString();
402 
403  if (isNumber) {
404  int index = Integer.parseInt(parsedTag);
405 
406  // make sure number is in the range of the provided parameters.
407  if (params.length <= index) {
408  reappendTag(buffer, tag);
409  }
410  // replace number with parameter argument.
411  else {
412 
413  Object param = params[index];
414 
415  if (param instanceof IDynamicText) {
416  param = ((IDynamicText) param).nextText();
417  }
418  else if (param instanceof IFormatterArg) {
419  IChatModifier modifier = new SimpleChatModifier(buffer.getModifier());
420  if (buffer.isModified()) {
421  buffer.reset();
422  }
423  ((IFormatterArg) param).getComponents(buffer.results);
424  buffer.reset(modifier);
425  return;
426  }
427  else if (param instanceof IChatComponent) {
428  IChatModifier modifier = new SimpleChatModifier(buffer.getModifier());
429  if (buffer.isModified()) {
430  buffer.reset();
431  }
432  buffer.results.add((IChatComponent)param);
433  buffer.reset(modifier);
434  return;
435  }
436  else if (param instanceof IChatMessage) {
437  IChatModifier modifier = new SimpleChatModifier(buffer.getModifier());
438  if (buffer.isModified()) {
439  buffer.reset();
440  }
441  ((IChatMessage) param).getComponents(buffer.results);
442  buffer.reset(modifier);
443  return;
444  }
445 
446  String toAppend = String.valueOf(param);
447 
448  // append parameter argument
449  if (settings.isArgsFormatted()) {
450  IChatModifier modifier = new SimpleChatModifier(buffer.getModifier());
451  ITextFormatterResult argResult = format(new ParseContext(context, toAppend), true);
452  //result.appendAll(argResult);
453 
454  if ((argResult.isParsed() && argResult.isColorParsed())
455  || (!argResult.isParsed() && toAppend.indexOf(TextFormat.CHAR) != -1)) {
456 
457  // make sure colors from inserted text do not continue
458  // into template text
459  buffer.reset(modifier);
460  }
461  }
462  else {
463  boolean hasColorCode = toAppend.indexOf(TextFormat.CHAR) != -1;
464  IChatModifier modifier = hasColorCode
465  ? new SimpleChatModifier(buffer.getModifier())
466  : null;
467 
468  buffer.append(toAppend);
469 
470  if (hasColorCode) {
471  // make sure colors from inserted text do not continue
472  // into template text
473  buffer.reset(modifier);
474  }
475  }
476  }
477  }
478  else {
479 
480  // check for custom formatter
481  ITagFormatter formatter = settings.getTagPolicy() == FormatPolicy.IGNORE
482  ? null
483  : getFormatter(parsedTag, formatters);
484 
485  if (formatter == null && settings.getColorPolicy() != FormatPolicy.IGNORE) {
486 
487  // check for color formatter
488  formatter = getFormatter(parsedTag, _colors);
489 
490  if (formatter != null) {
491 
492  // remove color tag if color policy is remove
493  if (settings.getColorPolicy() == FormatPolicy.REMOVE) {
494  formatter = ERASER;
495  }
496  else {
497  result.setParsedColor(true);
498  }
499  }
500  }
501  // remove tag if tag policy is remove
502  else if (formatter != null && settings.getTagPolicy() == FormatPolicy.REMOVE) {
503  formatter = ERASER;
504  }
505 
506  if (formatter != null) {
507 
508  if (shouldResetAfterTag(formatter.getTag()))
509  buffer.reset();
510 
511  // formatter appends replacement text to format buffer
512  formatter.append(buffer, tag);
513  }
514  else {
515  // no formatter, append tag to result buffer
516  reappendTag(buffer, tag);
517  }
518  }
519  }

529  private void processBackslash(ParseContext context) {
530 
531  TextFormatterSettings settings = context.settings;
532  FormatResultBuffer buffer = context.buffer;
533  TextParser parser = context.parser;
534 
535  // make sure the backslash isn't escaped
536  int s = 0;
537  int bsCount = 0;
538  while (parser.current() != 0) {
539  if (parser.peek(s - 1) == '\\') {
540  bsCount++;
541  }
542  else {
543  break;
544  }
545  s--;
546  }
547  if (bsCount % 2 != 0)
548  return;
549 
550  // look at next character
551  char next = parser.peek(1);
552 
553  // handle new line character
554  if ((next == 'n' || next == 'r') && settings.getLineReturnPolicy() != FormatPolicy.IGNORE) {
555 
556  if (settings.getLineReturnPolicy() != FormatPolicy.REMOVE) {
557  buffer.newLine();
558  }
559 
560  parser.skip(1);
561  }
562  // handle unicode
563  else if (next == 'u' && settings.getUnicodePolicy() != FormatPolicy.IGNORE) {
564 
565  parser.skip(1);
566  char unicode = parseUnicode(context);
567  if (unicode == 0) {
568  // append non unicode text
569  buffer.append("\\u");
570  buffer.incrementCharCount(2);
571  }
572  else {
573  if (settings.getUnicodePolicy() != FormatPolicy.REMOVE) {
574  buffer.append(unicode);
575  buffer.incrementCharCount(1);
576  }
577 
578  parser.skip(4);
579  }
580 
581  }
582  // unused backslash
583  else {
584  buffer.append('\\');
585  }
586  }
