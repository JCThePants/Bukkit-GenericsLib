162  @Override
163  protected void onRun() {
164 
165  Deque<BlockInfo> multiBlocks = new ArrayDeque<>(blockQueue.size());
166 
167  while (!blockQueue.isEmpty()) {
168  BlockInfo info = blockQueue.remove();
169 
170  // skip multi-blocks and restore afterwards
171  if (Materials.isMultiBlock(info.material)) {
172  multiBlocks.add(info);
173  continue;
174  }
175 
176  restoreBlock(info);
177  }
178 
179  // Restore block Pairs
180  // keyed to block x, y z value as a string
181  Multimap<String, BlockInfo> placedMultiBlocks =
182  MultimapBuilder.hashKeys(multiBlocks.size()).hashSetValues(3).build();
183 
184  // Get block pairs
185  while (!multiBlocks.isEmpty()) {
186  BlockInfo info = multiBlocks.remove();
187 
188  int x = info.x;
189  int y = info.y;
190  int z = info.z;
191 
192  String lowerKey = getKey(x, y - 1, z);
193  Collection<BlockInfo> lowerBlock = placedMultiBlocks.get(lowerKey);
194 
195  if (lowerBlock != null) {
196  placedMultiBlocks.put(lowerKey, info);
197  continue;
198  }
199 
200  String upperKey = getKey(x, y + 1, z);
201  Collection<BlockInfo> upperBlock = placedMultiBlocks.get(upperKey);
202  if (upperBlock != null) {
203  placedMultiBlocks.put(upperKey, info);
204  continue;
205  }
206 
207  placedMultiBlocks.put(getKey(x, y, z), info);
208  }
209 
210  // Restore pairs
211  Set<String> keys = placedMultiBlocks.keySet();
212 
213  for (String key : keys) {
214 
215  Collection<BlockInfo> multiBlockSetSet = placedMultiBlocks.get(key);
216  if (multiBlockSetSet == null)
217  continue;
218 
219  List<BlockInfo> multiBlockSet = new ArrayList<>(multiBlockSetSet);
220 
221  Collections.sort(multiBlockSet);
222 
223  for (BlockInfo info : multiBlockSet) {
224  restoreBlock(info);
225  }
226  }
227 
228  // restore serializables
229  while (!serializables.isEmpty()) {
230  IAppliedSerializable meta = serializables.remove();
231  meta.apply();
232  }
233 
234  complete();
235  }
