141  * Attempt to run runnable if dependency plugin is loaded.
142  */
143  private boolean runDepends(String pluginNames, Runnable runnable) {
144 
145  String[] names = TextUtils.PATTERN_COMMA.split(pluginNames);
146 
147  for (String pluginName : names) {
148  Plugin plugin = Bukkit.getPluginManager().getPlugin(pluginName.trim());
149  if (plugin == null)
150  return false;
151 
152  if (!plugin.isEnabled())
153  return false;
154 
155  if (plugin instanceof NucleusPlugin && !((NucleusPlugin) plugin).isLoaded())
156  return false;
157  }
158 
159  try {
160  runnable.run();
161  } catch (Exception e) {
162  e.printStackTrace();
163  }
164 
165  return true;
166  }
