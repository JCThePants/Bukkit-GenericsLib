119  public static String getFull(int potionId) {
120 
121  if (potionId <= 64) {
122 
123  switch (potionId) {
124  case 0:
125  return NucLang.get(_WATER).toString();
126  case 16:
127  return NucLang.get(_AWKWARD).toString();
128  case 32:
129  return NucLang.get(_THICK).toString();
130  case 64:
131  return NucLang.get(_MUNDANE).toString();
132  default:
133  throw new IllegalArgumentException(
134  "Failed to get PotionType for Potion name Id: " + potionId);
135  }
136  }
137 
138  PotionType type = PotionType.getByDamageValue(potionId & 15);
139 
140  StringBuilder buffer = buffer();
141 
142  if ((potionId & 16384) == 16384) {
143  append(buffer, _SPLASH);
144  buffer.append(' ');
145  }
146 
147  appendSimple(buffer, type);
148 
149  if ((potionId & 32) == 32) {
150 
151  // see if a level one potion is valid, if not, consider
152  // this the level 1 potion
153  if (PotionUtils.isValidPotionId(potionId & ~32)) {
154  buffer.append(' ');
155  append(buffer, _LEVEL2);
156  }
157  }
158 
159  if ((potionId & 64) == 64) {
160  buffer.append(' ');
161  append(buffer, _EXTENDED_DURATION);
162  }
163 
164  return buffer.toString();
165  }
