166  /*
167  * Serialize an item stack to a string and append to buffer.
168  */
169  private void appendItemStackString(StringBuilder buffy, @Nullable ItemStack stack,
170  SerializerOutputType outputType) {
171 
172  if (stack == null) {
173  stack = new ItemStack(Material.AIR, -1);
174  }
175 
176  // material name
177  if (outputType == SerializerOutputType.COLOR)
178  buffy.append(ChatColor.GREEN);
179 
180  buffy.append(stack.getType().name());
181 
182  // material data
183  short data = stack.getData().getData();
184  if (data != 0) {
185  if (outputType == SerializerOutputType.COLOR)
186  buffy.append(ChatColor.DARK_GRAY);
187  buffy.append(':');
188 
189  if (outputType == SerializerOutputType.COLOR)
190  buffy.append(ChatColor.YELLOW);
191  buffy.append(data);
192 
193  if (outputType == SerializerOutputType.COLOR)
194  buffy.append(ChatColor.GRAY);
195  }
196  else if (outputType == SerializerOutputType.COLOR) {
197  buffy.append(ChatColor.GRAY);
198  }
199 
200  // quantity
201 
202  if (stack.getAmount() != 1) {
203  buffy.append(';');
204 
205  if (outputType == SerializerOutputType.COLOR)
206  buffy.append(ChatColor.AQUA);
207 
208  buffy.append(stack.getAmount());
209  }
210 
211  Collection<IItemMetaHandler> handlers = _metaHandlers.getHandlers();
212 
213  List<ItemMetaValue> metaObjects = new ArrayList<>(10);
214 
215  for (IItemMetaHandler handler : handlers) {
216  metaObjects.addAll(handler.getMeta(stack));
217  }
218 
219  if (!metaObjects.isEmpty()) {
220  buffy.append('{');
221 
222  for (int i=0, last = metaObjects.size() - 1; i < metaObjects.size(); i++) {
223 
224  ItemMetaValue meta = metaObjects.get(i);
225 
226  buffy.append(meta.getName());
227 
228  buffy.append(':');
229  buffy.append('"');
230 
231  Matcher matcher = TextUtils.PATTERN_DOUBLE_QUOTE.matcher(meta.getRawData());
232  buffy.append(matcher.replaceAll("\\\""));
233 
234  buffy.append('"');
235 
236  if (i < last) {
237  buffy.append(',');
238  }
239  }
240 
241  buffy.append('}');
242  }
243  }
244 }
