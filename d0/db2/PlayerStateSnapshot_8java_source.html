226  @Override
227  public void deserialize(IDataNode dataNode) throws DeserializeException {
228 
229  if ((_playerId = dataNode.getUUID("player-id")) == null)
230  throw new DeserializeException("player-id not found.");
231 
232  if ((_items = dataNode.getItemStacks("items")) == null)
233  throw new DeserializeException("items not found.");
234 
235  if ((_armor = dataNode.getItemStacks("armor")) == null)
236  throw new DeserializeException("armor not found.");
237 
238  if ((_location = dataNode.getLocation("location")) == null)
239  throw new DeserializeException("location not found.");
240 
241  if ((_gameMode = dataNode.getEnum("gamemode", GameMode.class)) == null)
242  throw new DeserializeException("gamemode not found.");
243 
244  _health = dataNode.getDouble("health", 20);
245  _food = dataNode.getInteger("food", 20);
246  _level = dataNode.getInteger("levels", 0);
247  _exp = (float)dataNode.getDouble("exp", 0);
248  _flight = dataNode.getBoolean("flight", false);
249  _allowFlight = dataNode.getBoolean("allow-flight", false);
250  _fireTicks = dataNode.getInteger("fire-ticks", 0);
251  _fallDistance = (float)dataNode.getDouble("fall-distance", 0);
252 
253  try {
254 
255  IDataNode potionsNode = dataNode.getNode("potions");
256 
257  _potions = new ArrayList<>(potionsNode.size());
258 
259  for (IDataNode potionNode : potionsNode) {
260 
261  PotionEffect effect = getEffectFromNode(potionNode);
262  if (effect == null)
263  continue;
264 
265  _potions.add(effect);
266  }
267  }
268  catch (Exception ex) {
269  ex.printStackTrace();
270  throw new DeserializeException("Exception while reading potions.");
271  }
272  }

277  @Nullable
278  private static PotionEffect getEffectFromNode(IDataNode dataNode) {
279  PreCon.notNull(dataNode);
280 
281  int amplifier = dataNode.getInteger("amplifier");
282  int duration = dataNode.getInteger("duration");
283  boolean isAmbient = dataNode.getBoolean("ambient");
284 
285  String typeName = dataNode.getString("effect-name");
286  if (typeName == null)
287  return null;
288 
289  PotionEffectType type = PotionEffectType.getByName(typeName);
290  if (type == null)
291  return null;
292 
293  return new PotionEffect(type, duration, amplifier, isAmbient);
294  }
