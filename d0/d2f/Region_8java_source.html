290  @Override

293  synchronized (getSync()) {

295  if (getWorld() == null)
296  return false;
297 
298  int xlen = getXEnd();
299  int ylen = getYEnd();
300  int zlen = getZEnd();
301 
302  for (int x = getXStart(); x <= xlen; x++) {
303 
304  for (int y = getYStart(); y <= ylen; y++) {
305 
306  for (int z = getZStart(); z <= zlen; z++) {
307 
308  Block block = getWorld().getBlockAt(x, y, z);
309  if (block.getType() != material)
310  continue;
311 
312  return true;
313  }
314  }
315  }
316 
317  return false;

373  @Override
374  public final void removeEntities(Class<?>... entityTypes) {
375 
376  synchronized (getSync()) {
377  Collection<IChunkCoords> chunks = getChunkCoords();
378  for (IChunkCoords chunkInfo : chunks) {
379 
380  Chunk chunk = chunkInfo.getChunk();
381  if (chunk == null)
382  return;
383 
384  for (Entity entity : chunk.getEntities()) {
385 
386  if (contains(entity.getLocation())) {
387 
388  if (entityTypes == null || entityTypes.length == 0) {
389  entity.remove();
390  continue;
391  }
392 
393  for (Class<?> itemType : entityTypes) {
394 
395  if (!itemType.isInstance(entity))
396  continue;
397 
398  entity.remove();
399  break;
400  }
401  }
402  }
