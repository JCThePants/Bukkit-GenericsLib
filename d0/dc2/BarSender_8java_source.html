328  @Override
329  public void run() {
330 
331  List<BarDistributor> distributors = new ArrayList<>(10);
332 
333  synchronized (_sync){
334  if (_playerMap.isEmpty())
335  return;
336 
337  // copy distributors to prevent concurrent modification errors.
338  distributors.addAll(_playerMap.values());
339  }
340 
341  long now = System.currentTimeMillis();
342 
343  for (BarDistributor distributor : distributors) {
344 
345  PlayerBar playerBar;
346 
347  synchronized (distributor.sync) {
348  // get the current action bar
349  playerBar = distributor.current();
350  if (playerBar == null)
351  continue;
352 
353  while (!distributor.isHighestPriority(playerBar.priority())) {
354  playerBar = distributor.next();
355  if (playerBar == null)
356  throw new AssertionError("Null player bar in bar distributor.");
357  }
358  }
359 
360  // remove expired action bars
361  if (playerBar.expires() > 0 && playerBar.expires() <= now) {
362  removeBar(playerBar);
363  NucMsg.debug("Removing Bar");
364  continue;
365  }
366 
367  // send action bar packet if time to update
368  if (playerBar.nextUpdate() == 0 ||
369  playerBar.nextUpdate() <= System.currentTimeMillis()) {
370 
371  playerBar.send();
372  }
373  }
374 
375  distributors.clear();
376  }

406  }

410  if (interval <= 0) {
411  interval = MAX_REFRESH_RATE;
412  }
413  else {
414  interval = Math.min(interval * 50, MAX_REFRESH_RATE);
415  interval = Math.max(interval, MIN_REFRESH_RATE);
416  }
