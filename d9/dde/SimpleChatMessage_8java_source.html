218  private static void generateLines(Collection<? extends IChatComponent> components, List<IChatLine> output) {
219 
220  IChatLine current = new SimpleChatLine();
221  TextColor color = null;
222 
223  for (IChatComponent component : components) {
224 
225  if (component instanceof NewLineComponent) {
226  output.add(current);
227  current = new SimpleChatLine();
228  continue;
229  }
230 
231  boolean hasColor = component.getModifier().getColor() != null;
232 
233  if (component.getModifier().isReset()) {
234  color = null;
235  }
236  else {
237  // continue previous color if no new color has been set
238  if (color != null && !hasColor && !component.getText().isEmpty()) {
239  component.getModifier().setColor(color);
240  }
241  }
242 
243  current.append(component);
244 
245  if (hasColor)
246  color = component.getModifier().getColor();
247  }
248 
249  output.add(current);
250  }

264  private static void fixLines(IChatLineSettings settings, List<IChatLine> lines) {
265  int maxLen = settings.getMaxLineLen();
266  if (maxLen <= 0)
267  return;
268 
269  double prefixLen = 0;
270  IChatMessage prefix = settings.getLinePrefix();
271  if (prefix != null) {
272  prefixLen = getCharWidth(prefix.getText());
273  }
274 
275  List<IChatComponent> lineComponents = new ArrayList<>(5);
276  StringBuilder buffer = new StringBuilder(20);
277 
278  ListIterator<IChatLine> lineIterator = lines.listIterator();
279  while (lineIterator.hasNext()) {
280 
281  IChatLine line = lineIterator.next();
282  line.getComponents(lineComponents);
283 
284  double len = prefixLen;
285 
286  ListIterator<IChatComponent> iterator = lineComponents.listIterator();
287  while (iterator.hasNext()) {
288  IChatComponent component = iterator.next();
289  component.getText(buffer);
290 
291  String lineText = buffer.toString();
292  buffer.setLength(0);
293  String[] words = TextUtils.PATTERN_SPACE.split(lineText);
294 
295  // iterate words and measure character length to determine if the line
296  // needs to be broken into a new line.
297  for (int i=0; i < words.length; i++) {
298  len += getCharWidth(words[i]);
299  len += 1; // add space as part of count
300 
301  if (len < maxLen || buffer.length() == 0) {
302  buffer.append(words[i]);
303  if (i < words.length - 1) {
304  buffer.append(' ');
305  }
306  }
307  else {
308 
309  // replace component with shortened version.
310  iterator.remove();
311  iterator.add(new SimpleChatComponent(buffer.toString(), component.getModifier()));
312 
313  // create new line.
314  SimpleChatLine newLine = new SimpleChatLine();
315 
316  newLine.append(new SimpleChatComponent(TextUtils.concat(i, words.length, words, " "),
317  component.getModifier()));
318 
319  // transfer excess components to new line.
320  while (iterator.hasNext()) {
321  IChatComponent transferred = iterator.next();
322  iterator.remove();
323  newLine.append(transferred);
324  }
325 
326  lineIterator.remove(); // remove old line
327  lineIterator.add(new SimpleChatLine(lineComponents)); // add modified line
328  lineIterator.add(newLine); // add new line
329  lineIterator.previous();
330  break;
331  }
332  }
333 
334  buffer.setLength(0);
335  }
336 
337  lineComponents.clear();
338  }

341  private static double getCharWidth(String text) {
342 
343  double width = 0;
344 
345  for (int i=0; i < text.length(); i++) {
346  char ch = text.charAt(i);
347 
348  switch (ch) {
349  case '.':
350  case ',':
351  case ':':
352  case 'i':
353  case 'l':
354  width += 0.5D;
355  break;
356  case 'm':
357  case '-':
358  width += 1.5D;
359  break;
360  default:
361  width += Character.isUpperCase(ch) ? 1.5D : 1D;
362  }
363  }
364  return width;
365  }
