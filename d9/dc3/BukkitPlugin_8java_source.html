267  private void loadScriptManager() {
268 
269  // allow script engines to find nucleus classes
270  Thread.currentThread().setContextClassLoader(getClassLoader());
271 
272  File scriptFolder = new File(getDataFolder(), "scripts");
273  if (!scriptFolder.exists() && !scriptFolder.mkdirs()) {
274  throw new RuntimeException("Failed to create script folder.");
275  }
276 
277  _scriptManager = new InternalScriptManager(scriptFolder);
278  _scriptManager.reload();

284  public void notifyProvidersReady() {
285 
286  if (_isModulesReady)
287  return;
288 
289  _isModulesReady = true;
290 
291  onEnablePlugin();
292 
293  // enable Nucleus plugins
294  for (NucleusPlugin plugin : NucleusPlugin._enabled) {
295  if (plugin instanceof BukkitPlugin)
296  continue;
297 
298  // skip plugins that were disabled before this point.
299  if (!plugin.isEnabled())
300  continue;
301 
302  try {
303  plugin._isEnabled = true;
304  plugin.onEnablePlugin();
305  }
306  catch (Throwable e) {
307  e.printStackTrace();
308  plugin._isEnabled = false;
309  Bukkit.getPluginManager().disablePlugin(plugin);
310  }
311  }
312 
313  NucleusLoadedEvent event = new NucleusLoadedEvent();
314  Bukkit.getPluginManager().callEvent(event);
315  }

317  /*
318  * Init class loader into global factory.
319  * Reflection used to prevent compile issues with "internal" package.
320  *
321  * Equivalent code:
322  *
323  * ContextFactory factory = ContextFactory.getGlobal();
324  * ClassLoader loader = Nucleus.class.getClassLoader();
325  *
326  * if (loader == null) {
327  * factory.initApplicationClassLoader(loader);
328  * }
329  */
330  private void initRhinoClassLoader() throws ClassNotFoundException,
331  NoSuchMethodException, InvocationTargetException, IllegalAccessException {
332 
333  Class<?> contextFactoryClass = Class.forName("sun.org.mozilla.javascript.internal.ContextFactory");
334 
335  Method getGlobal = contextFactoryClass.getDeclaredMethod("getGlobal");
336  getGlobal.setAccessible(true);
337 
338  ClassLoader classLoader = Nucleus.class.getClassLoader();
339 
340  Object contextFactory = getGlobal.invoke(null);
341 
342  Method getApplicationClassLoader = contextFactoryClass.getDeclaredMethod(
343  "getApplicationClassLoader");
344  getApplicationClassLoader.setAccessible(true);
345 
346  ClassLoader current = (ClassLoader) getApplicationClassLoader.invoke(contextFactory);
347 
348  if (current == null) {
349 
350  Method initApplicationClassLoader = contextFactoryClass.getDeclaredMethod(
351  "initApplicationClassLoader", ClassLoader.class);
352  initApplicationClassLoader.setAccessible(true);
353 
354  initApplicationClassLoader.invoke(contextFactory, classLoader);
355  }
356  }
